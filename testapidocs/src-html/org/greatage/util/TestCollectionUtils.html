105            @DataProvider
106            public Object[][] mapFactoryData() {
107                    final Map<String, Integer> testMap = new HashMap<String, Integer>();
108                    testMap.put("1", 1);
109                    testMap.put("2", 2);
110                    testMap.put("3", 3);
111    
112                    return new Object[][]{
113                                    {CollectionUtils.newMap(), HashMap.class, Collections.emptyMap()},
114                                    {CollectionUtils.newMap("1", 1, "2", 2, "3", 3), HashMap.class, testMap},
115                                    {CollectionUtils.newMap(testMap), HashMap.class, testMap},
116                                    {CollectionUtils.newConcurrentMap(), ConcurrentHashMap.class, Collections.emptyMap()},
117                                    {CollectionUtils.newConcurrentMap("1", 1, "2", 2, "3", 3), ConcurrentHashMap.class, testMap},
118                                    {CollectionUtils.newConcurrentMap(testMap), ConcurrentHashMap.class, testMap},
119                    };
120            }
121    
122            @DataProvider
123            public Object[][] mapFactoryWrongData() {
124                    return new Object[][]{
125                                    {null},
126                                    {new Object[]{1, 2, 3}},
127                    };
128            }
129    
130            @Test(dataProvider = "isEmptyData")
131            public void testIsEmpty(final Collection collection, final boolean expected) {
132                    final boolean actual = CollectionUtils.isEmpty(collection);
133                    assertEquals(actual, expected);
134            }
135    
136            @Test(dataProvider = "collectionFactoryData")
137            public void testCollectionFactory(final Collection collection,
138                                                                              final Class<? extends Collection> collectionClass,
139                                                                              final Object[] expected) {
140                    assertNotNull(collection);
141                    assertTrue(collectionClass.isInstance(collection));
142                    assertEquals(collection.size(), expected.length);
143                    assertTrue(collection.containsAll(Arrays.asList(expected)));
144            }
145    
146            @Test(dataProvider = "mapFactoryData")
147            public void testMapFactory(final Map map,
148                                                               final Class<? extends Map> mapClass,
149                                                               final Map expected) {
150                    assertNotNull(map);
151                    assertTrue(mapClass.isInstance(map));
152                    assertEquals(map, expected);
153            }
154    
155            @Test(dataProvider = "mapFactoryWrongData", expectedExceptions = IllegalArgumentException.class)
156            public void testMapFactoryWrong(final Object[] values) {
157                    CollectionUtils.newMap(values);
158            }
