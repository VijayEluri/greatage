001    package org.greatage.util;
002    
003    import org.testng.Assert;
004    import org.testng.annotations.Test;
005    
006    /**
007     * @author Ivan Khalopik
008     * @since 1.0
009     */
010    public class TestLocker extends Assert {
011    
012            @Test
013            public void testLocker() {
014                    final Locker locker = new Locker();
015                    locker.check();
016                    locker.lock();
017            }
018    
019            @Test
020            public void testLocker1() {
021                    final Locker locker = new Locker();
022                    locker.check();
023                    locker.check();
024            }
025    
026            @Test(expectedExceptions = IllegalStateException.class)
027            public void testLocker2() {
028                    final Locker locker = new Locker();
029                    locker.lock();
030                    locker.lock();
031            }
032    
033            @Test(expectedExceptions = IllegalStateException.class)
034            public void testLocker3() {
035                    final Locker locker = new Locker();
036                    locker.lock();
037                    locker.check();
038            }
039    
040            @Test
041            public void testLockerToString() {
042                    final Locker locker = new Locker();
043                    locker.check();
044                    assertEquals(locker.toString(), "Locker(locked=false)");
045                    locker.lock();
046                    assertEquals(locker.toString(), "Locker(locked=true)");
047            }
048    
049            @Test
050            public void testMultiLocker() {
051                    final MultiLocker<Class> locker = new MultiLocker<Class>();
052                    locker.check(getClass());
053                    locker.lock(getClass());
054    
055                    locker.check(MultiLocker.class);
056                    locker.lock(MultiLocker.class);
057            }
058    
059            @Test
060            public void testMultiLocker1() {
061                    final MultiLocker<Class> locker = new MultiLocker<Class>();
062                    locker.check(getClass());
063                    locker.check(getClass());
064    
065                    locker.check(MultiLocker.class);
066                    locker.check(MultiLocker.class);
067            }
068    
069            @Test(expectedExceptions = IllegalStateException.class)
070            public void testMultiLocker2() {
071                    final MultiLocker<Class> locker = new MultiLocker<Class>();
072                    locker.check(getClass());
073                    locker.lock(getClass());
074                    locker.lock(MultiLocker.class);
075                    locker.lock(getClass());
076            }
077    
078            @Test(expectedExceptions = IllegalStateException.class)
079            public void testMultiLocker3() {
080                    final MultiLocker<Class> locker = new MultiLocker<Class>();
081                    locker.lock(getClass());
082                    locker.lock(MultiLocker.class);
083                    locker.check(getClass());
084            }
085    
086            @Test
087            public void testMultiLockerToString() {
088                    final MultiLocker<Class> locker = new MultiLocker<Class>();
089                    locker.check(getClass());
090                    assertEquals(locker.toString(), "MultiLocker(lockers={})");
091                    locker.lock(getClass());
092                    assertEquals(locker.toString(), "MultiLocker(lockers={class org.greatage.util.TestLocker=Locker(locked=true)})");
093            }
094    }
