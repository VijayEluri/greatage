041                    return new Object[][]{
042                                    {new byte[]{}, ""},

050            @Test(dataProvider = "encodeBase64Data")
051            public void testEncodeBase64(final byte[] data, final String expected) {
052                    final String actual = new String(EncodeUtils.encodeBase64(data));
053                    assertEquals(actual, expected);
054            }
055    
056            @Test(dataProvider = "encodeData")
057            public void testEncode(final String data, final String algorithm, final String expected) {
058                    final String actual = StringUtils.toHexString(EncodeUtils.encode(data.getBytes(), algorithm));
059                    assertEquals(actual, expected);
060            }
