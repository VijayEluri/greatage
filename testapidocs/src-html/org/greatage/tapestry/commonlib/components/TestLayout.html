005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.dom.Document;
008    import org.greatage.tapestry.internal.BaseComponentTest;
009    import org.greatage.tapestry.internal.TestConstants;
010    import org.testng.annotations.BeforeClass;
011    import org.testng.annotations.DataProvider;
012    import org.testng.annotations.Test;
013    
014    /**
015     * @author Ivan Khalopik
016     * @version $Revision: 198 $ $Date: 2010-07-09 11:07:50 +0300 (Пт, 09 июл 2010) $
017     */
018    public class TestLayout extends BaseComponentTest {
019            private static final String STYLESHEET_FROMAT = "<link type=\"text/css\" rel=\"stylesheet\" href=\"/foo/assets/test/%s\"></link>";
020            private String document;
021    
022            @BeforeClass(groups = TestConstants.INTEGRATION_GROUP)
023            public void generateDocument() {
024                    final Document doc = generatePage("TestPageForLayout");
025                    document = doc.toString();
026                    assertNotNull(document);
027            }
028    
029            @Test(dataProvider = "layoutAssetData", groups = TestConstants.INTEGRATION_GROUP)
030            public void testLayoutAsset(String expected) {
031                    final String fullExpected = String.format(STYLESHEET_FROMAT, "common/components/" + expected);
032                    assertTrue(document.contains(fullExpected));
033            }
034    
035            @Test(dataProvider = "layoutSkinData", groups = TestConstants.INTEGRATION_GROUP)
036            public void testLayoutSkin(String expected) {
037                    final String fullExpected = String.format(STYLESHEET_FROMAT, "theme/" + expected);
038                    assertTrue(document.contains(fullExpected));
039            }
040    
041            @DataProvider(name = "layoutAssetData")
042            public Object[][] getLayoutAssetData() {
043                    return new Object[][]{
044                                    {"layout.css"},
045                                    {"reset.css"},
046                                    {"base.css"},
047                                    {"fonts.css"},
048                    };
049            }
050    
051            @DataProvider(name = "layoutSkinData")
052            public Object[][] getLayoutSkinData() {
053                    return new Object[][]{
054                                    {"white.css"},
055                    };
056            }
057    }
