005    package org.greatage.tapestry.internal;
006    
007    import org.apache.tapestry5.dom.Document;
008    import org.apache.tapestry5.dom.Element;
009    import org.apache.tapestry5.test.PageTester;
010    import org.greatage.tapestry.services.CommonModule;
011    import org.testng.Assert;
012    import org.testng.annotations.AfterSuite;
013    import org.testng.annotations.BeforeSuite;
014    
015    /**
016     * @author Ivan Khalopik
017     * @version $Revision: 109 $ $Date: 2010-04-30 11:49:41 +0300 (Пт, 30 апр 2010) $
018     */
019    public class BaseComponentTest extends Assert {
020            private static PageTester PAGE_TESTER;
021    
022            @BeforeSuite(groups = TestConstants.INTEGRATION_GROUP)
023            public void setupPageTester() {
024                    PAGE_TESTER = new PageTester("org.greatage.tapestry.testapp", "testApp", PageTester.DEFAULT_CONTEXT_PATH, CommonModule.class);
025                    assertNotNull(PAGE_TESTER);
026            }
027    
028            @AfterSuite(groups = TestConstants.INTEGRATION_GROUP)
029            public void cleanupPageTester() {
030                    PAGE_TESTER = null;
031            }
032    
033            protected PageTester getPageTester() {
034                    return PAGE_TESTER;
035            }
036    
037            protected Document generatePage(String pageName) {
038                    assertNotNull(PAGE_TESTER);
039                    final Document document = PAGE_TESTER.renderPage(pageName);
040                    assertNotNull(document);
041                    return document;
042            }
043    
044            protected String getChildMarkup(Document document, String containerId) {
045                    final Element element = document.getElementById(containerId);
046                    assertNotNull(element);
047                    return element.getChildMarkup();
048            }
049    
050    }
