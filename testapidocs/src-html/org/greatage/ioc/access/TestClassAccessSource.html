001    package org.greatage.ioc.access;
002    
003    import org.testng.Assert;
004    import org.testng.annotations.BeforeClass;
005    import org.testng.annotations.DataProvider;
006    import org.testng.annotations.Test;
007    
008    /**
009     * @author Ivan Khalopik
010     * @since 1.1
011     */
012    public class TestClassAccessSource extends Assert {
013            private ClassAccessSource accessSource;
014    
015            @BeforeClass
016            public void setupAccessSource() {
017                    accessSource = new ClassAccessSourceImpl();
018            }
019    
020            @DataProvider
021            public Object[][] propertyAccessGetData() {
022                    return new Object[][] {
023                                    {MockObject.class, "message", new MockObject("hello"), "hello"},
024                                    {MockObject.class, "type", new MockObject("hello"), 10},
025                    };
026            }
027    
028            @DataProvider
029            public Object[][] propertyAccessSetData() {
030                    return new Object[][] {
031                                    {MockObject.class, "type", new MockObject("hello"), 10},
032                    };
033            }
034    
035            @Test(dataProvider = "propertyAccessGetData")
036            public void testPropertyAccessGet(final Class clazz, final String property, final Object instance,
037                                                                              final Object expected) {
038                    final PropertyAccess access = accessSource.getAccess(clazz).getPropertyAccess(property);
039                    final Object actual = access.get(instance);
040                    assertEquals(actual, expected);
041            }
042    
043            @Test(dataProvider = "propertyAccessSetData")
044            public void testPropertyAccessSet(final Class clazz, final String property, final Object instance,
045                                                                              final Object expected) {
046                    final PropertyAccess access = accessSource.getAccess(clazz).getPropertyAccess(property);
047                    access.set(instance, expected);
048                    final Object actual = access.get(instance);
049                    assertEquals(actual, expected);
050            }
051    }
