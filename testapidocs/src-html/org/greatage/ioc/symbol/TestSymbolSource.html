005    package org.greatage.ioc.symbol;
006    
007    import org.greatage.ioc.symbol.DefaultSymbolProvider;
008    import org.greatage.ioc.symbol.SymbolProvider;
009    import org.greatage.ioc.symbol.SymbolSource;
010    import org.greatage.ioc.symbol.SymbolSourceImpl;
011    import org.greatage.util.CollectionUtils;
012    import org.testng.Assert;
013    import org.testng.annotations.BeforeClass;
014    import org.testng.annotations.DataProvider;
015    import org.testng.annotations.Test;
016    
017    import java.util.List;
018    
019    /**
020     * @author Ivan Khalopik
021     * @since 1.0
022     */
023    public class TestSymbolSource extends Assert {
024            private SymbolSource symbolSource;
025    
026            @BeforeClass
027            public void setupSymbolSource() {
028                    final SymbolProvider symbolProvider = new DefaultSymbolProvider(CollectionUtils.<String, String>newMap(
029                                    "s1", "test1",
030                                    "s2", "test2",
031                                    "s3", "test3"
032                    ));
033                    final List<SymbolProvider> providers = CollectionUtils.newList(symbolProvider);
034                    symbolSource = new SymbolSourceImpl(providers);
035            }
036    
037            @DataProvider
038            public Object[][] symbolSourceData() {
039                    return new Object[][]{
040                                    {"s1", "test1"},
041                                    {"${s1}", "test1"},
042                                    {"H-${s1}", "H-test1"},
043                                    {"H-${s1}-${s2}", "H-test1-test2"},
044                    };
045            }
046    
047            @DataProvider
048            public Object[][] symbolSourceWrongData() {
049                    return new Object[][]{
050                                    {null},
051                                    {"s4"},
052                                    {"${s4}"},
053                                    {"${s1"},
054    //                              {"H-${s1}-${s3"}, todo: add this test
055                                    {"${s1}-${s4}"},
056                    };
057            }
058    
059            @Test(dataProvider = "symbolSourceData")
060            public void testSymbolSource(final String expression, final String expected) {
061                    final String actual = symbolSource.getValue(expression);
062                    assertEquals(actual, expected);
063            }
064    
065            @Test(dataProvider = "symbolSourceWrongData", expectedExceptions = RuntimeException.class)
066            public void testSymbolSourceWrong(final String expression) {
067                    symbolSource.getValue(expression);
068            }
069    }
