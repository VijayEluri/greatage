014     * @author Ivan Khalopik

016     */
017    public class TestTypeCoercer extends Assert {
018            private TypeCoercer typeCoercer;
019    
020            @DataProvider
021            public Object[][] typeCoercerData() {
022                    return new Object[][]{
023                                    {null, String.class, null},
024                                    {null, Boolean.class, null},
025                                    {null, Integer.class, null},
026                                    {null, TimeUnit.class, null},
027                                    {null, TypeCoercer.class, null},
028    
029                                    {true, String.class, "true"},
030                                    {Boolean.TRUE, String.class, "true"},
031                                    {false, String.class, "false"},
032                                    {Boolean.FALSE, String.class, "false"},
033    
034                                    {0, String.class, "0"},
035                                    {-0, String.class, "0"},
036                                    {100045, String.class, "100045"},
037                                    {-145, String.class, "-145"},
038    
039                                    {TimeUnit.SECONDS, String.class, "seconds"},
040                                    {TimeUnit.MILLISECONDS, CharSequence.class, "milliseconds"},
041                                    {ElementType.TYPE, String.class, "type"},
042    
043                                    {"type", ElementType.class, ElementType.TYPE},
044                                    {"seconds", TimeUnit.class, TimeUnit.SECONDS},
045                                    {"SECONDS", TimeUnit.class, TimeUnit.SECONDS},
046                                    {"sEcoNds", TimeUnit.class, TimeUnit.SECONDS},
047    
048                                    {-145, CharSequence.class, "-145"},
049                    };
050            }
051    
052            @DataProvider
053            public Object[][] typeCoercerWrongData() {
054                    return new Object[][]{
055                                    {"1", null},
056                                    {"1", TypeCoercer.class},
057                    };
058            }
059    
060            @BeforeTest
061            public void setupTypeCoercer() {
062                    final Set<Coercion> coercions = CollectionUtils.newSet();
063                    coercions.add(new BooleanToStringCoercion());
064                    coercions.add(new NumberToStringCoercion());
065                    coercions.add(new EnumToStringCoercion());
066                    coercions.add(new StringToBooleanCoercion());
067                    coercions.add(new StringToIntegerCoercion());
068                    coercions.add(new StringToDoubleCoercion());
069                    final CoercionProvider coercionProvider = new DefaultCoercionProvider(coercions);
070    
071                    final Set<CoercionProvider> coercionProviders = CollectionUtils.newSet();
072                    coercionProviders.add(coercionProvider);
073                    coercionProviders.add(new StringToEnumCoercionProvider());
074    
075                    typeCoercer = new TypeCoercerImpl(coercionProviders);
076            }
077    
078            @Test(dataProvider = "typeCoercerData")
079            public <S, T> void testTypeCoercer(final S source, final Class<T> targetClass, final T expected) {
080                    final T actual = typeCoercer.coerce(source, targetClass);
081                    assertEquals(actual, expected);
082            }
083    
084            @Test(dataProvider = "typeCoercerWrongData", expectedExceptions = {AssertionError.class, CoerceException.class})
085            public <S, T> void testTypeCoercer(final S source, final Class<T> targetClass) {
086                    typeCoercer.coerce(source, targetClass);
087            }
088    }
