005    package org.greatage.ioc.mock.modules;
006    
007    import org.greatage.ioc.OrderedConfiguration;
008    import org.greatage.ioc.annotations.Build;
009    import org.greatage.ioc.annotations.Contribute;
010    import org.greatage.ioc.mock.MockMessageService;
011    import org.greatage.ioc.mock.MockMessageServiceImpl;
012    import org.greatage.ioc.mock.MockTalkService;
013    import org.greatage.ioc.mock.MockTalkServiceImpl;
014    
015    import java.util.List;
016    
017    /**
018     * @author Ivan Khalopik
019     * @since 1.0
020     */
021    public class MockConfigureModule {
022    
023            @Build
024            public MockMessageService buildMessageService(final List<String> messages) {
025                    return new MockMessageServiceImpl(messages);
026            }
027    
028            @Build
029            public MockTalkService buildTalkService(final MockMessageService messageService) {
030                    return new MockTalkServiceImpl(messageService);
031            }
032    
033            @Contribute(MockMessageService.class)
034            public void contributeMessageService(final OrderedConfiguration<String> configuration) {
035                    configuration.add("!!!", "end");
036                    configuration.add("hello", "hello");
037                    configuration.add("world", "world", "after:hello", "before:end");
038            }
039    }
