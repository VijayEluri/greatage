005    package org.greatage.ioc.resource;
006    
007    import org.greatage.util.CollectionUtils;
008    import org.greatage.util.I18nUtils;
009    import org.testng.Assert;
010    import org.testng.annotations.BeforeClass;
011    import org.testng.annotations.DataProvider;
012    import org.testng.annotations.Test;
013    
014    import java.util.Locale;
015    import java.util.Map;
016    
017    /**
018     * @author Ivan Khalopik
019     * @since 1.0
020     */
021    public class TestMessagesSource extends Assert {
022            private MessagesSource messagesSource;
023    
024            @BeforeClass
025            public void setupMessagesSource() {
026                    final ClasspathResourceLocator resourceLocator = new ClasspathResourceLocator();
027                    messagesSource = new MessagesSourceImpl(resourceLocator);
028            }
029    
030            @DataProvider
031            public Object[][] getMessagesData() {
032                    return new Object[][]{
033                                    {TestMessagesSource.class, null, CollectionUtils.newMap("test", "Test", "class", "Class")},
034                                    {TestMessagesSource.class, I18nUtils.ROOT_LOCALE, CollectionUtils.newMap("test", "Test", "class", "Class")},
035                                    {TestMessagesSource.class, Locale.ENGLISH, CollectionUtils.newMap("test", "Test", "class", "Class")},
036                                    {TestMessagesSource.class, new Locale("ru"), CollectionUtils.newMap("test", "Тест", "class", "Класс")},
037                                    {TestMessagesSource.class, new Locale("ru", "RU", "xxx"), CollectionUtils.newMap("test", "Тест", "class", "Класс")},
038                    };
039            }
040    
041            @Test(dataProvider = "getMessagesData")
042            public void testGetMessages(Class clazz, Locale locale, Map<String, String> expected) {
043                    final Messages messages = messagesSource.getMessages(clazz, locale);
044                    assertNotNull(messages);
045                    for (Map.Entry<String, String> entry : expected.entrySet()) {
046                            assertEquals(messages.get(entry.getKey()), entry.getValue());
047                    }
048            }
049    
050    }
