005    package org.greatage.javassist;
006    
007    import org.testng.Assert;
008    import org.testng.annotations.BeforeClass;
009    import org.testng.annotations.DataProvider;
010    import org.testng.annotations.Test;
011    
012    /**
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public class TestClassBuilder extends Assert {
017            private ClassPoolEx pool;
018    
019            @BeforeClass
020            public void setupClassPool() {
021                    pool = new ClassPoolEx();
022            }
023    
024            @DataProvider
025            public Object[][] constructorData() {
026                    return new Object[][]{
027                                    {"TestClass_0", false, null},
028                                    {"TestClass_1", false, Object.class},
029                                    {"TestClass_2", false, MockInterfaceImpl2.class},
030                                    {"TestClass_3", false, MockInterface.class},
031    
032                                    {"TestInterface_0", true, null},
033                                    {"TestInterface_1", true, MockInterface.class},
034                    };
035            }
036    
037            @DataProvider
038            public Object[][] constructorWrongData() {
039                    return new Object[][]{
040                                    {"TestInterface_0", true, Object.class},
041                                    {"TestInterface_0", false, Object.class},
042                    };
043            }
044    
045            @Test(dataProvider = "constructorData")
046            public <T> void testConstructor(String className, boolean isInterface, Class<T> superClass) {
047                    final ClassBuilder<T> builder = new ClassBuilder<T>(pool, className, isInterface, superClass);
048                    final Class<T> newClass = builder.build();
049                    assertNotNull(newClass);
050                    assertEquals(newClass.getName(), className);
051                    assertEquals(newClass.isInterface(), isInterface);
052                    if (superClass != null) {
053                            assertTrue(superClass.isAssignableFrom(newClass));
054                    }
055                    if (!isInterface) {
056                            assertEquals(newClass.getSuperclass(),
057                                            superClass == null || superClass.isInterface() ? Object.class : superClass);
058                    }
059            }
060    
061            @Test(dataProvider = "constructorWrongData", expectedExceptions = RuntimeException.class)
062            public <T> void testConstructorWrong(String className, boolean isInterface, Class<T> superClass) {
063                    final ClassBuilder<T> builder = new ClassBuilder<T>(pool, className, isInterface, superClass);
064                    builder.build();
065            }
066    }
