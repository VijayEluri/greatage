121 	private <T, PK extends Serializable, E extends Entity<PK>>
122 	T execute(final EntityFilter<PK, E> filter, final Pagination pagination, final QueryCallback<T> callback) {
123 		return executor.execute(new JdoCallback<T>() {
124 			public T doInJdo(final PersistenceManager pm) throws JDOException {
125 				final JdoCriteria criteria = JdoCriteria.forClass(pm, getImplementation(filter.getEntityClass()));
126 				if (filterProcessor != null) {
127 					filterProcessor.process(criteria, filter, pagination);
128 				}
129 				return callback.doInQuery(criteria.assign());
130 			}
131 		});
132 	}
