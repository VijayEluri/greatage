40  		return execute(filter, Pagination.ALL, new CriteriaCallback<Number>() {

133 	/**
134 	 * Executes inner hibernate selection by filter with specified callback. First creates {@link DetachedCriteria} object
135 	 * according to filter and pagination parameters and then executes it with criteria callback.
136 	 *
137 	 * @param filter	 selection filter
138 	 * @param pagination selection pagination
139 	 * @param callback   criteria callback
140 	 * @param <T>        type of returned value
141 	 * @param <PK>       type of entities primary key
142 	 * @param <E>        type of entities
143 	 * @return criteria execution result
144 	 */
145 	private <T, PK extends Serializable, E extends Entity<PK>>
146 	T execute(final EntityFilter<PK, E> filter, final Pagination pagination, final CriteriaCallback<T> callback) {
147 		final HibernateCriteria entityCriteria = HibernateCriteria.forClass(getImplementation(filter.getEntityClass()));
148 		if (filterProcessor != null) {
149 			filterProcessor.process(entityCriteria, filter, pagination);
150 		}
151 		return executor.execute(new HibernateCallback<T>() {
152 			public T doInSession(final Session session) throws Throwable {
153 				return callback.doInCriteria(entityCriteria.assign(session));
154 			}
155 		});
156 	}

162 		 *
163 		 * @param criteria hibernate criteria holder
164 		 * @return criteria execution result
165 		 */
166 		T doInCriteria(Criteria criteria);
167 
168 	}
