26  public class HibernateModule {
27  
28  	@Bind
29  	public static void bind(final ServiceBinder binder) {
30  		binder.bind(HibernateConfiguration.class, HibernateAnnotationConfiguration.class).withId("HibernateAnnotationConfiguration");
31  		binder.bind(HibernateConfiguration.class, HibernatePropertyConfiguration.class).withId("HibernatePropertyConfiguration");
32  		binder.bind(HibernateExecutor.class, HibernateExecutorImpl.class).withScope(ScopeConstants.THREAD);
33  		binder.bind(EntityFilterProcessor.class, CompositeFilterProcessor.class).withId("HibernateFilterProcessor");
34  	}
35  
36  	@Contribute(value = SymbolProvider.class, serviceId = "ApplicationSymbolProvider")
37  	public void contributeApplicationSymbolProvider(final MappedConfiguration<String, String> configuration) {
38  		configuration.add(DomainConstants.HIBERNATE_PROPERTIES, "hibernate.properties");
39  	}
40  
41  	@Build
42  	public SessionFactory buildSessionFactory(final List<HibernateConfiguration> configurations) {
43  		final Configuration configuration = new Configuration();
44  		for (HibernateConfiguration hibernateConfiguration : configurations) {
45  			hibernateConfiguration.configure(configuration);
46  		}
47  		return configuration.buildSessionFactory();
48  	}
49  
50  	@Contribute(SessionFactory.class)
51  	public void contributeSessionFactory(final OrderedConfiguration<HibernateConfiguration> configuration,
52  										 @Inject("HibernatePropertyConfiguration") final HibernateConfiguration propertyConfiguration,
53  										 @Inject("HibernateAnnotationConfiguration") final HibernateConfiguration annotationConfiguration,
54  										 @Symbol(DomainConstants.HIBERNATE_PROPERTIES) final String hibernatePropertiesResource,
55  										 final ResourceLocator resourceLocator) throws IOException {
56  		configuration.add(new HibernateBaseConfiguration(resourceLocator, hibernatePropertiesResource), "Base");
57  		configuration.add(propertyConfiguration, "Property", "after:Base");
58  		configuration.add(annotationConfiguration, "Annotation", "after:Property");
59  	}
60  
61  	@Build("HibernateRepository")
62  	public EntityRepository buildHibernateRepository(final Map<Class, Class> aliases,
63  													 final HibernateExecutor executor,
64  													 @Inject("HibernateFilterProcessor") final EntityFilterProcessor processor) {
65  		return new HibernateRepository(aliases, executor, processor);
66  	}
67  
68  	@Contribute(value = EntityFilterProcessor.class, serviceId = "HibernateFilterProcessor")
69  	public void contributeHibernateFilterProcessor(final org.greatage.ioc.Configuration<EntityFilterProcessor> configuration) {
70  		configuration.addInstance(BaseFilterProcessor.class);
71  	}
