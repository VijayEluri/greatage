24  public class HibernateCriteria implements EntityCriteria {
25  	private final Criteria criteria;
26  	private final EntityCriteria root;
27  
28  	private final Map<String, EntityProperty> properties = new HashMap<String, EntityProperty>();
29  	private final Map<String, HibernateCriteria> criterias = new HashMap<String, HibernateCriteria>();
30  
31  	private final List<String> names = new ArrayList<String>();
32  
33  	private HibernateCriteria(final Criteria criteria, final EntityCriteria root) {
34  		this.criteria = criteria;
35  		this.root = root;
36  	}
37  
38  	public static HibernateCriteria forClass(final Class<? extends Entity> entityClass) {
39  		final CriteriaImpl criteria = new CriteriaImpl(entityClass.getName(), null);
40  		return new HibernateCriteria(criteria, null);
41  	}
42  
43  	public String getAlias() {
44  		return criteria.getAlias();
45  	}
46  
47  	public EntityCriteria root() {
48  		return root != null ? root : this;
49  	}
50  
51  	public HibernateCriteria getCriteria(final String path) {
52  		if (!criterias.containsKey(path)) {
53  			criterias.put(path, createCriteria(path));
54  		}
55  		return criterias.get(path);
56  	}
57  
58  	public EntityProperty getProperty(final String path) {
59  		if (!properties.containsKey(path)) {
60  			properties.put(path, createProperty(path));
61  		}
62  		return properties.get(path);
63  	}
64  
65  	public void add(final EntityCriterion criterion) {
66  		criteria.add(((HibernateCriterion) criterion).getCriterion());
67  	}
68  
69  	public void setPagination(final Pagination pagination) {
70  		if (pagination.getStart() > 0) {
71  			criteria.setFirstResult(pagination.getStart());
72  		}
73  		if (pagination.getCount() >= 0) {
74  			criteria.setMaxResults(pagination.getCount());
75  		}
76  	}
77  
78  	Criteria assign(final Session session) {
79  		CriteriaImpl criteriaImpl = (CriteriaImpl) criteria;
80  		criteriaImpl.setSession((SessionImplementor) session);
81  		return criteriaImpl;
82  	}
83  
84  	private HibernateCriteria createCriteria(final String path) {
85  		if (StringUtils.isEmpty(path)) {
86  			throw new IllegalArgumentException("Empty path");
87  		}
88  		final int i = path.lastIndexOf('.');
89  		final String property = i > 0 ? path.substring(i + 1) : path;
90  		final Criteria subCriteria = criteria.createCriteria(path, allocateName(property));
91  		return new HibernateCriteria(subCriteria, root());
92  	}
93  
94  	private EntityProperty createProperty(final String path) {
95  		if (StringUtils.isEmpty(path)) {
96  			throw new IllegalArgumentException("Empty path");
97  		}
98  		final int i = path.lastIndexOf('.');
99  		final HibernateCriteria entityCriteria = i > 0 ? getCriteria(path.substring(0, i)) : this;
100 		final String property = i > 0 ? path.substring(i + 1) : path;
101 		return new HibernateProperty(entityCriteria.criteria, property);
102 	}
103 
104 	private String allocateName(final String name) {
105 		String result = name;
106 		int i = 0;
107 		while (names.contains(result)) {
108 			result = name + "_" + i++;
109 		}
110 		names.add(result);
111 		return result;
112 	}

114 	@Override
115 	public String toString() {
116 		final DescriptionBuilder builder = new DescriptionBuilder(getClass());
117 		builder.append(criteria);
118 		return builder.toString();
119 	}
