111 	protected <T, PK extends Serializable, E extends Entity<PK>>
112 	T execute(final String queryString, final EntityFilter<PK, E> filter, final Pagination pagination, final QueryCallback<T> callback) {
113 		return executor.execute(new JpaCallback<T>() {
114 			public T doInJpa(EntityManager em) throws PersistenceException {
115 				final Query query = em.createQuery(queryString.replaceAll("entityClass", filter.getEntityClass().getName()));
116 				if (pagination.getStart() > 0) {
117 					query.setFirstResult(pagination.getStart());
118 				}
119 				if (pagination.getCount() >= 0) {
120 					query.setMaxResults(pagination.getCount());
121 				}
122 				return callback.doInQuery(query);
123 			}
124 		});
125 	}
