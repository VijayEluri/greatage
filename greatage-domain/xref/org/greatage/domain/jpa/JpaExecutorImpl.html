9   import javax.persistence.EntityManager;
10  import javax.persistence.EntityManagerFactory;
11  import javax.persistence.EntityTransaction;

13  /**
14   * @author Ivan Khalopik

17  public class JpaExecutorImpl implements JpaExecutor {
18  	private final EntityManagerFactory entityManagerFactory;
19  
20  	private EntityManager entityManager;
21  
22  	public JpaExecutorImpl(final EntityManagerFactory entityManagerFactory) {
23  		this.entityManagerFactory = entityManagerFactory;
24  	}
25  
26  	public <T> T execute(final JpaCallback<T> callback) {
27  		try {
28  			final EntityManager entityManager = getEntityManager();
29  			return callback.doInJpa(entityManager);
30  		} catch (RuntimeException ex) {
31  			throw ex;
32  		} catch (Throwable throwable) {
33  			throw new RuntimeException(throwable);
34  		}
35  	}
36  
37  	private EntityManager getEntityManager() {
38  		if (entityManager == null) {
39  			entityManager = entityManagerFactory.createEntityManager();
40  		}
41  		return entityManager;
42  	}
43  
44  	public Transaction begin() {
45  		final EntityTransaction transaction = getEntityManager().getTransaction();
46  		transaction.begin();
47  		return new JpaTransaction(transaction);
48  	}
49  }
