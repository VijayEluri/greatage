5   package org.greatage.tapestry.internal;
6   
7   import org.apache.tapestry5.services.*;
8   import org.greatage.security.AnonymousAuthenticationToken;
9   import org.greatage.security.Authentication;
10  import org.greatage.security.AuthenticationManager;
11  import org.greatage.security.SecurityContext;
12  
13  import java.io.IOException;
14  
15  /**
16   * @author Ivan Khalopik
17   * @since 1.0
18   */
19  public class UserPersistenceFilter implements RequestFilter {
20  	private static final String SECURITY_CONTEXT_KEY = "SECURITY_CONTEXT";
21  
22  	private final AuthenticationManager authenticationManager;
23  	private final SecurityContext securityContext;
24  
25  	public UserPersistenceFilter(final AuthenticationManager authenticationManager, final SecurityContext securityContext) {
26  		this.authenticationManager = authenticationManager;
27  		this.securityContext = securityContext;
28  	}
29  
30  	@SuppressWarnings({"unchecked"})
31  	public boolean service(final Request request, final Response response, final RequestHandler handler) throws IOException {
32  		try {
33  			final Authentication authentication = loadAuthentication(request);
34  			if (authentication != null) {
35  				securityContext.initCurrentUser(authentication);
36  			} else {
37  				authenticationManager.signIn(new AnonymousAuthenticationToken());
38  			}
39  			return handler.service(request, response);
40  		} finally {
41  			final Authentication authentication = securityContext.getCurrentUser();
42  			securityContext.clearCurrentUser();
43  			saveAuthentication(request, authentication);
44  		}
45  	}
46  
47  	private Authentication loadAuthentication(final Request request) {
48  		final Session session = getSession(request, true);
49  		final Object user = session != null ? session.getAttribute(SECURITY_CONTEXT_KEY) : null;
50  		return user != null && user instanceof Authentication ? (Authentication) user : null;
51  	}
52  
53  	private void saveAuthentication(final Request request, final Authentication user) {
54  		final Session session = getSession(request, true);
55  		session.setAttribute(SECURITY_CONTEXT_KEY, user);
56  	}
57  
58  	private Session getSession(final Request request, final boolean allowCreate) {
59  		final Session session = request.getSession(false);
60  		return session == null && allowCreate ? request.getSession(true) : session;
61  	}
62  }
