5   package org.greatage.tapestry.grid;
6   
7   import org.apache.tapestry5.grid.ColumnSort;
8   import org.apache.tapestry5.grid.GridDataSource;
9   import org.apache.tapestry5.grid.SortConstraint;
10  import org.greatage.domain.Entity;
11  import org.greatage.domain.EntityService;
12  import org.greatage.domain.PaginationBuilder;
13  
14  import java.io.Serializable;
15  import java.util.List;
16  
17  /**
18   * Needs ga:ga-core dependency.
19   *
20   * @author Ivan Khalopik
21   */
22  public class EntityDataSource<PK extends Serializable, E extends Entity<PK>> implements GridDataSource {
23  	private final EntityService<PK, E> entityService;
24  	private final Class<E> rowType;
25  
26  	private List<E> selection;
27  	private int indexFrom;
28  
29  	public EntityDataSource(final EntityService<PK, E> entityService) {
30  		this.entityService = entityService; //todo: defence
31  		this.rowType = entityService.getEntityClass();
32  	}
33  
34  	public int getAvailableRows() {
35  		return entityService.getEntitiesCount();
36  	}
37  
38  	public E getRowValue(final int index) {
39  		return selection.get(index - indexFrom);
40  	}
41  
42  	public Class<E> getRowType() {
43  		return rowType;
44  	}
45  
46  	public void prepare(final int startIndex, final int endIndex, final List<SortConstraint> sortConstraints) {
47  		final PaginationBuilder builder = PaginationBuilder.create().start(startIndex).end(endIndex + 1);
48  		for (SortConstraint constraint : sortConstraints) {
49  			final String property = constraint.getPropertyModel().getPropertyName();
50  			final boolean ascending = constraint.getColumnSort() == ColumnSort.ASCENDING;
51  			builder.sort(property, ascending);
52  		}
53  		selection = entityService.getEntities(builder);
54  		indexFrom = startIndex;
55  	}
56  }
