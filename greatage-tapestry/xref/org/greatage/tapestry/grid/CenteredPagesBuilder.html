7   import java.util.ArrayList;
8   import java.util.List;

13  public class CenteredPagesBuilder implements PagesBuilder {
14  	private static final int DEFAULT_RANGE = 5;
15  	private final int range;
16  
17  	public CenteredPagesBuilder() {
18  		this(DEFAULT_RANGE);
19  	}
20  
21  	public CenteredPagesBuilder(int range) {
22  		this.range = range;
23  	}
24  
25  	public List<Page> build(PaginationModel model) {
26  		//todo: defence
27  //		Defense.notNull(model, "model");
28  		final int currentPage = model.getCurrentPage();
29  		final int halfRange = range / 2;
30  		int startPage = currentPage - halfRange;
31  		if (startPage < model.getFirstPage()) {
32  			startPage = model.getFirstPage();
33  		}
34  		int endPage = startPage + (range - 1);
35  		if (endPage > model.getLastPage()) {
36  			endPage = model.getLastPage();
37  		}
38  		startPage = endPage - (range - 1);
39  		if (startPage < model.getFirstPage()) {
40  			startPage = model.getFirstPage();
41  		}
42  		final List<Page> pages = new ArrayList<Page>();
43  		pages.add(Page.first());
44  		pages.add(Page.previous());
45  		pages.add(Page.separator());
46  		for (int i = startPage; i <= endPage; i++) {
47  			pages.add(Page.page(i));
48  		}
49  		pages.add(Page.separator());
50  		pages.add(Page.next());
51  		pages.add(Page.last());
52  
53  		return pages;
54  	}
55  }
