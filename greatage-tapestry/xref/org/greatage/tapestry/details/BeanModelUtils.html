7   import org.apache.tapestry5.beaneditor.BeanModel;

9   import java.util.ArrayList;
10  import java.util.Arrays;
11  import java.util.List;

13  /**
14   * @author Ivan Khalopik
15   */
16  public abstract class BeanModelUtils {
17  	private static final String COLUMNS_DELIMITER = "\\s*,\\s*";
18  
19  	public static void setColumns(final BeanModel<?> model, final String columns) {
20  		final List<String> propertyNames = model.getPropertyNames();
21  		final List<String> includeColumns = new ArrayList<String>();
22  		final List<String> addColumns = new ArrayList<String>();
23  		final String[] cols = columns.split(COLUMNS_DELIMITER);
24  		for (String column : cols) {
25  			if (propertyNames.contains(column)) {
26  				includeColumns.add(column);
27  			} else {
28  				addColumns.add(column);
29  			}
30  		}
31  		model.include(includeColumns.toArray(new String[]{""}));
32  		for (String column : addColumns) {
33  			model.add(column, null);
34  		}
35  		model.reorder(cols);
36  	}
37  
38  	public static void setSortColumns(final BeanModel<?> model, final String sortColumns) {
39  		final List<String> propertyNames = model.getPropertyNames();
40  		final List<String> columns = Arrays.asList(sortColumns.split(COLUMNS_DELIMITER));
41  		for (String name : propertyNames) {
42  			model.get(name).sortable(columns.contains(name));
43  		}
44  	}
45  }
