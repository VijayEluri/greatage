21  @SupportsInformalParameters
22  public class Container extends AbstractComponent {
23  
24  	@Parameter(defaultPrefix = BindingConstants.LITERAL)

26

32  	}
33  
34  	public String getClassName() {
35  		final List<String> classes = CollectionFactory.newList();
36  		classes.add(CSSConstants.CONTAINER_CLASS);
37  		if (className != null) {
38  			classes.add(className);
39  		}
40  		return TapestryInternalUtils.toClassAttributeValue(classes);
41  	}
42  
43  	protected void beginRender(MarkupWriter writer) {
44  		writer.element(element, "class", getClassName());
45  		getResources().renderInformalParameters(writer);
46  	}
