46  	public List<String> getSortColumns() {
47  		if (sortColumns == null) {
48  			sortColumns = createSortColumns();
49  		}
50  		return sortColumns;
51  	}
52  
53  	public boolean isCurrent() {
54  		return getSortModel().getColumnSort(sortColumn) != ColumnSort.UNSORTED;
55  	}
56  
57  	public String getSorterClass() {
58  		return CSSConstants.TOOLBAR_MENU_CLASS;
59  	}
60  
61  	@SuppressWarnings({"unchecked"})
62  	private List<String> createSortColumns() {
63  		final List<String> result = new ArrayList<String>();
64  		final List<String> propertyNames = getDataModel().getPropertyNames();
65  		for (String name : propertyNames) {
66  			final PropertyModel propertyModel = getDataModel().get(name);
67  			if (propertyModel.isSortable()) {
68  				result.add(propertyModel.getPropertyName());
69  			}
70  		}
71  		return result;
72  	}
