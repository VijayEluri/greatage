82  	public String getCellClass() {
83  		final List<String> classes = CollectionFactory.newList();
84  		if (columnClass != null) {
85  			classes.add(columnClass);
86  		}
87  		classes.add(getModel().getId());
88  		switch (sort) {
89  			case ASCENDING:
90  				classes.add(CSSConstants.GRID_ASCENDING_CLASS);
91  				break;
92  			case DESCENDING:
93  				classes.add(CSSConstants.GRID_DESCENDING_CLASS);
94  				break;
95  			default:
96  				if (model.isSortable()) {
97  					classes.add(CSSConstants.GRID_UNSORTED_CLASS);
98  				}
99  		}
100 		return TapestryInternalUtils.toClassAttributeValue(classes);
101 	}
