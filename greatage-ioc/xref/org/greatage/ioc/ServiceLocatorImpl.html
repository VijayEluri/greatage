21  public class ServiceLocatorImpl implements ServiceLocator {
22  	private final Map<String, ServiceStatus<?>> servicesById = CollectionUtils.newConcurrentMap();
23  	private final Set<Class<?>> internalServices = CollectionUtils.newSet();
24  
25  	private final Logger logger;
26  
27  	ServiceLocatorImpl(final List<Module> modules) {
28  		final Map<String, Service<?>> services = CollectionUtils.newMap();
29  		for (Module module : modules) {
30  			for (Service service : module.getServices()) {
31  				final String serviceId = service.getServiceId();
32  				if (services.containsKey(serviceId) && !service.isOverride()) {
33  					throw new IllegalStateException(String.format("Service with id '%s' already declared", serviceId));
34  				}
35  				services.put(serviceId, service);
36  			}
37  		}
38  
39  		internalServices.add(LoggerSource.class);
40  		internalServices.add(ProxyFactory.class);
41  		internalServices.add(ScopeManager.class);
42  
43  
44  		final ServiceStatus<ServiceLocator> serviceLocatorStatus = new SimpleHolder<ServiceLocator>(ServiceLocator.class.getSimpleName(), ServiceLocator.class, this);
45  		servicesById.put(serviceLocatorStatus.getServiceId(), serviceLocatorStatus);
46  		int maxLength = serviceLocatorStatus.getServiceId().length();
47  		for (Map.Entry<String, Service<?>> entry : services.entrySet()) {
48  			final String serviceId = entry.getKey();
49  			final Service<?> service = entry.getValue();
50  			final List<Contributor<?>> contributors = CollectionUtils.newList();
51  			final List<Decorator<?>> decorators = CollectionUtils.newList();
52  			final List<Interceptor<?>> interceptors = CollectionUtils.newList();
53  			for (Module module : modules) {
54  				contributors.addAll(module.getContributors(service));
55  				decorators.addAll(module.getDecorators(service));
56  				interceptors.addAll(module.getInterceptors(service));
57  			}
58  			final ServiceStatus<?> status = createServiceHolder(service, contributors, decorators, interceptors);
59  			servicesById.put(serviceId, status);
60  			if (serviceId.length() > maxLength) {
61  				maxLength = serviceId.length();
62  			}
63  		}
64  
65  		final StringBuilder statisticsBuilder = new StringBuilder("Services:\n");
66  		final String format = "%" + maxLength + "s[%s] : %s\n";
67  		for (ServiceStatus<?> status : servicesById.values()) {
68  			statisticsBuilder.append(String.format(format, status.getServiceId(), status.getServiceScope(), status.getServiceClass()));
69  		}
70  
71  		final LoggerSource loggerSource = getService(LoggerSource.class);
72  		logger = loggerSource.getLogger(ServiceLocator.class);
73  		logger.info(statisticsBuilder.toString());
74  	}
75  
76  	public Set<String> getServiceIds() {
77  		return CollectionUtils.newSet(servicesById.keySet());
78  	}
79  
80  	public ServiceStatus<?> getServiceStatus(final String id) {
81  		return servicesById.get(id);
82  	}
83  
84  	public <T> T getService(final String id, final Class<T> serviceClass) {
85  		if (servicesById.containsKey(id)) {
86  			final ServiceStatus<?> status = servicesById.get(id);
87  			final Object service = status.getService();
88  			return serviceClass.cast(service);
89  		}
90  		throw new IllegalStateException(String.format("Can't find service with id %s", id));
91  	}
92  
93  	public <T> T getService(final Class<T> serviceClass) {
94  		final Set<T> services = findServices(serviceClass);
95  		if (services.size() == 1) {
96  			return services.iterator().next();
97  		}
98  		throw new IllegalStateException(String.format("Can't find service of class %s", serviceClass));
99  	}
100 
101 	public <T> Set<T> findServices(final Class<T> serviceClass) {
102 		final Set<T> result = CollectionUtils.newSet();
103 		for (ServiceStatus<?> serviceStatus : servicesById.values()) {
104 			if (serviceClass.isAssignableFrom(serviceStatus.getServiceClass())) {
105 				final Object service = serviceStatus.getService();
106 				result.add(serviceClass.cast(service));
107 			}
108 		}
109 		return result;
110 	}
111 
112 	@SuppressWarnings({"unchecked"})
113 	private ServiceStatus<?> createServiceHolder(final Service<?> service,
114 												 final List<Contributor<?>> contributors,
115 												 final List<Decorator<?>> decorators,
116 												 final List<Interceptor<?>> interceptors) {
117 		return isInternal(service) ?
118 				new InternalService(this, service, contributors, decorators) :
119 				new ScopedService(this, service, contributors, decorators, interceptors);
120 	}
121 
122 	private boolean isInternal(final Service<?> service) {
123 		return internalServices.contains(service.getServiceClass());
124 	}
