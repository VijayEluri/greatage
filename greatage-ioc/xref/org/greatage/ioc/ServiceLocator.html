7   import java.util.Set;

9   /**
10   * This class represents object which can provide access to services by their id, or (when appropriate) by just service
11   * interface. It is an entry point to Great Age IoC container.
12   *
13   * @author Ivan Khalopik
14   * @since 1.0
15   */
16  public interface ServiceLocator {
17  
18  	/**
19  	 * Gets all service ids provided by this service locator.
20  	 *
21  	 * @return all service ids or empty set
22  	 */
23  	Set<String> getServiceIds();
24  
25  	/**
26  	 * Gets service status by specified service id.
27  	 *
28  	 * @param id unique service id
29  	 * @return service status or null if not found
30  	 */
31  	ServiceStatus<?> getServiceStatus(String id);
32  
33  	/**
34  	 * Gets a service instance by specified service id. It returns the service's proxy that implements the same interface
35  	 * as the actual service and instantiates the actual service only as needed with all configured method advices.
36  	 *
37  	 * @param id		   unique service id, case is sensitive
38  	 * @param serviceClass service interface
39  	 * @param <T>          service type
40  	 * @return service instance that implement specified service interface
41  	 * @throws RuntimeException if the service is not found, or if an error occurs instantiating it
42  	 */
43  	<T> T getService(String id, Class<T> serviceClass);
44  
45  	/**
46  	 * Gets a service instance by its service interface. It returns the service's proxy that implements the same interface
47  	 * as the actual service and instantiates the actual service only as needed with all configured method advices.
48  	 *
49  	 * @param serviceClass service interface
50  	 * @param <T>          service type
51  	 * @return service instance that implement specified service interface
52  	 * @throws RuntimeException if there are no ore more than one services found, or if an error occurs instantiating it
53  	 */
54  	<T> T getService(Class<T> serviceClass);
55  
56  	/**
57  	 * Retrieves services by their service interface. It returns the service's proxies that implements the same interface
58  	 * as the actual service and instantiates the actual service only as needed with all configured method advices.
59  	 *
60  	 * @param serviceClass service interface
61  	 * @param <T>          service type
62  	 * @return services that implement specified service interface or empty set
63  	 * @throws RuntimeException if an error occurs instantiating it
64  	 */
65  	<T> Set<T> findServices(Class<T> serviceClass);
66  }
