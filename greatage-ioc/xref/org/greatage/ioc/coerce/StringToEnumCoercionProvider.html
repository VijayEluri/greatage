14  public class StringToEnumCoercionProvider implements CoercionProvider {

16  
17  	/**

20  	@SuppressWarnings("unchecked")
21  	public <S, T> Coercion<S, T> getCoercion(final Class<S> sourceClass, final Class<T> targetClass) {
22  		if (String.class.isAssignableFrom(sourceClass) && Enum.class.isAssignableFrom(targetClass)) {
23  			if (!coercionsByTargetClass.containsKey(targetClass)) {
24  				coercionsByTargetClass.put(targetClass, new StringToEnumCoercion(targetClass));
25  			}
26  			return coercionsByTargetClass.get(targetClass);
27  		}
28  		return null;
