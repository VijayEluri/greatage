5   package org.greatage.ioc;
6   
7   import org.greatage.util.Locker;
8   
9   import java.util.Collection;
10  import java.util.List;
11  import java.util.Map;
12  
13  /**
14   * This class represents service resources proxy that is used during service build phase and adds resulting service
15   * configuration as additional resource. This resources are calculated with service contributors during configure
16   * phase.
17   *
18   * @param <T> service type
19   * @author Ivan Khalopik
20   * @since 1.0
21   */
22  public class ServiceBuildResources<T> extends ServiceAdditionalResources<T> {
23  	private final List<Contributor<T>> contributors;
24  	private final Locker locker = new Locker();
25  
26  	/**
27  	 * Creates new instance of service resources proxy for build phase with defined service resources delegate and list of
28  	 * service contributors.
29  	 *
30  	 * @param delegate	 service resources delegate
31  	 * @param contributors service contributors
32  	 */
33  	ServiceBuildResources(final ServiceResources<T> delegate, final List<Contributor<T>> contributors) {
34  		super(delegate);
35  		this.contributors = contributors;
36  	}
37  
38  	/**
39  	 * {@inheritDoc} Retrieves correspondent service configuration.
40  	 *
41  	 * @return correspondent service configuration or null
42  	 */
43  	@Override
44  	public <E> E getAdditionalResource(final Class<E> resourceClass) {
45  		if (Map.class.equals(resourceClass)) {
46  			locker.lock();
47  			return resourceClass.cast(getMappedConfiguration());
48  		}
49  		if (List.class.equals(resourceClass)) {
50  			locker.lock();
51  			return resourceClass.cast(getOrderedConfiguration());
52  		}
53  		if (Collection.class.equals(resourceClass)) {
54  			locker.lock();
55  			return resourceClass.cast(getConfiguration());
56  		}
57  		return null;
58  	}
59  
60  	/**
61  	 * Creates mapped service configuration contributed by all configured service contributors.
62  	 *
63  	 * @param <K> type of configuration keys
64  	 * @param <V> type of configuration values
65  	 * @return map as service mapped configuration, not null
66  	 */
67  	private <K, V> Map<K, V> getMappedConfiguration() {
68  		final MappedConfigurationImpl<T, K, V> configuration = new MappedConfigurationImpl<T, K, V>(getDelegate());
69  		configure(configuration);
70  		return configuration.build();
71  	}
72  
73  	/**
74  	 * Creates ordered service configuration contributed by all configured service contributors.
75  	 *
76  	 * @param <V> type of configuration items
77  	 * @return list as service ordered configuration, not null
78  	 */
79  	private <V> List<V> getOrderedConfiguration() {
80  		final OrderedConfigurationImpl<T, V> configuration = new OrderedConfigurationImpl<T, V>(getDelegate());
81  		configure(configuration);
82  		return configuration.build();
83  	}
84  
85  	/**
86  	 * Creates unordered service configuration contributed by all configured service contributors.
87  	 *
88  	 * @param <V> type of configuration items
89  	 * @return collection as service unordered configuration, not null
90  	 */
91  	private <V> Collection<V> getConfiguration() {
92  		final ConfigurationImpl<T, V> configuration = new ConfigurationImpl<T, V>(getDelegate());
93  		configure(configuration);
94  		return configuration.build();
95  	}
96  
97  	/**
98  	 * Contributes service configuration by triggering configure phase.
99  	 *
100 	 * @param configuration correspondent service configuration instance
101 	 */
102 	private void configure(final Object configuration) {
103 		final ServiceConfigureResources<T> resources = new ServiceConfigureResources<T>(getDelegate(), configuration);
104 		for (Contributor<T> contributor : contributors) {
105 			contributor.contribute(resources);
106 		}
107 	}
108 }
