5   package org.greatage.ioc;

7   import org.greatage.util.Locker;
8   
9   import java.util.Collection;
10  import java.util.List;
11  import java.util.Map;
12  
13  /**
14   * @author Ivan Khalopik
15   * @since 1.0
16   */
17  public class ServiceBuildResources<T> extends ServiceAdditionalResources<T> {
18  	private final List<Contributor<T>> contributors;
19  	private final Locker locker = new Locker();
20  
21  	ServiceBuildResources(final ServiceResources<T> delegate, final List<Contributor<T>> contributors) {
22  		super(delegate);
23  		this.contributors = contributors;
24  	}
25  
26  	@Override
27  	public <E> E getAdditionalResource(final Class<E> resourceClass) {
28  		if (Map.class.equals(resourceClass)) {
29  			locker.lock();
30  			return resourceClass.cast(getMappedConfiguration());
31  		}
32  		if (List.class.equals(resourceClass)) {
33  			locker.lock();
34  			return resourceClass.cast(getOrderedConfiguration());
35  		}
36  		if (Collection.class.equals(resourceClass)) {
37  			locker.lock();
38  			return resourceClass.cast(getConfiguration());
39  		}
40  		return null;
41  	}
42  
43  	private <K, V> Map<K, V> getMappedConfiguration() {
44  		final MappedConfigurationImpl<T, K, V> configuration = new MappedConfigurationImpl<T, K, V>(getDelegate());
45  		configure(configuration);
46  		return configuration.build();
47  	}
48  
49  	private <V> List<V> getOrderedConfiguration() {
50  		final OrderedConfigurationImpl<T, V> configuration = new OrderedConfigurationImpl<T, V>(getDelegate());
51  		configure(configuration);
52  		return configuration.build();
53  	}
54  
55  	private <V> Collection<V> getConfiguration() {
56  		final ConfigurationImpl<T, V> configuration = new ConfigurationImpl<T, V>(getDelegate());
57  		configure(configuration);
58  		return configuration.build();
59  	}
60  
61  	private void configure(Object configuration) {
62  		final ServiceConfigureResources<T> resources = new ServiceConfigureResources<T>(getDelegate(), configuration);
63  		for (Contributor<T> contributor : contributors) {
64  			contributor.contribute(resources);
65  		}
66  	}
67  }
