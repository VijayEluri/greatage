9   /**
10   * This class represents {@link ServiceResources} proxy implementation that is used for extending its default
11   * functionality by adding additional resources. Its implementations are used for different service build phases.
12   *
13   * @param <T> service type
14   * @author Ivan Khalopik
15   * @since 1.0
16   */
17  public abstract class ServiceAdditionalResources<T> implements ServiceResources<T> {
18  	private final ServiceResources<T> delegate;
19  
20  	/**
21  	 * Creates new instance of service resources proxy with defined service resources delegate instance.
22  	 *
23  	 * @param delegate service resources delegate instance
24  	 */
25  	ServiceAdditionalResources(final ServiceResources<T> delegate) {
26  		this.delegate = delegate;
27  	}
28  
29  	/**
30  	 * Gets service resources delegate instance. It is used to delegate almost all methods of {@link ServiceResources}
31  	 * interface.
32  	 *
33  	 * @return service resources delegate instance
34  	 */
35  	protected ServiceResources<T> getDelegate() {
36  		return delegate;
37  	}
38  
39  	/**
40  	 * {@inheritDoc}
41  	 */
42  	public String getServiceId() {
43  		return delegate.getServiceId();
44  	}
45  
46  	/**
47  	 * {@inheritDoc}
48  	 */
49  	public Class<T> getServiceClass() {
50  		return delegate.getServiceClass();
51  	}
52  
53  	/**
54  	 * {@inheritDoc}
55  	 */
56  	public String getServiceScope() {
57  		return delegate.getServiceScope();
58  	}
59  
60  	/**
61  	 * {@inheritDoc} Firstly tries to retrieve resource from additional resources.
62  	 */
63  	public <E> E getResource(final Class<E> resourceClass, final Annotation... annotations) {
64  		final E resource = getAdditionalResource(resourceClass);
65  		return resource != null ? resource : delegate.getResource(resourceClass, annotations);
66  	}

68  	/**

73  	 * @param <E>           resource type

75  	 */
76  	protected abstract <E> E getAdditionalResource(final Class<E> resourceClass);
