7   import org.greatage.ioc.annotations.*;
8   import org.greatage.util.CollectionUtils;

10  import java.lang.reflect.Method;
11  import java.util.Collection;
12  import java.util.List;

14  /**
15   * @author Ivan Khalopik
16   * @since 1.0
17   */
18  public class ModuleImpl<T> extends ServiceImpl<T> implements Module {
19  	private final List<Service> services = CollectionUtils.newList();
20  	private final List<Contributor> contributors = CollectionUtils.newList();
21  	private final List<Decorator> decorators = CollectionUtils.newList();
22  	private final List<Interceptor> interceptors = CollectionUtils.newList();
23  
24  	ModuleImpl(final Class<T> moduleClass) {
25  		super(moduleClass.getSimpleName(), moduleClass, ScopeConstants.GLOBAL, false);
26  		services.add(this);
27  		for (Method method : moduleClass.getMethods()) {
28  			if (method.isAnnotationPresent(Build.class)) {
29  				services.add(new ServiceFactory(moduleClass, method));
30  			} else if (method.isAnnotationPresent(Contribute.class)) {
31  				contributors.add(new ContributorImpl(moduleClass, method));
32  			} else if (method.isAnnotationPresent(Decorate.class)) {
33  				decorators.add(new DecoratorImpl(moduleClass, method));
34  			} else if (method.isAnnotationPresent(Intercept.class)) {
35  				interceptors.add(new InterceptorImpl(moduleClass, method));
36  			} else if (method.isAnnotationPresent(Bind.class)) {
37  				final ServiceBinderImpl binder = new ServiceBinderImpl();
38  				try {
39  					method.invoke(null, binder);
40  					services.addAll(binder.getServices());
41  				} catch (Exception e) {
42  					throw new RuntimeException("Exception in bind method", e);
43  				}
44  			}
45  		}
46  	}
47  
48  	public Collection<Service> getServices() {
49  		return services;
50  	}
51  
52  	public <T> List<Contributor<T>> getContributors(final Service<T> service) {
53  		final List<Contributor<T>> result = CollectionUtils.newList();
54  		for (Contributor contributor : contributors) {
55  			if (contributor.supports(service)) {
56  				//noinspection unchecked
57  				result.add(contributor);
58  			}
59  		}
60  		return result;
61  	}
62  
63  	public <T> List<Decorator<T>> getDecorators(final Service<T> service) {
64  		final List<Decorator<T>> result = CollectionUtils.newList();
65  		for (Decorator decorator : decorators) {
66  			if (decorator.supports(service)) {
67  				//noinspection unchecked
68  				result.add(decorator);
69  			}
70  		}
71  		return result;
72  	}
73  
74  	public <T> List<Interceptor<T>> getInterceptors(final Service<T> service) {
75  		final List<Interceptor<T>> result = CollectionUtils.newList();
76  		for (Interceptor interceptor : interceptors) {
77  			if (interceptor.supports(service)) {
78  				//noinspection unchecked
79  				result.add(interceptor);
80  			}
81  		}
82  		return result;
83  	}
