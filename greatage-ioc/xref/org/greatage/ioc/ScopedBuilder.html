5   package org.greatage.ioc;
6   
7   import org.greatage.ioc.proxy.ObjectBuilder;
8   import org.greatage.ioc.scope.Scope;
9   import org.greatage.ioc.scope.ScopeManager;
10  
11  /**
12   * This class represents service builder decorator that builds service using service scope.
13   *
14   * @param <T> service type
15   * @author Ivan Khalopik
16   * @since 1.0
17   */
18  public class ScopedBuilder<T> implements ObjectBuilder<T> {
19  	private final ServiceResources<T> resources;
20  	private final ObjectBuilder<T> builder;
21  
22  	private Scope scope;
23  
24  	/**
25  	 * Creates new instance of service builder decorator that builds service using service scope with defined service
26  	 * resources and decorated service builder.
27  	 *
28  	 * @param resources service resources
29  	 * @param builder   decorated service builder
30  	 */
31  	ScopedBuilder(final ServiceResources<T> resources, final ObjectBuilder<T> builder) {
32  		this.resources = resources;
33  		this.builder = builder;
34  	}
35  
36  	/**
37  	 * {@inheritDoc}
38  	 */
39  	public Class<T> getObjectClass() {
40  		return builder.getObjectClass();
41  	}
42  
43  	/**
44  	 * {@inheritDoc} Delegates object creation to service scope.
45  	 */
46  	public T build() {
47  		return getScope().get(resources, builder);
48  	}
49  
50  	/**
51  	 * Gets service scope instance or creates it if it is not created yet.
52  	 *
53  	 * @return service scope instance
54  	 */
55  	private Scope getScope() {
56  		if (scope == null) {
57  			final ScopeManager scopeManager = resources.getResource(ScopeManager.class);
58  			scope = scopeManager.getScope(resources.getServiceScope());
59  			if (scope == null) {
60  				throw new IllegalStateException(
61  						String.format("Wrong scope specified for service (%s, %s). scope=%s",
62  								resources.getServiceId(), resources.getServiceClass(), resources.getServiceScope()));
63  			}
64  		}
65  		return scope;
66  	}
67  }
