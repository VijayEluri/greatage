10  import org.greatage.ioc.logging.Logger;
11  import org.greatage.ioc.logging.LoggerSource;
12  import org.greatage.ioc.symbol.SymbolSource;
13  import org.greatage.util.StringUtils;

15  import java.lang.annotation.Annotation;

17  /**
18   * This class represents default {@link ServiceResources} implementation that is used for retrieving default service
19   * resources such as other services, symbols or specific service logger instance.
20   *
21   * @param <T> service type
22   * @author Ivan Khalopik
23   * @since 1.0
24   */
25  public class ServiceInitialResources<T> implements ServiceResources<T> {
26  	private final ServiceLocator locator;
27  	private final String serviceId;
28  	private final Class<T> serviceClass;
29  	private final String serviceScope;
30  
31  	/**
32  	 * Creates new instance of service initial resources with defined service locator and service definition.
33  	 *
34  	 * @param locator service locator
35  	 * @param service service definition
36  	 */
37  	ServiceInitialResources(final ServiceLocator locator, final Service<T> service) {
38  		this(locator, service.getServiceId(), service.getServiceClass(), service.getScope());
39  	}
40  
41  	/**
42  	 * Creates new instance of service initial resources with defined service locator, service identifier, service class
43  	 * and scope.
44  	 *
45  	 * @param locator	  service locator
46  	 * @param serviceId	service identifier
47  	 * @param serviceClass service class
48  	 * @param serviceScope service scope
49  	 */
50  	ServiceInitialResources(final ServiceLocator locator, final String serviceId, final Class<T> serviceClass,
51  							final String serviceScope) {
52  		this.locator = locator;
53  		this.serviceId = serviceId;
54  		this.serviceClass = serviceClass;
55  		this.serviceScope = serviceScope;
56  	}
57  
58  	/**
59  	 * {@inheritDoc}
60  	 */
61  	public String getServiceId() {
62  		return serviceId;
63  	}
64  
65  	/**
66  	 * {@inheritDoc}
67  	 */
68  	public Class<T> getServiceClass() {
69  		return serviceClass;
70  	}
71  
72  	/**
73  	 * {@inheritDoc}
74  	 */
75  	public String getServiceScope() {
76  		return serviceScope;
77  	}
78  
79  	/**
80  	 * {@inheritDoc}
81  	 */
82  	public <E> E getResource(final Class<E> resourceClass, final Annotation... annotations) {
83  		// if resource type is Logger trying to retrieve service specific logger using LoggerSource service
84  		if (Logger.class.equals(resourceClass)) {
85  			if (LoggerSource.class.equals(serviceClass)) {
86  				return resourceClass.cast(new ConsoleLogger(LoggerSource.class.getName()));
87  			}
88  			final LoggerSource loggerSource = locator.getService(LoggerSource.class);
89  			return resourceClass.cast(loggerSource.getLogger(serviceClass));
90  		}
91  
92  		// if annotated with Symbol annotation trying to find symbol using SymbolSource service
93  		final Symbol symbol = findAnnotation(annotations, Symbol.class);
94  		if (symbol != null) {
95  			// if return type is not String throw an exception
96  			if (!String.class.equals(resourceClass)) {
97  				throw new IllegalArgumentException("Symbol annotation can be used only with String parameters");
98  			}
99  			final SymbolSource symbolSource = getResource(SymbolSource.class);
100 			final String value = symbolSource.getValue(symbol.value());
101 			return resourceClass.cast(value);
102 		}
103 
104 		// if annotated with Inject annotation trying to find service with specified service identifier
105 		final Inject inject = findAnnotation(annotations, Inject.class);
106 		if (inject != null && !StringUtils.isEmpty(inject.value())) {
107 			return locator.getService(inject.value(), resourceClass);
108 		}
109 
110 		// trying to find service by resource type by default
111 		return locator.getService(resourceClass);
112 	}
113 
114 	/**
115 	 * Finds annotation with specified class in annotation array.
116 	 *
117 	 * @param annotations	 annotation array
118 	 * @param annotationClass annotation class to find
119 	 * @param <A>             annotation type
120 	 * @return annotation instance or null if not found
121 	 */
122 	private static <A extends Annotation> A findAnnotation(final Annotation[] annotations,
123 														   final Class<A> annotationClass) {
124 		for (Annotation annotation : annotations) {
125 			if (annotationClass.isInstance(annotation)) {
126 				return annotationClass.cast(annotation);
127 			}
128 		}
129 		return null;
130 	}
