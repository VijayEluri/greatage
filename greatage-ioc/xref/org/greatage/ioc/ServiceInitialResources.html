21  public class ServiceInitialResources<T> implements ServiceResources<T> {
22  	private final ServiceLocator locator;
23  	private final String serviceId;
24  	private final Class<T> serviceClass;
25  	private final String serviceScope;
26  
27  	ServiceInitialResources(final ServiceLocator locator, final Service<T> service) {
28  		this(locator, service.getServiceId(), service.getServiceClass(), service.getScope());
29  	}
30  
31  	ServiceInitialResources(final ServiceLocator locator, final String serviceId, final Class<T> serviceClass, final String serviceScope) {
32  		this.locator = locator;
33  		this.serviceId = serviceId;
34  		this.serviceClass = serviceClass;
35  		this.serviceScope = serviceScope;
36  	}
37  
38  	public String getServiceId() {
39  		return serviceId;
40  	}
41  
42  	public Class<T> getServiceClass() {
43  		return serviceClass;
44  	}
45  
46  	public String getServiceScope() {
47  		return serviceScope;
48  	}
49  
50  	public <E> E getResource(final Class<E> resourceClass, final Annotation... annotations) {
51  		if (Logger.class.equals(resourceClass)) {
52  			if (LoggerSource.class.equals(serviceClass)) {
53  				return resourceClass.cast(new DummyLogger(LoggerSource.class.getName()));
54  			}
55  			final LoggerSource loggerSource = locator.getService(LoggerSource.class);
56  			return resourceClass.cast(loggerSource.getLogger(serviceClass));
57  		}
58  
59  		final Symbol symbol = findAnnotation(annotations, Symbol.class);
60  		if (symbol != null) {
61  			if (!String.class.equals(resourceClass)) {
62  				throw new IllegalArgumentException("Symbol annotation can be used only with String parameters");
63  			}
64  			final SymbolSource symbolSource = getResource(SymbolSource.class);
65  			final String value = symbolSource.getValue(symbol.value());
66  			return resourceClass.cast(value);
67  		}
68  
69  		final Inject inject = findAnnotation(annotations, Inject.class);
70  		if (inject != null && !StringUtils.isEmpty(inject.value())) {
71  			return locator.getService(inject.value(), resourceClass);
72  		}
73  		return locator.getService(resourceClass);
74  	}
75  
76  	private static <A extends Annotation> A findAnnotation(final Annotation[] annotations, final Class<A> annotationClass) {
77  		for (Annotation annotation : annotations) {
78  			if (annotationClass.isInstance(annotation)) return annotationClass.cast(annotation);
79  		}
80  		return null;
81  	}
82  }
