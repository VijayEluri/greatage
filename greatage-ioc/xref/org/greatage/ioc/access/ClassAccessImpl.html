18  /**
19   * This class represents default {@link ClassAccess} implementation that uses {@code java.beans} utilities to retrieve
20   * all bean properties.
21   *
22   * @param <T> bean type
23   * @author Ivan Khalopik
24   * @since 1.1
25   */
26  public class ClassAccessImpl<T> implements ClassAccess<T> {
27  	private final Map<String, PropertyAccess<T>> properties = CollectionUtils.newMap();
28  	private final Class<T> type;
29  
30  	/**
31  	 * Creates new class access instance for specified class which is configured with its properties using {@code
32  	 * java.beans} utilities
33  	 *
34  	 * @param type bean class
35  	 * @throws PropertyAccessException when error occurs while resolving bean properties
36  	 */
37  	ClassAccessImpl(final Class<T> type) {
38  		this.type = type;
39  
40  		try {
41  			final BeanInfo info = Introspector.getBeanInfo(type);
42  			final List<PropertyDescriptor> descriptors = CollectionUtils.newList();
43  			Collections.addAll(descriptors, info.getPropertyDescriptors());
44  
45  			if (type.isInterface() || Modifier.isAbstract(type.getModifiers())) {
46  				final Queue<Class> interfaces = new LinkedList<Class>();
47  				interfaces.addAll(Arrays.asList(type.getInterfaces()));
48  
49  				while (!interfaces.isEmpty()) {
50  					final Class anInterface = interfaces.remove();
51  					final BeanInfo info1 = Introspector.getBeanInfo(anInterface);
52  
53  					Collections.addAll(descriptors, info1.getPropertyDescriptors());
54  					Collections.addAll(interfaces, anInterface.getInterfaces());
55  				}
56  			}
57  
58  			for (PropertyDescriptor descriptor : descriptors) {
59  				if (descriptor.getPropertyType() != null && !properties.containsKey(descriptor.getName())) {
60  					final PropertyAccess<T> propertyAccess = new PropertyAccessImpl<T>(this, descriptor);
61  					properties.put(propertyAccess.getName(), propertyAccess);
62  				}
63  			}
64  
65  			for (Field field : type.getFields()) {
66  				if (!Modifier.isStatic(field.getModifiers()) && !properties.containsKey(field.getName())) {
67  					final PropertyAccess<T> propertyAccess = new PropertyAccessImpl<T>(this, field);
68  					properties.put(propertyAccess.getName(), propertyAccess);
69  				}
70  			}
71  		} catch (Throwable ex) {
72  			throw new PropertyAccessException(
73  					String.format("Can not create class access instance for '%s' class", type), ex);
74  		}
75  	}
76  
77  	/**
78  	 * {@inheritDoc}
79  	 */
80  	public Set<String> getPropertyNames() {
81  		return properties.keySet();
82  	}
83  
84  	/**
85  	 * {@inheritDoc}
86  	 */
87  	public Class<T> getType() {
88  		return type;
89  	}

91  	/**
92  	 * {@inheritDoc}
93  	 */
94  	public PropertyAccess<T> getPropertyAccess(final String propertyName) {
95  		assert propertyName != null;
96  
97  		return properties.get(propertyName);
98  	}

100 	/**
101 	 * {@inheritDoc}
102 	 */
103 	public Object get(final T instance, final String propertyName) {
104 		return propertyAccess(propertyName).get(instance);
105 	}
106 
107 	/**
108 	 * {@inheritDoc}
109 	 */
110 	public void set(final T instance, final String propertyName, final Object value) {
111 		propertyAccess(propertyName).set(instance, value);
112 	}
113 
114 	/**
115 	 * Gets property access instance for specified property, if it is not exists {@link PropertyAccessException} will be
116 	 * thrown.
117 	 *
118 	 * @param propertyName property name
119 	 * @return property access instance, not null
120 	 * @throws PropertyAccessException if bean has no such property
121 	 */
122 	private PropertyAccess<T> propertyAccess(final String propertyName) {
123 		if (!properties.containsKey(propertyName)) {
124 			throw new PropertyAccessException(
125 					String.format("Class '%s' has no property with name '%s'", type, propertyName));
126 		}
127 		return properties.get(propertyName);
128 	}
