23  	private final Map<String, PropertyAccess> properties = CollectionUtils.newMap();
24  	private final Class type;

26  	public ClassAccessImpl(final Class type) {
27  		this.type = type;
28  
29  		try {
30  			final BeanInfo info = Introspector.getBeanInfo(type);
31  			final List<PropertyDescriptor> descriptors = CollectionUtils.newList();
32  			Collections.addAll(descriptors, info.getPropertyDescriptors());
33  
34  			if (type.isInterface() || Modifier.isAbstract(type.getModifiers())) {
35  				final Queue<Class> interfaces = new LinkedList<Class>();
36  				interfaces.addAll(Arrays.asList(type.getInterfaces()));
37  
38  				while (!interfaces.isEmpty()) {
39  					final Class anInterface = interfaces.remove();
40  					final BeanInfo info1 = Introspector.getBeanInfo(anInterface);
41  
42  					Collections.addAll(descriptors, info1.getPropertyDescriptors());
43  					Collections.addAll(interfaces, anInterface.getInterfaces());
44  				}
45  			}
46  
47  			for (PropertyDescriptor descriptor : descriptors) {
48  				if (descriptor.getPropertyType() != null && !properties.containsKey(descriptor.getName())) {
49  					final PropertyAccess propertyAccess = new PropertyAccessImpl(this, descriptor);
50  					properties.put(propertyAccess.getName(), propertyAccess);
51  				}
52  			}
53  
54  			for (Field field : type.getFields()) {
55  				if (!Modifier.isStatic(field.getModifiers()) && !properties.containsKey(field.getName())) {
56  					final PropertyAccess propertyAccess = new PropertyAccessImpl(this, field);
57  					properties.put(propertyAccess.getName(), propertyAccess);
58  				}
59  			}
60  		} catch (Throwable ex) {
61  			throw new RuntimeException(ex);
62  		}
63  	}
