40  	@Bind
41  	public static void bind(final ServiceBinder binder) {
42  		binder.bind(ProxyFactory.class, JavassistProxyFactory.class);
43  		binder.bind(LoggerSource.class, Log4jLoggerSource.class);
44  		binder.bind(ScopeManager.class, ScopeManagerImpl.class);
45  		binder.bind(SymbolSource.class, SymbolSourceImpl.class).withId("GASymbolSource");
46  		binder.bind(SymbolProvider.class, DefaultSymbolProvider.class).withId("ApplicationSymbolProvider");
47  		binder.bind(ResourceLocator.class, ClasspathResourceLocator.class);
48  		binder.bind(MessagesSource.class, MessagesSourceImpl.class);
49  		binder.bind(CacheSource.class, SimpleCacheSource.class).withId("SimpleCacheSource");
50  
51  		binder.bind(ClassAccessSource.class, ClassAccessSourceImpl.class);
52  	}
53  
54  	@Contribute(ScopeManager.class)
55  	public static void contributeScopeManager(final MappedConfiguration<String, Scope> configuration) {
56  		configuration.addInstance(ScopeConstants.GLOBAL, GlobalScope.class);
57  		configuration.addInstance(ScopeConstants.PROTOTYPE, PrototypeScope.class);
58  		configuration.addInstance(ScopeConstants.THREAD, ThreadScope.class);
59  	}
60  
61  	@Contribute(SymbolSource.class)
62  	public static void contributeSymbolSource(final OrderedConfiguration<SymbolProvider> configuration,
63  											  @Inject("ApplicationSymbolProvider") final SymbolProvider applicationSymbolProvider) {
64  		configuration.add(applicationSymbolProvider, "Application");
65  		configuration.addInstance(SystemSymbolProvider.class, "System", "after:Application");
66  	}
