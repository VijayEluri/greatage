44  /**
45   * This class represents base module for Great Age IoC container that configures all needed core services. This are
46   * {@link ProxyFactory}, {@link LoggerSource}, {@link ScopeManager}, {@link SymbolSource}, {@link SymbolProvider},
47   * {@link ResourceLocator}, {@link MessagesSource}, {@link ClassAccessSource}.
48   *
49   * @author Ivan Khalopik
50   * @since 1.0
51   */
52  public class IOCModule {
53  
54  	/**
55  	 * Binds all needed core services with their default implementations. This are {@link ProxyFactory}, {@link
56  	 * LoggerSource}, {@link ScopeManager}, {@link SymbolSource}, {@link SymbolProvider}, {@link ResourceLocator}, {@link
57  	 * MessagesSource}, {@link ClassAccessSource}.
58  	 *
59  	 * @param binder service binder
60  	 */
61  	@Bind
62  	public static void bind(final ServiceBinder binder) {
63  		binder.bind(ProxyFactory.class, JavassistProxyFactory.class);
64  		binder.bind(LoggerSource.class, Log4jLoggerSource.class);
65  		binder.bind(ScopeManager.class, ScopeManagerImpl.class);
66  		binder.bind(TypeCoercer.class, TypeCoercerImpl.class);
67  		binder.bind(CoercionProvider.class, DefaultCoercionProvider.class);
68  		binder.bind(SymbolSource.class, SymbolSourceImpl.class);
69  		binder.bind(SymbolProvider.class, DefaultSymbolProvider.class);
70  
71  		binder.bind(ResourceLocator.class, ClasspathResourceLocator.class);
72  		binder.bind(MessagesSource.class, MessagesSourceImpl.class);
73  
74  		binder.bind(ClassAccessSource.class, ClassAccessSourceImpl.class);
75  	}
76  
77  	/**
78  	 * Configures scope manager service to understand all common scopes. This are <tt>global</tt>, <tt>prototype</tt>,
79  	 * <tt>thread</tt>,
80  	 *
81  	 * @param configuration scope manager mapped configuration
82  	 */
83  	@Contribute(ScopeManager.class)
84  	public static void contributeScopeManager(final MappedConfiguration<String, Scope> configuration) {
85  		configuration.addInstance(ScopeConstants.GLOBAL, GlobalScope.class);
86  		configuration.addInstance(ScopeConstants.PROTOTYPE, PrototypeScope.class);
87  		configuration.addInstance(ScopeConstants.THREAD, ThreadScope.class);
88  	}
89  
90  	/**
91  	 * Configures symbol source service with configured application and system symbol providers.
92  	 *
93  	 * @param configuration  symbol source ordered configuration
94  	 * @param symbolProvider configured application symbol provider
95  	 */
96  	@Contribute(SymbolSource.class)
97  	public static void contributeSymbolSource(final OrderedConfiguration<SymbolProvider> configuration,
98  											  final SymbolProvider symbolProvider) {
99  		configuration.add(symbolProvider, "Application");
100 		configuration.addInstance(SystemSymbolProvider.class, "System", "after:Application");
101 	}
102 
103 	/**
104 	 * Configures type coercer service with configured default coercion provider and string-to-enum provider.
105 	 *
106 	 * @param configuration	type coercer configuration
107 	 * @param coercionProvider default coercion provider service
108 	 */
109 	@Contribute(TypeCoercer.class)
110 	public static void contributeTypeCoercer(final Configuration<CoercionProvider> configuration,
111 											 final CoercionProvider coercionProvider) {
112 		configuration.add(coercionProvider);
113 		configuration.addInstance(StringToEnumCoercionProvider.class);
114 	}
115 
116 	/**
117 	 * Configures default coercion provider with all basic coercions.
118 	 *
119 	 * @param configuration default coercion provider configuration
120 	 */
121 	@Contribute(CoercionProvider.class)
122 	public static void contributeCoercionProvider(final Configuration<Coercion> configuration) {
123 		configuration.addInstance(BooleanToStringCoercion.class);
124 		configuration.addInstance(NumberToStringCoercion.class);
125 		configuration.addInstance(EnumToStringCoercion.class);
126 		configuration.addInstance(StringToBooleanCoercion.class);
127 		configuration.addInstance(StringToIntegerCoercion.class);
128 		configuration.addInstance(StringToDoubleCoercion.class);
129 	}
