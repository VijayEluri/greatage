5   package org.greatage.hibernate.type;

7   import java.sql.PreparedStatement;
8   import java.sql.ResultSet;
9   import java.sql.SQLException;
10  import java.sql.Types;
11  
12  /**
13   * This class represent custom hibernate type that maps enum constants as strings. This string representation will be
14   * taken from {@link org.greatage.hibernate.type.NamedEnum#getName()} method if enum implements this interface or from
15   * standard enum method {@link Enum#name()}.
16   *
17   * @param <E> type of enum
18   * @author Ivan Khalopik
19   * @since 1.0
20   */
21  public class NamedEnumUserType<E extends Enum> extends AbstractEnumUserType<E, String> {
22  
23  	/**
24  	 * Creates new instance of custom hibernate type that maps enum constants as strings.
25  	 */
26  	public NamedEnumUserType() {
27  		super(Types.VARCHAR);
28  	}
29  
30  	/**
31  	 * {@inheritDoc}
32  	 */
33  	@Override
34  	public String get(final ResultSet rs, final String name) throws SQLException {
35  		return rs.getString(name);
36  	}
37  
38  	/**
39  	 * {@inheritDoc}
40  	 */
41  	@Override
42  	public void set(final PreparedStatement st, final String value, final int index) throws SQLException {
43  		st.setString(index, value);
44  	}
45  
46  	/**
47  	 * Converts enum constant to correspondent string value. This string representation will be taken from {@link
48  	 * org.greatage.hibernate.type.NamedEnum#getName()} method if enum implements this interface or from standard enum
49  	 * method {@link Enum#name()}.
50  	 *
51  	 * @param enumConstant enum constant
52  	 * @return unique string representation of enum constant
53  	 */
54  	@Override
55  	protected String enumToValue(final E enumConstant) {
56  		return enumConstant instanceof NamedEnum ? ((NamedEnum) enumConstant).getName() : enumConstant.name();
57  	}
58  }
