7   import org.hibernate.HibernateException;

9   import java.sql.PreparedStatement;
10  import java.sql.ResultSet;
11  import java.sql.SQLException;
12  import java.sql.Types;

14  /**
15   * @author Ivan Khalopik
16   */
17  public class NamedEnumUserType<E extends Enum> extends AbstractEnumUserType<E, String> {
18  	protected NamedEnumUserType() {
19  		super(Types.VARCHAR);
20  	}
21  
22  	@SuppressWarnings({"unchecked"})
23  	public void nullSafeSet(final PreparedStatement st, final Object value, final int index) throws HibernateException, SQLException {
24  		if (value == null) {
25  			st.setNull(index, getSqlType());
26  		} else {
27  			st.setString(index, getValue((E) value));
28  		}
29  	}
30  
31  	@Override
32  	public String get(final ResultSet rs, final String name) throws HibernateException, SQLException {
33  		return rs.getString(name);
34  	}
35  
36  	@Override
37  	public void set(final PreparedStatement st, final String value, final int index) throws HibernateException, SQLException {
38  		st.setString(index, value);
39  	}
40  
41  	@Override
42  	protected String enumToValue(final E e) {
43  		return e instanceof NamedEnum ? ((NamedEnum) e).getName() : e.name();
44  	}
45  }
