5   package org.greatage.util;

7   import java.util.Map;
8   
9   /**
10   * This class represent utility that locks some code by key to prevent use it again.
11   *
12   * @param <K> type of keys
13   * @author Ivan Khalopik
14   * @since 1.0
15   */
16  public class MultiLocker<K> {
17  	private final Map<K, Locker> lockers = CollectionUtils.newConcurrentMap();
18  
19  	/**
20  	 * Checks if specified key is locked.
21  	 *
22  	 * @param key key
23  	 * @throws IllegalStateException if instance is locked
24  	 */
25  	public synchronized void check(final K key) {
26  		if (lockers.containsKey(key)) {
27  			final Locker locker = lockers.get(key);
28  			locker.check();
29  		}
30  	}
31  
32  	/**
33  	 * Locks specified key to prevent use it again.
34  	 *
35  	 * @param key key
36  	 * @throws IllegalStateException if instance is locked
37  	 */
38  	public synchronized void lock(final K key) {
39  		if (!lockers.containsKey(key)) {
40  			lockers.put(key, new Locker());
41  		}
42  		lockers.get(key).lock();
43  	}
44  
45  	@Override
46  	public String toString() {
47  		final DescriptionBuilder builder = new DescriptionBuilder(getClass());
48  		builder.append("lockers", lockers);
49  		return builder.toString();
50  	}
51  }
