005    package org.greatage.hibernate;
006    
007    import org.hibernate.proxy.HibernateProxy;
008    
009    /**
010     * @author Ivan Khalopik
011     */
012    public abstract class HibernateUtils {
013    
014            /**
015             * Get the true, underlying class of a proxied persistent class.
016             *
017             * @param entityClass proxied persistent class
018             * @param <T>         type of entity
019             * @return true, underlying class of a proxied persistent class
020             */
021            @SuppressWarnings("unchecked")
022            public static <T> Class<T> getRealClass(final Class<T> entityClass) {
023                    if (HibernateProxy.class.isAssignableFrom(entityClass)) {
024                            return (Class<T>) entityClass.getSuperclass();
025                    }
026                    return entityClass;
027            }
028    }
