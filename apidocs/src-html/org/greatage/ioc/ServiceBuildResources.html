005    package org.greatage.ioc;
006    
007    import org.greatage.util.Locker;
008    
009    import java.util.Collection;
010    import java.util.List;
011    import java.util.Map;
012    
013    /**
014     * This class represents service resources proxy that is used during service build phase and adds resulting service
015     * configuration as additional resource. This resources are calculated with service contributors during configure
016     * phase.
017     *
018     * @param <T> service type
019     * @author Ivan Khalopik
020     * @since 1.0
021     */
022    public class ServiceBuildResources<T> extends ServiceAdditionalResources<T> {
023            private final List<Contributor<T>> contributors;
024            private final Locker locker = new Locker();
025    
026            /**
027             * Creates new instance of service resources proxy for build phase with defined service resources delegate and list of
028             * service contributors.
029             *
030             * @param delegate       service resources delegate
031             * @param contributors service contributors
032             */
033            ServiceBuildResources(final ServiceResources<T> delegate, final List<Contributor<T>> contributors) {
034                    super(delegate);
035                    this.contributors = contributors;
036            }
037    
038            /**
039             * {@inheritDoc} Retrieves correspondent service configuration.
040             *
041             * @return correspondent service configuration or null
042             */
043            @Override
044            public <E> E getAdditionalResource(final Class<E> resourceClass) {
045                    if (Map.class.equals(resourceClass)) {
046                            locker.lock();
047                            return resourceClass.cast(getMappedConfiguration());
048                    }
049                    if (List.class.equals(resourceClass)) {
050                            locker.lock();
051                            return resourceClass.cast(getOrderedConfiguration());
052                    }
053                    if (Collection.class.equals(resourceClass)) {
054                            locker.lock();
055                            return resourceClass.cast(getConfiguration());
056                    }
057                    return null;
058            }
059    
060            /**
061             * Creates mapped service configuration contributed by all configured service contributors.
062             *
063             * @param <K> type of configuration keys
064             * @param <V> type of configuration values
065             * @return map as service mapped configuration, not null
066             */
067            private <K, V> Map<K, V> getMappedConfiguration() {
068                    final MappedConfigurationImpl<T, K, V> configuration = new MappedConfigurationImpl<T, K, V>(getDelegate());
069                    configure(configuration);
070                    return configuration.build();
071            }
072    
073            /**
074             * Creates ordered service configuration contributed by all configured service contributors.
075             *
076             * @param <V> type of configuration items
077             * @return list as service ordered configuration, not null
078             */
079            private <V> List<V> getOrderedConfiguration() {
080                    final OrderedConfigurationImpl<T, V> configuration = new OrderedConfigurationImpl<T, V>(getDelegate());
081                    configure(configuration);
082                    return configuration.build();
083            }
084    
085            /**
086             * Creates unordered service configuration contributed by all configured service contributors.
087             *
088             * @param <V> type of configuration items
089             * @return collection as service unordered configuration, not null
090             */
091            private <V> Collection<V> getConfiguration() {
092                    final ConfigurationImpl<T, V> configuration = new ConfigurationImpl<T, V>(getDelegate());
093                    configure(configuration);
094                    return configuration.build();
095            }
096    
097            /**
098             * Contributes service configuration by triggering configure phase.
099             *
100             * @param configuration correspondent service configuration instance
101             */
102            private void configure(final Object configuration) {
103                    final ServiceConfigureResources<T> resources = new ServiceConfigureResources<T>(getDelegate(), configuration);
104                    for (Contributor<T> contributor : contributors) {
105                            contributor.contribute(resources);
106                    }
107            }
108    }
