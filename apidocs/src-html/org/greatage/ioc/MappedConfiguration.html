005    package org.greatage.ioc;
006    
007    /**
008     * This class represent mapped service configuration that is used by service contributors to configure service instance.
009     * It is injected to service resources as a map of keyed items at build time.
010     *
011     * @param <K> type of configuration keys
012     * @param <V> type of configuration values
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public interface MappedConfiguration<K, V> {
017    
018            /**
019             * Adds simple keyed item to service configuration.
020             *
021             * @param key   item key
022             * @param value item instance
023             * @return this configuration instance
024             */
025            MappedConfiguration<K, V> add(K key, V value);
026    
027            /**
028             * Instantiates instance of specified class with all injected dependencies and adds this keyed item to service
029             * configuration.
030             *
031             * @param key           item key
032             * @param valueClass item class
033             * @return this configuration instance
034             */
035            MappedConfiguration<K, V> addInstance(K key, Class<? extends V> valueClass);
036    }
