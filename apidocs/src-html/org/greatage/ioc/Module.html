005    package org.greatage.ioc;
006    
007    import java.util.Collection;
008    import java.util.List;
009    
010    /**
011     * This class represents module definition that is used to build {@link org.greatage.ioc.ServiceLocator}. By default it
012     * is configured by simple class with all needed methods that represent service build, configure, decorate and advice
013     * points. It is also may be annotated by {@link org.greatage.ioc.annotations.Dependency} class to define children
014     * modules.
015     *
016     * @author Ivan Khalopik
017     * @see org.greatage.ioc.Service
018     * @see org.greatage.ioc.Contributor
019     * @see org.greatage.ioc.Decorator
020     * @see org.greatage.ioc.Interceptor
021     * @since 1.0
022     */
023    public interface Module {
024    
025            /**
026             * Gets collection of all service definitions of this module.
027             *
028             * @return all service definitions
029             */
030            Collection<Service> getServices();
031    
032            /**
033             * Gets an ordered list of all service contributor definitions for specified service.
034             *
035             * @param service service definition
036             * @param <T>     type of service
037             * @return ordered list of all service contributor definitions for specified service or empty list
038             */
039            <T> List<Contributor<T>> getContributors(Service<T> service);
040    
041            /**
042             * Gets an ordered list of all service decorator definitions for specified service.
043             *
044             * @param service service definition
045             * @param <T>     type of service
046             * @return ordered list of all service decorator definitions for specified service or empty list
047             */
048            <T> List<Decorator<T>> getDecorators(Service<T> service);
049    
050            /**
051             * Gets an ordered list of all service interceptor definitions for specified service.
052             *
053             * @param service service definition
054             * @param <T>     type of service
055             * @return ordered list of all service interceptor definitions for specified service or empty list
056             */
057            <T> List<Interceptor<T>> getInterceptors(Service<T> service);
058    }
