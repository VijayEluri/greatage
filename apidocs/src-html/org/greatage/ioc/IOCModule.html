044    /**
045     * This class represents base module for Great Age IoC container that configures all needed core services. This are
046     * {@link ProxyFactory}, {@link LoggerSource}, {@link ScopeManager}, {@link SymbolSource}, {@link SymbolProvider},
047     * {@link ResourceLocator}, {@link MessagesSource}, {@link ClassAccessSource}.
048     *
049     * @author Ivan Khalopik
050     * @since 1.0
051     */
052    public class IOCModule {
053    
054            /**
055             * Binds all needed core services with their default implementations. This are {@link ProxyFactory}, {@link
056             * LoggerSource}, {@link ScopeManager}, {@link SymbolSource}, {@link SymbolProvider}, {@link ResourceLocator}, {@link
057             * MessagesSource}, {@link ClassAccessSource}.
058             *
059             * @param binder service binder
060             */
061            @Bind
062            public static void bind(final ServiceBinder binder) {
063                    binder.bind(ProxyFactory.class, JavassistProxyFactory.class);
064                    binder.bind(LoggerSource.class, Log4jLoggerSource.class);
065                    binder.bind(ScopeManager.class, ScopeManagerImpl.class);
066                    binder.bind(TypeCoercer.class, TypeCoercerImpl.class);
067                    binder.bind(CoercionProvider.class, DefaultCoercionProvider.class);
068                    binder.bind(SymbolSource.class, SymbolSourceImpl.class);
069                    binder.bind(SymbolProvider.class, DefaultSymbolProvider.class);
070    
071                    binder.bind(ResourceLocator.class, ClasspathResourceLocator.class);
072                    binder.bind(MessagesSource.class, MessagesSourceImpl.class);
073    
074                    binder.bind(ClassAccessSource.class, ClassAccessSourceImpl.class);
075            }
076    
077            /**
078             * Configures scope manager service to understand all common scopes. This are <tt>global</tt>, <tt>prototype</tt>,
079             * <tt>thread</tt>,
080             *
081             * @param configuration scope manager mapped configuration
082             */
083            @Contribute(ScopeManager.class)
084            public static void contributeScopeManager(final MappedConfiguration<String, Scope> configuration) {
085                    configuration.addInstance(ScopeConstants.GLOBAL, GlobalScope.class);
086                    configuration.addInstance(ScopeConstants.PROTOTYPE, PrototypeScope.class);
087                    configuration.addInstance(ScopeConstants.THREAD, ThreadScope.class);
088            }
089    
090            /**
091             * Configures symbol source service with configured application and system symbol providers.
092             *
093             * @param configuration  symbol source ordered configuration
094             * @param symbolProvider configured application symbol provider
095             */
096            @Contribute(SymbolSource.class)
097            public static void contributeSymbolSource(final OrderedConfiguration<SymbolProvider> configuration,
098                                                                                              final SymbolProvider symbolProvider) {
099                    configuration.add(symbolProvider, "Application");
100                    configuration.addInstance(SystemSymbolProvider.class, "System", "after:Application");
101            }
102    
103            /**
104             * Configures type coercer service with configured default coercion provider and string-to-enum provider.
105             *
106             * @param configuration type coercer configuration
107             * @param coercionProvider default coercion provider service
108             */
109            @Contribute(TypeCoercer.class)
110            public static void contributeTypeCoercer(final Configuration<CoercionProvider> configuration,
111                                                                                             final CoercionProvider coercionProvider) {
112                    configuration.add(coercionProvider);
113                    configuration.addInstance(StringToEnumCoercionProvider.class);
114            }
115    
116            /**
117             * Configures default coercion provider with all basic coercions.
118             *
119             * @param configuration default coercion provider configuration
120             */
121            @Contribute(CoercionProvider.class)
122            public static void contributeCoercionProvider(final Configuration<Coercion> configuration) {
123                    configuration.addInstance(BooleanToStringCoercion.class);
124                    configuration.addInstance(NumberToStringCoercion.class);
125                    configuration.addInstance(EnumToStringCoercion.class);
126                    configuration.addInstance(StringToBooleanCoercion.class);
127                    configuration.addInstance(StringToIntegerCoercion.class);
128                    configuration.addInstance(StringToDoubleCoercion.class);
129            }
130    }
