005    package org.greatage.ioc;
006    
007    import org.greatage.util.Ordered;
008    
009    /**
010     * This class represents service decorator definition that distributively decorates services. By default it is
011     * configured by module decorate methods annotated with {@link org.greatage.ioc.annotations.Decorate} class.
012     *
013     * @param <T> service type
014     * @author Ivan Khalopik
015     * @since 1.0
016     */
017    public interface Decorator<T> extends Ordered {
018    
019            /**
020             * Checks if this service decorator definition supports specified service.
021             *
022             * @param service service definition
023             * @return true if this service decorator definition supports specified service, false otherwise
024             */
025            boolean supports(Service service);
026    
027            /**
028             * Decorates service using existing service instance and configured service resource. It retrieves service instance
029             * from service resources by service instance.
030             *
031             * @param resources configured service resources
032             * @return decorated service instance
033             */
034            T decorate(ServiceResources<T> resources);
035    }
