005    package org.greatage.ioc;
006    
007    import org.greatage.ioc.scope.ScopeConstants;
008    
009    /**
010     * This class represents {@link ServiceStatus} implementation for {@link ServiceLocator} service. It always has hardly
011     * specified service identifier, class, scope and instance.
012     *
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public class ServiceLocatorStatus implements ServiceStatus<ServiceLocator> {
017            private final ServiceLocator locator;
018    
019            /**
020             * Creates new instance of service locator status with defined service locator.
021             *
022             * @param locator service locator
023             */
024            ServiceLocatorStatus(final ServiceLocator locator) {
025                    this.locator = locator;
026            }
027    
028            /**
029             * {@inheritDoc} Always returns <tt>ServiceLocator</tt> service identifier.
030             */
031            public String getServiceId() {
032                    return ServiceLocator.class.getName();
033            }
034    
035            /**
036             * {@inheritDoc} Always returns <tt>ServiceLocator</tt> service class.
037             */
038            public Class<ServiceLocator> getServiceClass() {
039                    return ServiceLocator.class;
040            }
041    
042            /**
043             * {@inheritDoc} Always returns <tt>global</tt> service scope.
044             */
045            public String getServiceScope() {
046                    return ScopeConstants.GLOBAL;
047            }
048    
049            /**
050             * {@inheritDoc} Always returns <tt>ServiceLocator</tt> instance.
051             */
052            public ServiceLocator getService() {
053                    return locator;
054            }
055    }
