005    package org.greatage.ioc.logging;
006    
007    import java.util.logging.Level;
008    
009    /**
010     * This class represents logger implementation for JDK 1.4 logging API. It uses next level mapping: trace -> finest,
011     * debug -> fine, info -> info, warn -> warning, error -> severe.
012     *
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public class Jdk14Logger extends AbstractLogger {
017            private final java.util.logging.Logger delegate;
018    
019            /**
020             * Creates new logger instance for JDK 1.4 logging API.
021             *
022             * @param logger JDK 1.4 logger
023             */
024            public Jdk14Logger(final java.util.logging.Logger logger) {
025                    delegate = logger;
026            }
027    
028            /**
029             * {@inheritDoc}
030             */
031            public void trace(final Throwable exception, final String format, final Object... parameters) {
032                    log(Level.FINEST, exception, format, parameters);
033            }
034    
035            /**
036             * {@inheritDoc}
037             */
038            public void debug(final Throwable exception, final String format, final Object... parameters) {
039                    log(Level.FINE, exception, format, parameters);
040            }
041    
042            /**
043             * {@inheritDoc}
044             */
045            public void info(final Throwable exception, final String format, final Object... parameters) {
046                    log(Level.INFO, exception, format, parameters);
047            }
048    
049            /**
050             * {@inheritDoc}
051             */
052            public void warn(final Throwable exception, final String format, final Object... parameters) {
053                    log(Level.WARNING, exception, format, parameters);
054            }
055    
056            /**
057             * {@inheritDoc}
058             */
059            public void error(final Throwable exception, final String format, final Object... parameters) {
060                    log(Level.SEVERE, exception, format, parameters);
061            }
062    
063            /**
064             * {@inheritDoc}
065             */
066            @Override
067            protected String getName() {
068                    return delegate.getName();
069            }
070    
071            /**
072             * Log a formatted with specified parameters message and attached exception at the specified level.
073             *
074             * @param level   log level
075             * @param exception  attached exception
076             * @param format         message format
077             * @param parameters message parameters
078             */
079            private void log(final Level level, final Throwable exception, final String format, final Object... parameters) {
080                    delegate.log(level, String.format(format, parameters), exception);
081            }
082    }
