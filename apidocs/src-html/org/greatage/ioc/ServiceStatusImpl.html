005    package org.greatage.ioc;
006    
007    import org.greatage.ioc.proxy.MethodAdvice;
008    import org.greatage.ioc.proxy.ObjectBuilder;
009    import org.greatage.ioc.proxy.ProxyFactory;
010    import org.greatage.util.DescriptionBuilder;
011    import org.greatage.util.Locker;
012    import org.greatage.util.OrderingUtils;
013    
014    import java.util.ArrayList;
015    import java.util.List;
016    
017    /**
018     * This class represents implementation of {@link ServiceStatus} that is used by default for all services. It lazily
019     * creates, configures, decorates and intercepts service using {@link ProxyFactory} service and scoped builder.
020     *
021     * @param <T> service type
022     * @author Ivan Khalopik
023     * @since 1.0
024     */
025    public class ServiceStatusImpl<T> implements ServiceStatus<T> {
026            private final ServiceResources<T> resources;
027            private final ObjectBuilder<T> builder;
028            private final List<Interceptor<T>> interceptors;
029    
030            private final Locker locker = new Locker();
031    
032            private T serviceInstance;
033    
034            /**
035             * Creates new instance of service status that is used by default for all services. It lazily creates, configures,
036             * decorates and intercepts service using {@link ProxyFactory} service and scoped builder.
037             *
038             * @param locator         service locator
039             * @param service         service definition
040             * @param contributors service contributors
041             * @param decorators   service decorators
042             * @param interceptors service interceptors
043             */
044            ServiceStatusImpl(final ServiceLocator locator,
045                                              final Service<T> service,
046                                              final List<Contributor<T>> contributors,
047                                              final List<Decorator<T>> decorators,
048                                              final List<Interceptor<T>> interceptors) {
049                    this.resources = new ServiceInitialResources<T>(locator, service);
050                    final ServiceBuilder<T> serviceBuilder = new ServiceBuilder<T>(resources, service, contributors, decorators);
051                    this.builder = new ScopedBuilder<T>(resources, serviceBuilder);
052                    this.interceptors = interceptors;
053            }
054    
055            /**
056             * {@inheritDoc}
057             */
058            public String getServiceId() {
059                    return resources.getServiceId();
060            }
061    
062            /**
063             * {@inheritDoc}
064             */
065            public Class<T> getServiceClass() {
066                    return resources.getServiceClass();
067            }
068    
069            /**
070             * {@inheritDoc}
071             */
072            public String getServiceScope() {
073                    return resources.getServiceScope();
074            }
075    
076            /**
077             * {@inheritDoc} Creates service instance using {@link ProxyFactory} service and scoped service builder.
078             */
079            public T getService() {
080                    if (serviceInstance == null) {
081                            locker.lock();
082                            final ProxyFactory proxyFactory = resources.getResource(ProxyFactory.class);
083                            serviceInstance = proxyFactory.createProxy(builder, createAdvices());
084                    }
085                    return serviceInstance;
086            }
087    
088            /**
089             * Creates ordered list of method advices for service using service interceptor definitions.
090             *
091             * @return list of method advices for service or empty list
092             */
093            private List<MethodAdvice> createAdvices() {
094                    final List<MethodAdvice> advices = new ArrayList<MethodAdvice>();
095                    final List<Interceptor<T>> ordered = OrderingUtils.order(interceptors);
096                    for (Interceptor<T> interceptor : ordered) {
097                            final MethodAdvice advice = interceptor.intercept(resources);
098                            advices.add(advice);
099                    }
100                    return advices;
101            }
102    
103            /**
104             * {@inheritDoc}
105             */
106            @Override
107            public String toString() {
108                    final DescriptionBuilder db = new DescriptionBuilder(getClass());
109                    db.append("resources", resources);
110                    db.append("builder", builder);
111                    return db.toString();
112            }
113    }
