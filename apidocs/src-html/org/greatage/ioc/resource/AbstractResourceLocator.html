005    package org.greatage.ioc.resource;
006    
007    /**
008     * This class represents abstract {@link ResourceLocator} implementation that uses configured root resource for
009     * retrieving other resources.
010     *
011     * @author Ivan Khalopik
012     * @since 1.0
013     */
014    public abstract class AbstractResourceLocator implements ResourceLocator {
015            private final Resource rootResource;
016    
017            /**
018             * Creates new instance of resource locator with defined root resource that will be used for retrieving other
019             * resources.
020             *
021             * @param rootResource root resource
022             */
023            protected AbstractResourceLocator(final Resource rootResource) {
024                    this.rootResource = rootResource;
025            }
026    
027            /**
028             * {@inheritDoc}
029             */
030            public Resource getResource(final String path) {
031                    return rootResource.getChild(path);
032            }
033    }
