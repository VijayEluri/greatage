005    package org.greatage.ioc.resource;
006    
007    import org.greatage.util.DescriptionBuilder;
008    
009    import java.util.Locale;
010    
011    /**
012     * This class represents abstract implementation of messages source that delegates messages creation by class to
013     * creation by class name.
014     *
015     * @author Ivan Khalopik
016     * @since 1.0
017     */
018    public abstract class AbstractMessagesSource implements MessagesSource {
019    
020            /**
021             * {@inheritDoc} Delegates invocation to {@link #getMessages(String, java.util.Locale)} method.
022             */
023            public Messages getMessages(final Class clazz, final Locale locale) {
024                    final String resourceName = clazz.getName().replace('.', '/');
025                    return getMessages(resourceName, locale);
026            }
027    
028            /**
029             * {@inheritDoc}
030             */
031            @Override
032            public String toString() {
033                    return new DescriptionBuilder(getClass()).toString();
034            }
035    }
