005    package org.greatage.ioc.resource;
006    
007    import java.net.URL;
008    import java.util.Locale;
009    
010    /**
011     * This class represents {@link Resource} implementation that is based on application classpath.
012     *
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public class ClasspathResource extends AbstractResource {
017            private final ClassLoader classLoader;
018    
019            /**
020             * Creates new instance of classpath resource with defined class loader, parent resource, name and locale.
021             *
022             * @param classLoader resource class loader
023             * @param parent          parent resource
024             * @param name          resource name
025             * @param locale          resource locale
026             */
027            public ClasspathResource(final ClassLoader classLoader,
028                                                             final ClasspathResource parent,
029                                                             final String name,
030                                                             final Locale locale) {
031                    super(parent, name, locale);
032                    this.classLoader = classLoader;
033            }
034    
035            /**
036             * {@inheritDoc} Obtains URL using class loader.
037             */
038            @Override
039            protected URL toURL() {
040                    final String path = getPath();
041                    return classLoader.getResource(path);
042            }
043    
044            /**
045             * {@inheritDoc} Always creates classpath resources.
046             */
047            @Override
048            protected Resource createResource(final Resource parentResource,
049                                                                              final String resourceName,
050                                                                              final Locale resourceLocale) {
051                    return new ClasspathResource(classLoader, (ClasspathResource) parentResource, resourceName, resourceLocale);
052            }
053    }
