005    package org.greatage.ioc.resource;
006    
007    import java.util.Locale;
008    
009    /**
010     * This interface represents utility producing messages by class or name.
011     *
012     * @author Ivan Khalopik
013     * @since 1.0
014     */
015    public interface MessagesSource {
016    
017            /**
018             * Creates or retrieves localized messages in specified locale based on specified class.
019             *
020             * @param clazz  messages class
021             * @param locale messages locale
022             * @return localized messages
023             */
024            Messages getMessages(Class clazz, Locale locale);
025    
026            /**
027             * Creates or retrieves localized messages in specified locale based on specified name.
028             *
029             * @param name   messages name
030             * @param locale messages locale
031             * @return localized messages
032             */
033            Messages getMessages(String name, Locale locale);
034    }
