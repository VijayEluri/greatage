005    package org.greatage.ioc.resource;
006    
007    import java.io.IOException;
008    import java.io.InputStream;
009    import java.util.Locale;
010    
011    /**
012     * This interface represents application hierarchical localized resource that can be represented as input stream.
013     *
014     * @author Ivan Khalopik
015     * @since 1.0
016     */
017    public interface Resource {
018    
019            /**
020             * Gets resource name.
021             *
022             * @return resource name
023             */
024            String getName();
025    
026            /**
027             * Gets parent resource.
028             *
029             * @return instance of parent resource or null if it is in the root
030             */
031            Resource getParent();
032    
033            /**
034             * Gets resource locale for localized resources.
035             *
036             * @return resource locale or null for non localized resources
037             */
038            Locale getLocale();
039    
040            /**
041             * Checks if this resource exists.
042             *
043             * @return true if resource exists, false otherwise
044             */
045            boolean exists();
046    
047            /**
048             * Gets a child resource of this resource by specified relative path.
049             *
050             * @param path relative path to child resource
051             * @return child resource instance, not null
052             */
053            Resource getChild(String path);
054    
055            /**
056             * Gets the same resource in specified locale.
057             *
058             * @param locale resource locale
059             * @return the same resource in specified locale or null if not exists
060             */
061            Resource inLocale(Locale locale);
062    
063            /**
064             * Opens resource as an input stream.
065             *
066             * @return resource input stream representation or null if not exists
067             * @throws IOException if error occurs during resource opening
068             */
069            InputStream open() throws IOException;
070    }
