005    package org.greatage.ioc.scope;
006    
007    import org.greatage.util.CollectionUtils;
008    
009    import java.util.Map;
010    
011    /**
012     * This class represents {@link Scope} implementation that is used for services that have the same state for whole
013     * application. Default scope identifier is {@link ScopeConstants#GLOBAL}.
014     *
015     * @author Ivan Khalopik
016     * @since 1.0
017     */
018    public class GlobalScope extends AbstractScope {
019            private final Map<String, Object> services = CollectionUtils.newConcurrentMap();
020    
021            /**
022             * {@inheritDoc}
023             */
024            @Override
025            protected Map<String, Object> getServices() {
026                    return services;
027            }
028    }
