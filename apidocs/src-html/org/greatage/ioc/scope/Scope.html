005    package org.greatage.ioc.scope;
006    
007    import org.greatage.ioc.ServiceResources;
008    import org.greatage.ioc.proxy.ObjectBuilder;
009    
010    /**
011     * This interface represents service scope object that is used to obtain service instances according to their scope.
012     *
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public interface Scope {
017    
018            /**
019             * Gets service by its resource. If service is missed in this scope new instance will be built via specified builder.
020             *
021             * @param resources service resources that identifies the service
022             * @param builder   service builder
023             * @param <E>       type of service
024             * @return service instance bound to this scope
025             */
026            <E> E get(ServiceResources<E> resources, ObjectBuilder<E> builder);
027    
028            /**
029             * Close scope before end of lifetime.
030             */
031            void cleanup();
032    }
