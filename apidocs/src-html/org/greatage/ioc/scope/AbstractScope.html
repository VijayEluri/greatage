005    package org.greatage.ioc.scope;
006    
007    import org.greatage.ioc.ServiceResources;
008    import org.greatage.ioc.proxy.ObjectBuilder;
009    
010    import java.util.Map;
011    
012    /**
013     * This class represents abstract {@link Scope} implementation that look ups service instance inside the scope and if it
014     * is not found creates new instance using specified service builder.
015     *
016     * @author Ivan Khalopik
017     * @since 1.0
018     */
019    public abstract class AbstractScope implements Scope {
020    
021            /**
022             * {@inheritDoc}
023             */
024            public <E> E get(final ServiceResources<E> resources, final ObjectBuilder<E> builder) {
025                    if (!contains(resources)) {
026                            final E service = builder.build();
027                            put(resources, service);
028                    }
029                    return get(resources);
030            }
031    
032            /**
033             * {@inheritDoc}
034             */
035            public void cleanup() {
036                    //TODO: add functionality of closing services, add init method
037                    getServices().clear();
038            }
039    
040            /**
041             * Checks are there a service instance inside the scope.
042             *
043             * @param resources service resources
044             * @param <E>       service type
045             * @return true if there are a service instance inside the scope, false otherwise
046             */
047            protected <E> boolean contains(final ServiceResources<E> resources) {
048                    return getServices().containsKey(resources.getServiceId());
049            }
050    
051            /**
052             * Gets service instance by its resources.
053             *
054             * @param resources service resources
055             * @param <E>       service type
056             * @return service instance
057             */
058            protected <E> E get(final ServiceResources<E> resources) {
059                    final Class<E> serviceClass = resources.getServiceClass();
060                    return serviceClass.cast(getServices().get(resources.getServiceId()));
061            }
062    
063            /**
064             * Puts service instance to this scope.
065             *
066             * @param resources service resources
067             * @param service   service instance
068             * @param <E>       service type
069             */
070            protected <E> void put(final ServiceResources<E> resources, final E service) {
071                    getServices().put(resources.getServiceId(), service);
072            }
073    
074            /**
075             * Gets all service instances contained inside the scope mapped by their identifiers.
076             *
077             * @return all service instances mapped by their identifiers
078             */
079            protected abstract Map<String, Object> getServices();
080    }
