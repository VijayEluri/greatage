001    package org.greatage.ioc.access;
002    
003    import org.greatage.util.CollectionUtils;
004    
005    import java.util.Map;
006    
007    /**
008     * This class represents default {@link ClassAccessSource} implementation that gets class access instances by specified
009     * class and builds missing.
010     *
011     * @author Ivan Khalopik
012     * @since 1.1
013     */
014    public class ClassAccessSourceImpl implements ClassAccessSource {
015            private final Map<Class, ClassAccess> accesses = CollectionUtils.newConcurrentMap();
016    
017            /**
018             * {@inheritDoc}
019             */
020            @SuppressWarnings("unchecked")
021            public <T> ClassAccess<T> getAccess(final Class<T> clazz) {
022                    assert clazz != null;
023    
024                    if (!accesses.containsKey(clazz)) {
025                            final ClassAccess<T> access = buildClassAccess(clazz);
026                            accesses.put(clazz, access);
027                    }
028                    return accesses.get(clazz);
029            }
030    
031            /**
032             * Creates new class access instance for specified bean class.
033             *
034             * @param clazz bean class
035             * @param <T>   bean type
036             * @return new class access instance
037             * @throws PropertyAccessException when error occurs while getting class access
038             */
039            private <T> ClassAccess<T> buildClassAccess(final Class<T> clazz) {
040                    return new ClassAccessImpl<T>(clazz);
041            }
042    }
