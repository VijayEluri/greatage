001    package org.greatage.ioc.access;
002    
003    import java.util.Set;
004    
005    /**
006     * This interface represents helper that simplifies bean's property resolving mechanism. It provides set of methods for
007     * resolving all property names, getting and setting property values for specified bean instance.
008     *
009     * @param <T> bean type
010     * @author Ivan Khalopik
011     * @see ClassAccessSource
012     * @see PropertyAccess
013     * @since 1.1
014     */
015    public interface ClassAccess<T> {
016    
017            /**
018             * Obtains all property names for configured bean class.
019             *
020             * @return set with all bean property names or empty set
021             */
022            Set<String> getPropertyNames();
023    
024            /**
025             * Gets bean class.
026             *
027             * @return bean class
028             */
029            Class<T> getType();
030    
031            /**
032             * Gets property access instance for specified property.
033             *
034             * @param propertyName property name, not null
035             * @return property access instance for specified property or null if not found
036             */
037            PropertyAccess<T> getPropertyAccess(String propertyName);
038    
039            /**
040             * Gets bean property value for specified property.
041             *
042             * @param instance       bean instance
043             * @param propertyName property name
044             * @return bean property value
045             * @throws PropertyAccessException if bean has no such read property or error occurs while getting value
046             */
047            Object get(T instance, String propertyName);
048    
049            /**
050             * Sets bean property value for specified property.
051             *
052             * @param instance       bean instance
053             * @param propertyName property name
054             * @param value         property value
055             * @throws PropertyAccessException if bean has no such write property or error occurs while setting value
056             */
057            void set(T instance, String propertyName, Object value);
058    }
