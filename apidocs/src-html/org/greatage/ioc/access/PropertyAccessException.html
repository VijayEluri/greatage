001    package org.greatage.ioc.access;
002    
003    import org.greatage.ioc.ApplicationException;
004    
005    /**
006     * This class represents exception that is thrown by property access helpers.
007     *
008     * @author Ivan Khalopik
009     * @since 1.1
010     */
011    public class PropertyAccessException extends ApplicationException {
012    
013            /**
014             * Creates new instance of property access exception with defined detail message.
015             *
016             * @param message detail message
017             */
018            public PropertyAccessException(final String message) {
019                    super(message);
020            }
021    
022            /**
023             * Creates new instance of property access exception with defined detail message and cause throwable.
024             *
025             * @param message detail message
026             * @param cause   cause problem
027             */
028            public PropertyAccessException(final String message, final Throwable cause) {
029                    super(message, cause);
030            }
031    }
