005    package org.greatage.ioc;
006    
007    import org.greatage.ioc.annotations.Dependency;
008    import org.greatage.util.CollectionUtils;
009    import org.greatage.util.Locker;
010    
011    import java.util.List;
012    
013    /**
014     * This class represents utility that simplifies {@link ServiceLocator} building process.
015     *
016     * @author Ivan Khalopik
017     * @since 1.0
018     */
019    public class ServiceLocatorBuilder {
020            private final List<Module> modules = CollectionUtils.newList();
021            private final Locker locker = new Locker();
022    
023            /**
024             * Creates new service locator builder with defined {@link IOCModule} core module.
025             */
026            public ServiceLocatorBuilder() {
027                    addModule(IOCModule.class);
028            }
029    
030            /**
031             * Adds module instances to service locator builder.
032             *
033             * @param moduleInstances module instances
034             * @return this instance
035             */
036            public ServiceLocatorBuilder addModules(final Module... moduleInstances) {
037                    locker.check();
038                    for (Module module : moduleInstances) {
039                            addModule(module);
040                    }
041                    return this;
042            }
043    
044            /**
045             * Adds module classes to service locator builder.
046             *
047             * @param moduleClasses module classes
048             * @return this instance
049             */
050            public ServiceLocatorBuilder addModules(final Class<?>... moduleClasses) {
051                    locker.check();
052                    for (Class<?> moduleClass : moduleClasses) {
053                            addModule(moduleClass);
054                    }
055                    return this;
056            }
057    
058            /**
059             * Adds module instance to service locator builder.
060             *
061             * @param moduleInstance module instance
062             * @return this instance
063             */
064            public ServiceLocatorBuilder addModule(final Module moduleInstance) {
065                    locker.check();
066                    assert moduleInstance != null;
067    
068                    modules.add(moduleInstance);
069                    return this;
070            }
071    
072            /**
073             * Adds module class to service locator builder.
074             *
075             * @param <T>         module type
076             * @param moduleClass module class
077             * @return this instance
078             */
079            public <T> ServiceLocatorBuilder addModule(final Class<T> moduleClass) {
080                    locker.check();
081                    assert moduleClass != null;
082    
083                    // if module class is annotated with Dependency annotation, adds specified module classes
084                    final Dependency dependency = moduleClass.getAnnotation(Dependency.class);
085                    if (dependency != null) {
086                            addModules(dependency.value());
087                    }
088                    addModule(new ModuleImpl<T>(moduleClass));
089                    return this;
090            }
091    
092            /**
093             * Builds new service locator instance with all collected modules defined.
094             *
095             * @return new service locator instance
096             */
097            public ServiceLocator build() {
098                    locker.lock();
099                    return new ServiceLocatorImpl(modules);
100            }
101    
102            /**
103             * Creates new service locator instance for specified module classes + IOCModule.
104             *
105             * @param moduleClasses module classes
106             * @return new service locator instance
107             */
108            public static ServiceLocator createServiceLocator(final Class... moduleClasses) {
109                    final ServiceLocatorBuilder builder = new ServiceLocatorBuilder().addModules(moduleClasses);
110                    return builder.build();
111            }
112    }
