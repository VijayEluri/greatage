005    package org.greatage.ioc;
006    
007    import org.greatage.ioc.annotations.Contribute;
008    import org.greatage.ioc.logging.Logger;
009    import org.greatage.util.StringUtils;
010    
011    import java.lang.reflect.Method;
012    import java.lang.reflect.Modifier;
013    
014    /**
015     * This class represents default implementation of service contribution definition that distributively configures
016     * services. It is based on configuring service by invoking module method.
017     *
018     * @param <T> service type
019     * @author Ivan Khalopik
020     * @since 1.0
021     */
022    public class ContributorImpl<T> implements Contributor<T> {
023            private final Class<?> moduleClass;
024            private final Method configureMethod;
025    
026            private final Class<T> serviceClass;
027            private final String serviceId;
028    
029            /**
030             * Creates new instance of service contribution definition with defined module class and method used for service
031             * configuration. Configuration method must have void return type and be annotated with {@link Contribute} annotation.
032             *
033             * @param moduleClass    module class
034             * @param configureMethod module method used for service configuration
035             */
036            ContributorImpl(final Class<?> moduleClass, final Method configureMethod) {
037                    this.moduleClass = moduleClass;
038                    this.configureMethod = configureMethod;
039    
040                    if (!configureMethod.getReturnType().equals(Void.TYPE)) {
041                            throw new IllegalArgumentException("Configuration method can not return anu value");
042                    }
043    
044                    final Contribute contribute = configureMethod.getAnnotation(Contribute.class);
045                    serviceId = !StringUtils.isEmpty(contribute.serviceId()) ? contribute.serviceId() : null;
046                    //noinspection unchecked
047                    serviceClass = contribute.value();
048            }
049    
050            /**
051             * {@inheritDoc} This contributor supports service if its service identifier or service class correspond to configured
052             * ones.
053             */
054            public boolean supports(final Service service) {
055                    return serviceId != null ?
056                                    service.getServiceId().equals(serviceId) :
057                                    serviceClass.isAssignableFrom(service.getServiceClass());
058            }
059    
060            /**
061             * {@inheritDoc} It configures service by invoking configured module method.
062             */
063            public void contribute(final ServiceResources<T> resources) {
064                    final Logger logger = resources.getResource(Logger.class);
065                    logger.info("Configuring service (%s, %s) from module (%s, %s)", resources.getServiceId(),
066                                    resources.getServiceClass(), moduleClass, configureMethod);
067    
068                    try {
069                            final Object moduleInstance =
070                                            Modifier.isStatic(configureMethod.getModifiers()) ? null : resources.getResource(moduleClass);
071                            final Object[] parameters = InternalUtils.calculateParameters(resources, configureMethod);
072                            configureMethod.invoke(moduleInstance, parameters);
073                    } catch (Exception e) {
074                            throw new RuntimeException(
075                                            String.format("Can't create service configuration with id '%s'", resources.getServiceId()), e);
076                    }
077            }
078    }
