005    package org.greatage.ioc;
006    
007    import java.lang.annotation.Annotation;
008    import java.lang.reflect.Constructor;
009    import java.lang.reflect.Method;
010    
011    /**
012     * This class represents internal utility methods for calculating service dependencies.
013     *
014     * @author Ivan Khalopik
015     * @since 1.0
016     */
017    public class InternalUtils {
018    
019            /**
020             * Calculates service dependencies according to specified build constructor.
021             *
022             * @param resources   service resources
023             * @param constructor service build constructor
024             * @param <T>         service type
025             * @return array of calculated service dependencies
026             */
027            public static <T> Object[] calculateParameters(final ServiceResources<T> resources, final Constructor constructor) {
028                    return calculateInjections(resources, constructor.getParameterTypes(), constructor.getParameterAnnotations());
029            }
030    
031            /**
032             * Calculates service dependencies according to specified build method.
033             *
034             * @param resources service resources
035             * @param method        service build method
036             * @param <T>       service type
037             * @return array of calculated service dependencies
038             */
039            public static <T> Object[] calculateParameters(final ServiceResources<T> resources, final Method method) {
040                    return calculateInjections(resources, method.getParameterTypes(), method.getParameterAnnotations());
041            }
042    
043            /**
044             * Calculates service dependencies according to specified dependency types and annotations.
045             *
046             * @param resources   service resources
047             * @param types    dependency types
048             * @param annotations dependency annotations
049             * @param <T>         service type
050             * @return array of calculated service dependencies
051             */
052            private static <T> Object[] calculateInjections(final ServiceResources<T> resources,
053                                                                                                            final Class<?>[] types, final Annotation[][] annotations) {
054                    final int count = types.length;
055                    final Object[] parameters = new Object[count];
056                    for (int i = 0; i < count; i++) {
057                            parameters[i] = resources.getResource(types[i], annotations[i]);
058                    }
059                    return parameters;
060            }
061    }
