005    package org.greatage.ioc.symbol;
006    
007    import java.util.List;
008    import java.util.regex.Matcher;
009    import java.util.regex.Pattern;
010    
011    /**
012     * This class represents default {@link SymbolSource} implementation that obtains string values by their names. It also
013     * understands expressions like <tt>${another symbol}</tt>.
014     *
015     * @author Ivan Khalopik
016     * @since 1.0
017     */
018    public class SymbolSourceImpl implements SymbolSource {
019            private static final String VARIABLE_PREFIX = "\\$\\{";
020            private static final String VARIABLE_SUFFIX = "\\}";
021            private static final String VARIABLE_REGEX = ".*" + VARIABLE_PREFIX + "(\\S+)" + VARIABLE_SUFFIX + ".*";
022            private static final Pattern VARIABLE_PATTERN = Pattern.compile(VARIABLE_REGEX);
023    
024            private final List<SymbolProvider> providers;
025    
026            /**
027             * Creates new instance of symbol source service with defined ordered configuration of symbol providers.
028             *
029             * @param providers ordered configuration of symbol providers
030             */
031            public SymbolSourceImpl(final List<SymbolProvider> providers) {
032                    assert providers != null;
033    
034                    this.providers = providers;
035            }
036    
037            /**
038             * {@inheritDoc}
039             */
040            public String getValue(final String expression) {
041                    Matcher matcher = VARIABLE_PATTERN.matcher(expression);
042                    if (!matcher.matches()) {
043                            return getSymbol(expression);
044                    }
045    
046                    String result = expression;
047                    while (matcher.matches()) {
048                            final String key = matcher.group(1);
049                            result = result.replaceAll(VARIABLE_PREFIX + key + VARIABLE_SUFFIX, getSymbol(key));
050                            matcher = VARIABLE_PATTERN.matcher(result);
051                    }
052                    return result;
053            }
054    
055            /**
056             * Gets symbol value by specified symbol name. It uses all configured symbol providers to obtain symbol value.
057             *
058             * @param key symbol name
059             * @return symbol value, not null
060             * @throws IllegalStateException when symbol with specified name is not found
061             */
062            private String getSymbol(final String key) {
063                    for (SymbolProvider provider : providers) {
064                            final String symbol = provider.getSymbol(key);
065                            if (symbol != null) {
066                                    return symbol;
067                            }
068                    }
069                    throw new IllegalStateException("Missing variable: " + key);
070            }
071    }
