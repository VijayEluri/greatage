005    package org.greatage.ioc.proxy;
006    
007    import org.greatage.util.DescriptionBuilder;
008    
009    /**
010     * This class represents abstract proxy factory representation that helps to validate input parameters for creation
011     * proxy.
012     *
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public abstract class AbstractProxyFactory implements ProxyFactory {
017    
018            /**
019             * Checks if proxy instance can be created for specified class. It can not be created when original class is not
020             * interface and has no default constructor.
021             *
022             * @param <T>     object type
023             * @param builder object builder
024             * @throws IllegalArgumentException if proxy instance can not be created
025             */
026            protected <T> void validate(final ObjectBuilder<T> builder) {
027                    if (builder == null) {
028                            throw new IllegalArgumentException("Object builder must be specified");
029                    }
030                    final Class<T> proxyClass = builder.getObjectClass();
031                    if (proxyClass == null) {
032                            throw new IllegalArgumentException("Object builder must be configured with proxy class");
033                    }
034                    if (!proxyClass.isInterface()) {
035                            try {
036                                    proxyClass.getConstructor();
037                            } catch (NoSuchMethodException e) {
038                                    throw new IllegalArgumentException(
039                                                    String.format("Proxy class '%s' must have default constructor", proxyClass), e);
040                            }
041                    }
042            }
043    
044            /**
045             * {@inheritDoc}
046             */
047            @Override
048            public String toString() {
049                    return new DescriptionBuilder(getClass()).toString();
050            }
051    }
