005    package org.greatage.ioc.proxy;
006    
007    /**
008     * This class represent method advice that is used to advice object proxy methods with some additional functionality.
009     *
010     * @author Ivan Khalopik
011     * @since 1.0
012     */
013    public interface MethodAdvice {
014    
015            /**
016             * Advices method invocation. It can override method or add some additional functionality to method.
017             *
018             * @param invocation method invocation
019             * @param parameters invocation parameters
020             * @return method return result
021             * @throws Throwable when error occurs during invocation or may be added by this method advice
022             */
023            Object advice(Invocation invocation, Object... parameters) throws Throwable;
024    }
