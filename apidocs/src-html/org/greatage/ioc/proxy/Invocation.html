005    package org.greatage.ioc.proxy;
006    
007    import java.lang.annotation.Annotation;
008    
009    /**
010     * This class represent method invocation in environment of {@link org.greatage.ioc.proxy.MethodAdvice}. It provides
011     * information about method name, its signature and annotations.
012     *
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public interface Invocation {
017    
018            /**
019             * Gets method name.
020             *
021             * @return method name
022             */
023            String getName();
024    
025            /**
026             * Gets method annotation of specified annotation class if presents.
027             *
028             * @param annotationClass annotation class
029             * @param <T>             type of annotation
030             * @return method annotation of specified annotation class or null if missed
031             */
032            <T extends Annotation> T getAnnotation(Class<T> annotationClass);
033    
034            /**
035             * Gets method return type.
036             *
037             * @return method return type
038             */
039            Class<?> getReturnType();
040    
041            /**
042             * Gets method parameter types.
043             *
044             * @return method parameter types
045             */
046            Class<?>[] getParameterTypes();
047    
048            /**
049             * Proceeds an underlying method invocation with specified parameters.
050             *
051             * @param parameters method invocation parameters
052             * @return method invocation return result
053             * @throws Throwable when error occurs during invocation
054             */
055            Object proceed(Object... parameters) throws Throwable;
056    }
