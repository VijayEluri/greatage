005    package org.greatage.ioc.proxy;
006    
007    import java.util.List;
008    
009    /**
010     * This class represents utility producing proxy creation around object builders that instantiates the actual object
011     * only as needed with all configured method advices.
012     *
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public interface ProxyFactory {
017    
018            /**
019             * Creates proxy around specified object builder that instantiates the actual object only as needed with all configured
020             * method advices.
021             *
022             * @param objectBuilder object builder
023             * @param advices          method advices
024             * @param <T>           type of proxy object
025             * @return proxy around specified object builder
026             * @throws RuntimeException if proxy creation is impossible or if an error occurs instantiating it
027             */
028            <T> T createProxy(ObjectBuilder<T> objectBuilder, List<MethodAdvice> advices);
029    }
