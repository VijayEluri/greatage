005    package org.greatage.ioc.proxy;
006    
007    import org.greatage.util.DescriptionBuilder;
008    
009    import java.lang.reflect.Proxy;
010    import java.util.List;
011    
012    /**
013     * This class represents proxy factory implementation using JDK.
014     *
015     * @author Ivan Khalopik
016     * @since 1.0
017     */
018    public class JdkProxyFactory extends AbstractProxyFactory {
019    
020            /**
021             * {@inheritDoc}
022             */
023            public <T> T createProxy(final ObjectBuilder<T> builder, final List<MethodAdvice> advices) {
024                    validate(builder);
025    
026                    final ClassLoader classLoader = builder.getObjectClass().getClassLoader();
027    
028                    final Object proxyInstance = Proxy.newProxyInstance(classLoader,
029                                    new Class<?>[]{builder.getObjectClass()},
030                                    new JdkInvocationHandler<T>(builder, advices));
031                    return builder.getObjectClass().cast(proxyInstance);
032            }
033    
034            /**
035             * {@inheritDoc} It also can not be created when original class is not interface.
036             */
037            @Override
038            protected <T> void validate(final ObjectBuilder<T> builder) {
039                    super.validate(builder);
040                    if (!builder.getObjectClass().isInterface()) {
041                            throw new IllegalArgumentException("Proxy class must be an interface");
042                    }
043            }
044    
045            /**
046             * {@inheritDoc}
047             */
048            @Override
049            public String toString() {
050                    return new DescriptionBuilder(getClass()).toString();
051            }
052    }
