005    package org.greatage.ioc.proxy;
006    
007    import org.greatage.javassist.ClassBuilder;
008    import org.greatage.javassist.ClassPoolEx;
009    import org.greatage.util.DescriptionBuilder;
010    import org.greatage.util.ReflectionUtils;
011    
012    import java.lang.reflect.Method;
013    import java.lang.reflect.Modifier;
014    import java.util.List;
015    import java.util.concurrent.atomic.AtomicLong;
016    
017    /**
018     * This class represents proxy factory implementation using javassist library.
019     *
020     * @author Ivan Khalopik
021     * @since 1.0
022     */
023    public class JavassistProxyFactory extends AbstractProxyFactory {
024            private static final AtomicLong UID_GENERATOR = new AtomicLong(System.currentTimeMillis());
025    
026            private static final String HANDLER_FIELD = "_handler";
027    
028            private final ClassPoolEx pool;
029    
030            /**
031             * Creates new instance of java assist proxy factory with default class pool.
032             */
033            public JavassistProxyFactory() {
034                    this(new ClassPoolEx());
035            }
036    
037            /**
038             * Creates new instance of java assist proxy factory with specified class pool.
039             *
040             * @param pool class pool
041             */
042            public JavassistProxyFactory(final ClassPoolEx pool) {
043                    this.pool = pool;
044            }
045    
046            /**
047             * {@inheritDoc}
048             */
049            public <T> T createProxy(final ObjectBuilder<T> builder, final List<MethodAdvice> advices) {
050                    validate(builder);
051    
052                    final Class<T> proxyClass = createProxyClass(builder);
053                    return ReflectionUtils.newInstance(proxyClass, builder, advices);
054            }
055    
056            /**
057             * Generates proxy class around specified object builder.
058             *
059             * @param builder object builder
060             * @return proxy class around specified object builder
061             */
062            private <T> Class<T> createProxyClass(final ObjectBuilder<T> builder) {
063                    final Class<T> proxyClass = builder.getObjectClass();
064                    final String className = generateName(proxyClass);
065    
066                    final ClassBuilder<T> classBuilder = new ClassBuilder<T>(pool, className, false, proxyClass);
067    
068                    classBuilder.addField(HANDLER_FIELD, Modifier.PRIVATE | Modifier.FINAL, JavassistInvocationHandler.class);
069                    classBuilder.addConstructor(new Class[]{ObjectBuilder.class, List.class}, null,
070                                    String.format("%s = new %s($$);", HANDLER_FIELD, JavassistInvocationHandler.class.getName()));
071    
072                    for (Method method : proxyClass.getMethods()) {
073                            final int modifiers = method.getModifiers();
074                            if (Modifier.isPublic(modifiers) && !Modifier.isFinal(modifiers)) {
075                                    final String methodName = method.getName();
076                                    final String methodBody = String.format(
077                                                    "{ %s realMethod = %s.getDelegate().getClass().getMethod(\"%s\", $sig); return ($r) %s.invoke(realMethod, $args); }",
078                                                    Method.class.getName(), HANDLER_FIELD, methodName, HANDLER_FIELD);
079                                    classBuilder.addMethod(methodName, Modifier.PUBLIC, method.getReturnType(),
080                                                    method.getParameterTypes(), method.getExceptionTypes(), methodBody);
081                            }
082                    }
083    
084                    return classBuilder.build();
085            }
086    
087            /**
088             * Generates name for proxy class. The name will look like simplified original class name prefixed with <tt>$</tt> and
089             * suffixed with generated unique long number.
090             *
091             * @param inputClass original class
092             * @return generated name for proxy class
093             */
094            private String generateName(final Class inputClass) {
095                    final String uid = Long.toHexString(UID_GENERATOR.getAndIncrement());
096                    return "$" + inputClass.getSimpleName() + "_" + uid;
097            }
098    
099            /**
100             * {@inheritDoc}
101             */
102            @Override
103            public String toString() {
104                    final DescriptionBuilder builder = new DescriptionBuilder(getClass());
105                    builder.append("pool", pool);
106                    return builder.toString();
107            }
108    }
