005    package org.greatage.ioc;
006    
007    /**
008     * This class represents service binding options that is returned by {@link org.greatage.ioc.ServiceBinder} to define
009     * service unique id, service scope and is it overrides the existing service.
010     *
011     * @author Ivan Khalopik
012     * @since 1.0
013     */
014    public interface ServiceBindingOptions {
015    
016            /**
017             * Defines service unique id.
018             *
019             * @param id service unique id
020             * @return this service binding options
021             */
022            ServiceBindingOptions withId(String id);
023    
024            /**
025             * Defines service scope.
026             *
027             * @param scope service scope
028             * @return this service binding options
029             */
030            ServiceBindingOptions withScope(String scope);
031    
032            /**
033             * Defines option that service overrides the existing service definition.
034             *
035             * @return this service binding options
036             */
037            ServiceBindingOptions override();
038    }
