005    package org.greatage.ioc;
006    
007    /**
008     * This class represent ordered service configuration that is used by service contributors to configure service
009     * instance. All items will be ordered by their order constraints in {@link org.greatage.util.Ordered} style. It is
010     * injected to service resources as an ordered list of items at build time.
011     *
012     * @param <V> type of configuration items
013     * @author Ivan Khalopik
014     * @see org.greatage.util.OrderingUtils
015     * @since 1.0
016     */
017    public interface OrderedConfiguration<V> {
018    
019            /**
020             * Adds simple item to service configuration according to its order constrains.
021             *
022             * @param item          item instance
023             * @param orderId        unique order id
024             * @param constraints order constraints
025             * @return this configuration instance
026             */
027            OrderedConfiguration<V> add(V item, String orderId, String... constraints);
028    
029            /**
030             * Instantiates instance of specified class with all injected dependencies and adds this item to service configuration
031             * according to its order constrains.
032             *
033             * @param itemClass   item class
034             * @param orderId        unique order id
035             * @param constraints order constraints
036             * @return this configuration instance
037             */
038            OrderedConfiguration<V> addInstance(Class<? extends V> itemClass, String orderId, String... constraints);
039    }
