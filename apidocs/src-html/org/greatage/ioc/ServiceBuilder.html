005    package org.greatage.ioc;
006    
007    import org.greatage.ioc.proxy.ObjectBuilder;
008    import org.greatage.util.OrderingUtils;
009    
010    import java.util.List;
011    
012    /**
013     * This class represents service builder that is used for service creation by its service, contributors and decorators
014     * definitions.
015     *
016     * @param <T> service type
017     * @author Ivan Khalopik
018     * @since 1.0
019     */
020    public class ServiceBuilder<T> implements ObjectBuilder<T> {
021            private final Service<T> service;
022            private final ServiceResources<T> resources;
023            private final List<Contributor<T>> contributors;
024            private final List<Decorator<T>> decorators;
025    
026            /**
027             * Creates new instance of service builder with defined initial service resources, service, contributors and decorators
028             * definitions.
029             *
030             * @param resources     initial service resources
031             * @param service         service definition
032             * @param contributors service contributor definitions
033             * @param decorators   service decorator definitions
034             */
035            ServiceBuilder(final ServiceResources<T> resources,
036                                       final Service<T> service,
037                                       final List<Contributor<T>> contributors,
038                                       final List<Decorator<T>> decorators) {
039                    this.service = service;
040                    this.resources = resources;
041                    this.contributors = contributors;
042                    this.decorators = decorators;
043            }
044    
045            /**
046             * {@inheritDoc}
047             */
048            public Class<T> getObjectClass() {
049                    return service.getServiceClass();
050            }
051    
052            /**
053             * Builds, configures and decorates service instance. It builds new service instance every time when invoked.
054             *
055             * @return ready for use service instance
056             */
057            public T build() {
058                    final ServiceResources<T> buildResources = new ServiceBuildResources<T>(resources, contributors);
059                    final T serviceInstance = service.build(buildResources);
060                    return decorateService(serviceInstance);
061            }
062    
063            /**
064             * Implements decorate phase of service building.
065             *
066             * @param delegate service instance
067             * @return decorated service instance
068             * @throws IllegalStateException when decorator returns the same service instance or null
069             */
070            private T decorateService(final T delegate) {
071                    final List<Decorator<T>> ordered = OrderingUtils.order(decorators);
072                    T decoratedService = delegate;
073                    for (Decorator<T> decorator : ordered) {
074                            final ServiceDecorateResources<T> decorateResources =
075                                            new ServiceDecorateResources<T>(resources, decoratedService);
076                            decoratedService = decorator.decorate(decorateResources);
077                            if (decoratedService == null || decoratedService.equals(decorateResources.getServiceInstance())) {
078                                    throw new IllegalStateException("Decorator returns the same instance");
079                            }
080                    }
081                    return decoratedService;
082            }
083    }
