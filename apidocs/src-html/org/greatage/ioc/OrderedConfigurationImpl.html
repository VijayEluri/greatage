005    package org.greatage.ioc;
006    
007    import org.greatage.util.CollectionUtils;
008    import org.greatage.util.Ordered;
009    import org.greatage.util.OrderingUtils;
010    
011    import java.util.Arrays;
012    import java.util.List;
013    
014    /**
015     * This class represents default implementation of ordered service configuration. The resulting type of such
016     * configuration is {@link List}.
017     *
018     * @param <T> service type
019     * @param <V> type of configuration items
020     * @author Ivan Khalopik
021     * @since 1.0
022     */
023    public class OrderedConfigurationImpl<T, V> extends AbstractConfiguration<T, V, List<V>>
024                    implements OrderedConfiguration<V> {
025            private final List<OrderedItem> configuration = CollectionUtils.newList();
026    
027            /**
028             * Creates new instance of ordered service configuration with defined service resources.
029             *
030             * @param resources service resource
031             */
032            OrderedConfigurationImpl(final ServiceResources<T> resources) {
033                    super(resources);
034            }
035    
036            /**
037             * {@inheritDoc}
038             */
039            public OrderedConfiguration<V> add(final V item, final String id, final String... constraints) {
040                    configuration.add(new OrderedItem(item, id, constraints));
041                    return this;
042            }
043    
044            /**
045             * {@inheritDoc}
046             */
047            public OrderedConfiguration<V> addInstance(final Class<? extends V> clazz, final String id,
048                                                                                               final String... constraints) {
049                    add(newInstance(clazz), id, constraints);
050                    return this;
051            }
052    
053            /**
054             * Creates resulting instance of ordered service configuration as ordered list of configuration items.
055             *
056             * @return resulting ordered list of configuration items
057             */
058            public List<V> build() {
059                    final List<V> result = CollectionUtils.newList();
060    
061                    final List<OrderedItem> orderedItems = OrderingUtils.order(configuration);
062                    for (OrderedItem orderedItem : orderedItems) {
063                            result.add(orderedItem.getItem());
064                    }
065                    return result;
066            }
067    
068            class OrderedItem implements Ordered {
069                    private final String orderId;
070                    private final List<String> orderConstraints;
071                    private final V item;
072    
073                    OrderedItem(final V item, final String orderId, final String... orderConstraints) {
074                            this.item = item;
075                            this.orderId = orderId;
076                            this.orderConstraints = Arrays.asList(orderConstraints);
077                    }
078    
079                    public V getItem() {
080                            return item;
081                    }
082    
083                    public String getOrderId() {
084                            return orderId;
085                    }
086    
087                    public List<String> getOrderConstraints() {
088                            return orderConstraints;
089                    }
090            }
091    }
