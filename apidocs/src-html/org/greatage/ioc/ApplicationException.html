001    package org.greatage.ioc;
002    
003    /**
004     * This class represents base application exception that is thrown almost everywhere in the application powered with
005     * Great Age IoC container. All application specific exceptions must be extended from this class.
006     *
007     * @author Ivan Khalopik
008     * @since 1.1
009     */
010    public class ApplicationException extends RuntimeException {
011    
012            /**
013             * Creates new instance of application exception with defined detail message.
014             *
015             * @param message detail message
016             */
017            public ApplicationException(final String message) {
018                    super(message);
019            }
020    
021            /**
022             * Creates new instance of application exception with defined detail message and cause throwable.
023             *
024             * @param message detail message
025             * @param cause   cause throwable
026             */
027            public ApplicationException(final String message, final Throwable cause) {
028                    super(message, cause);
029            }
030    }
