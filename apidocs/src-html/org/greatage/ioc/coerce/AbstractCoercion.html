001    package org.greatage.ioc.coerce;
002    
003    /**
004     * This class represents abstract {@link Coercion} implementation that defines source and target classes in its
005     * constructor.
006     *
007     * @param <S> source type
008     * @param <T> target type
009     * @author Ivan Khalopik
010     * @since 1.1
011     */
012    public abstract class AbstractCoercion<S, T> implements Coercion<S, T> {
013            private final Class<S> sourceClass;
014            private final Class<T> targetClass;
015    
016            /**
017             * Creates new coercion instance with defined source and target classes.
018             *
019             * @param sourceClass source class, not null
020             * @param targetClass target class, not null
021             */
022            protected AbstractCoercion(final Class<S> sourceClass, final Class<T> targetClass) {
023                    assert sourceClass != null;
024                    assert targetClass != null;
025    
026                    this.sourceClass = sourceClass;
027                    this.targetClass = targetClass;
028            }
029    
030            /**
031             * {@inheritDoc}
032             */
033            public Class<T> getTargetClass() {
034                    return targetClass;
035            }
036    
037            /**
038             * {@inheritDoc}
039             */
040            public Class<S> getSourceClass() {
041                    return sourceClass;
042            }
043    }
