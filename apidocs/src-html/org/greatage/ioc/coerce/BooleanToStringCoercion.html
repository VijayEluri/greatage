001    package org.greatage.ioc.coerce;
002    
003    /**
004     * This class represents {@link Coercion} implementation that converts values from boolean to string.
005     *
006     * @author Ivan Khalopik
007     * @since 1.1
008     */
009    public class BooleanToStringCoercion extends AbstractCoercion<Boolean, String> {
010    
011            /**
012             * Creates new coercion instance that converts values from boolean to string.
013             */
014            public BooleanToStringCoercion() {
015                    super(Boolean.class, String.class);
016            }
017    
018            /**
019             * {@inheritDoc}
020             */
021            public String coerce(final Boolean source) {
022                    return source.toString();
023            }
024    }
