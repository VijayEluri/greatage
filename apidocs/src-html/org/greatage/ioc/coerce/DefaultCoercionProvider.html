001    package org.greatage.ioc.coerce;
002    
003    import java.util.Collection;
004    
005    /**
006     * This class represents {@link CoercionProvider} that can be configured with multiply {@link Coercion} instances.
007     *
008     * @author Ivan Khalopik
009     * @since 1.1
010     */
011    public class DefaultCoercionProvider implements CoercionProvider {
012            private final Collection<Coercion> coercions;
013    
014            /**
015             * Creates new coercion provider configured with multiply {@link Coercion} instances.
016             *
017             * @param coercions coercions configuration
018             */
019            public DefaultCoercionProvider(final Collection<Coercion> coercions) {
020                    assert coercions != null;
021    
022                    this.coercions = coercions;
023            }
024    
025            /**
026             * {@inheritDoc}
027             */
028            @SuppressWarnings("unchecked")
029            public <S, T> Coercion<S, T> getCoercion(final Class<S> sourceClass, final Class<T> targetClass) {
030                    for (Coercion coercion : coercions) {
031                            if (supports(coercion, sourceClass, targetClass)) {
032                                    return coercion;
033                            }
034                    }
035                    return null;
036            }
037    
038            /**
039             * Checks if specified coercion instance supports coercion from source to target class.
040             *
041             * @param coercion      coercion to check
042             * @param sourceClass source class
043             * @param targetClass target class
044             * @return true if specified coercion instance supports coercion from source to target class, false otherwise
045             */
046            private boolean supports(final Coercion coercion, final Class sourceClass, final Class targetClass) {
047                    if (coercion.getSourceClass().isAssignableFrom(sourceClass)) {
048                            if (targetClass.isAssignableFrom(coercion.getTargetClass())) {
049                                    return true;
050                            }
051                    }
052                    return false;
053            }
054    }
