001    package org.greatage.ioc.coerce;
002    
003    /**
004     * This class represents {@link Coercion} implementation that converts values from string to boolean.
005     *
006     * @author Ivan Khalopik
007     * @since 1.1
008     */
009    public class StringToBooleanCoercion extends AbstractCoercion<String, Boolean> {
010            private static final String VALUE_TRUE = "true";
011            private static final String VALUE_FALSE = "false";
012    
013            /**
014             * Creates new coercion instance that converts values from string to boolean.
015             */
016            public StringToBooleanCoercion() {
017                    super(String.class, Boolean.class);
018            }
019    
020            /**
021             * {@inheritDoc}
022             */
023            public Boolean coerce(final String source) {
024                    if (VALUE_TRUE.equalsIgnoreCase(source)) {
025                            return Boolean.TRUE;
026                    } else if (VALUE_FALSE.equalsIgnoreCase(source)) {
027                            return Boolean.FALSE;
028                    }
029                    throw new CoerceException(source, getTargetClass());
030            }
031    }
