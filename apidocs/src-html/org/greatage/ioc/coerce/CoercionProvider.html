001    package org.greatage.ioc.coerce;
002    
003    /**
004     * This interface represents {@link TypeCoercer} configuration point that provides {@link Coercion} instances by
005     * supported source and target coercion classes.
006     *
007     * @author Ivan Khalopik
008     * @since 1.1
009     */
010    public interface CoercionProvider {
011    
012            /**
013             * Gets coercion instance by specified source and target coercion classes. If there no such coercion this method must
014             * return {@code null}.
015             *
016             * @param sourceClass source class
017             * @param targetClass target class
018             * @param <S>         source type
019             * @param <T>         target type
020             * @return coercion instance or null if not found
021             */
022            <S, T> Coercion<S, T> getCoercion(Class<S> sourceClass, Class<T> targetClass);
023    }
