001    package org.greatage.ioc.coerce;
002    
003    /**
004     * This interface represents type coercer service that converts source values to specified target class.
005     *
006     * @author Ivan Khalopik
007     * @since 1.1
008     */
009    public interface TypeCoercer {
010    
011            /**
012             * Converts source value to specified target class.
013             *
014             * @param input   source value
015             * @param targetType target class
016             * @param <S>        source type
017             * @param <T>        target type
018             * @return converted source value
019             * @throws CoerceException if can not convert
020             */
021            <S, T> T coerce(S input, Class<T> targetType);
022    
023            /**
024             * Gets coercion instance that supports coercion from source to target class.
025             *
026             * @param sourceClass source class
027             * @param targetClass target class
028             * @param <S>         source type
029             * @param <T>         target type
030             * @return coercion instance or null if no correspondent coercion found
031             */
032            <S, T> Coercion<S, T> getCoercion(Class<S> sourceClass, Class<T> targetClass);
033    }
