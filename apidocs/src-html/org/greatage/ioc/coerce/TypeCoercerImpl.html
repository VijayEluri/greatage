001    package org.greatage.ioc.coerce;
002    
003    import org.greatage.util.CollectionUtils;
004    import org.greatage.util.CompositeKey;
005    
006    import java.util.Collection;
007    import java.util.Map;
008    
009    /**
010     * This class represents default {@link TypeCoercer} implementation that are configurable with {@link CoercionProvider}
011     * instances.
012     *
013     * @author Ivan Khalopik
014     * @since 1.1
015     */
016    public class TypeCoercerImpl implements TypeCoercer {
017            private final Collection<CoercionProvider> coercionProviders;
018            private final Map<CompositeKey, Coercion> cache = CollectionUtils.newConcurrentMap();
019    
020            /**
021             * Creates new instance of type coercer configured with specified coercion providers
022             *
023             * @param coercionProviders coercion providers
024             */
025            public TypeCoercerImpl(final Collection<CoercionProvider> coercionProviders) {
026                    assert coercionProviders != null;
027    
028                    this.coercionProviders = coercionProviders;
029            }
030    
031            /**
032             * {@inheritDoc}
033             */
034            public <S, T> T coerce(final S input, final Class<T> targetClass) {
035                    assert targetClass != null;
036    
037                    if (input == null || targetClass.isAssignableFrom(input.getClass())) {
038                            return targetClass.cast(input);
039                    }
040    
041                    @SuppressWarnings("unchecked")
042                    final Class<S> sourceClass = (Class<S>) input.getClass();
043    
044                    final Coercion<S, T> coercion = getCoercion(sourceClass, targetClass);
045                    if (coercion == null) {
046                            throw new CoerceException(input, targetClass);
047                    }
048                    return coercion.coerce(input);
049            }
050    
051            /**
052             * {@inheritDoc} It also caches results.
053             */
054            @SuppressWarnings("unchecked")
055            public <S, T> Coercion<S, T> getCoercion(final Class<S> sourceClass, final Class<T> targetClass) {
056                    final CompositeKey key = new CompositeKey(sourceClass, targetClass);
057                    if (cache.containsKey(key)) {
058                            return cache.get(key);
059                    }
060    
061                    for (CoercionProvider provider : coercionProviders) {
062                            final Coercion<S, T> coercion = provider.getCoercion(sourceClass, targetClass);
063                            if (coercion != null) {
064                                    cache.put(key, coercion);
065                                    return coercion;
066                            }
067                    }
068                    return null;
069            }
070    }
