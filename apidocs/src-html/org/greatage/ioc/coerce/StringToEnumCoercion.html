001    package org.greatage.ioc.coerce;
002    
003    import org.greatage.util.CollectionUtils;
004    
005    import java.util.Map;
006    
007    /**
008     * This class represents {@link Coercion} implementation that converts values from string to enum constant.
009     *
010     * @param <E> enum type
011     * @author Ivan Khalopik
012     * @since 1.1
013     */
014    public class StringToEnumCoercion<E extends Enum<E>> extends AbstractCoercion<String, E> {
015            private final Map<String, E> stringToEnum = CollectionUtils.newMap();
016    
017            /**
018             * Creates new coercion instance that converts values from string to enum constant.
019             *
020             * @param targetClass enum class
021             */
022            public StringToEnumCoercion(final Class<E> targetClass) {
023                    super(String.class, targetClass);
024                    for (E e : getTargetClass().getEnumConstants()) {
025                            stringToEnum.put(e.name().toLowerCase(), e);
026                    }
027            }
028    
029            /**
030             * {@inheritDoc}
031             */
032            public E coerce(final String source) {
033                    final String key = source.toLowerCase();
034                    if (!stringToEnum.containsKey(key)) {
035                            throw new CoerceException(source, getTargetClass());
036                    }
037                    return stringToEnum.get(key);
038            }
039    }
