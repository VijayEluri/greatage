001    package org.greatage.ioc.coerce;
002    
003    /**
004     * This class represents {@link Coercion} implementation that converts values from string to double.
005     *
006     * @author Ivan Khalopik
007     * @since 1.1
008     */
009    public class StringToDoubleCoercion extends AbstractCoercion<String, Double> {
010    
011            /**
012             * Creates new coercion instance that converts values from string to double.
013             */
014            public StringToDoubleCoercion() {
015                    super(String.class, Double.class);
016            }
017    
018            /**
019             * {@inheritDoc}
020             */
021            public Double coerce(final String source) {
022                    try {
023                            return Double.parseDouble(source);
024                    } catch (NumberFormatException e) {
025                            throw new CoerceException(source, getTargetClass(), e);
026                    }
027            }
028    }
