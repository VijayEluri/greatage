005    package org.greatage.ioc.annotations;
006    
007    import org.greatage.ioc.scope.ScopeConstants;
008    
009    import java.lang.annotation.Documented;
010    import java.lang.annotation.ElementType;
011    import java.lang.annotation.Retention;
012    import java.lang.annotation.RetentionPolicy;
013    import java.lang.annotation.Target;
014    
015    /**
016     * This annotation marks methods inside the IoC module as service build points that will build service instances with
017     * specified unique id, scope and override option. Such methods can get as arguments other services, collection, list
018     * and map for unordered, ordered and mapped configurations respectively.
019     *
020     * @author Ivan Khalopik
021     * @since 1.0
022     */
023    @Target(ElementType.METHOD)
024    @Retention(RetentionPolicy.RUNTIME)
025    @Documented
026    public @interface Build {
027    
028            /**
029             * This property defines service unique id. If it is empty id will be generated from simplified service interface
030             * name.
031             */
032            String value() default "";
033    
034            /**
035             * This property defines if service definition will be overridden or not.
036             */
037            boolean override() default false;
038    
039            /**
040             * This property defines service scope. This scope must present inside {@link org.greatage.ioc.scope.ScopeManager}
041             * configuration. The default scope is global.
042             */
043            String scope() default ScopeConstants.GLOBAL;
044    }
