005    package org.greatage.ioc.annotations;
006    
007    import java.lang.annotation.Documented;
008    import java.lang.annotation.ElementType;
009    import java.lang.annotation.Retention;
010    import java.lang.annotation.RetentionPolicy;
011    import java.lang.annotation.Target;
012    
013    /**
014     * This annotation marks methods inside the IoC module as bind methods that binds service interfaces to their
015     * automatically built implementations with specified binding options. Such bind method must be static and have one
016     * argument of type {@link org.greatage.ioc.ServiceBinder}.
017     *
018     * @author Ivan Khalopik
019     * @since 1.0
020     */
021    @Target(ElementType.METHOD)
022    @Retention(RetentionPolicy.RUNTIME)
023    @Documented
024    public @interface Bind {
025    }
