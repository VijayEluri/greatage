005    package org.greatage.ioc.annotations;
006    
007    import java.lang.annotation.Documented;
008    import java.lang.annotation.ElementType;
009    import java.lang.annotation.Retention;
010    import java.lang.annotation.RetentionPolicy;
011    import java.lang.annotation.Target;
012    
013    /**
014     * This annotation marks methods inside the IoC module as service contribute points that contributes to service
015     * configuration with specified service id or service interface. Such methods can get as arguments other services and
016     * also {@link org.greatage.ioc.Configuration}, {@link org.greatage.ioc.OrderedConfiguration} and {@link
017     * org.greatage.ioc.MappedConfiguration} for unordered, ordered and mapped service configuration respectively.
018     *
019     * @author Ivan Khalopik
020     * @since 1.0
021     */
022    @Target(ElementType.METHOD)
023    @Retention(RetentionPolicy.RUNTIME)
024    @Documented
025    public @interface Contribute {
026    
027            /**
028             * This property defines service interface for contribution.
029             */
030            Class value();
031    
032            /**
033             * This property defines service unique id for contribution.
034             */
035            String serviceId() default "";
036    }
