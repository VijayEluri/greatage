005    package org.greatage.ioc;
006    
007    import org.greatage.ioc.logging.Logger;
008    
009    import java.lang.reflect.Constructor;
010    
011    /**
012     * This class represent default implementation service definition that automatically instantiates service. It is based
013     * on building service by invoking service implementation constructor.
014     *
015     * @param <T> service type
016     * @author Ivan Khalopik
017     * @since 1.0
018     */
019    public class ServiceImpl<T> implements Service<T> {
020            private final Class<T> serviceClass;
021            private final Class<? extends T> implementationClass;
022            private final String serviceId;
023            private final String scope;
024            private final boolean override;
025    
026            /**
027             * Creates new instance of service definition with defined service identifier, service class, scope and override
028             * option.
029             *
030             * @param serviceId     service identifier
031             * @param serviceClass service class
032             * @param scope         service scope
033             * @param override       option that determines is service overrides its default definition
034             */
035            ServiceImpl(final String serviceId, final Class<T> serviceClass, final String scope, final boolean override) {
036                    this(serviceId, serviceClass, serviceClass, scope, override);
037            }
038    
039            /**
040             * Creates new instance of service definition with defined service identifier, service class, service implementation
041             * class, scope and override option.
042             *
043             * @param serviceId                service identifier
044             * @param serviceClass          service class
045             * @param implementationClass service implementation class
046             * @param scope                    service scope
047             * @param override                      option that determines is service overrides its default definition
048             */
049            ServiceImpl(final String serviceId,
050                                    final Class<T> serviceClass,
051                                    final Class<? extends T> implementationClass,
052                                    final String scope,
053                                    final boolean override) {
054                    this.serviceClass = serviceClass;
055                    this.implementationClass = implementationClass;
056                    this.serviceId = serviceId;
057                    this.scope = scope;
058                    this.override = override;
059            }
060    
061            /**
062             * {@inheritDoc}
063             */
064            public String getServiceId() {
065                    return serviceId;
066            }
067    
068            /**
069             * {@inheritDoc}
070             */
071            public Class<T> getServiceClass() {
072                    return serviceClass;
073            }
074    
075            /**
076             * {@inheritDoc}
077             */
078            public boolean isOverride() {
079                    return override;
080            }
081    
082            /**
083             * {@inheritDoc}
084             */
085            public String getScope() {
086                    return scope;
087            }
088    
089            /**
090             * {@inheritDoc} It automatically builds service instance by invoking service implementation constructor.
091             */
092            public T build(final ServiceResources<T> resources) {
093                    final Logger logger = resources.getResource(Logger.class);
094                    logger.info("Building service (%s, %s) from (%s)", serviceId, serviceClass, implementationClass);
095    
096                    try {
097                            final Constructor constructor = implementationClass.getConstructors()[0];
098                            final Object[] parameters = InternalUtils.calculateParameters(resources, constructor);
099                            return implementationClass.cast(constructor.newInstance(parameters));
100                    } catch (Exception e) {
101                            throw new RuntimeException(
102                                            String.format("Can't create service of class '%s' with id '%s'", serviceClass, serviceId), e);
103                    }
104            }
105    }
