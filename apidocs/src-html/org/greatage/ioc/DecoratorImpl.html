005    package org.greatage.ioc;
006    
007    import org.greatage.ioc.annotations.Decorate;
008    import org.greatage.ioc.annotations.Order;
009    import org.greatage.ioc.logging.Logger;
010    import org.greatage.util.CollectionUtils;
011    import org.greatage.util.StringUtils;
012    
013    import java.lang.reflect.Method;
014    import java.lang.reflect.Modifier;
015    import java.util.Arrays;
016    import java.util.List;
017    
018    /**
019     * This class represents default implementation of service decorator definition that distributively decorates services.
020     * It is based on decorating service by invoking module method.
021     *
022     * @param <T> service type
023     * @author Ivan Khalopik
024     * @since 1.0
025     */
026    public class DecoratorImpl<T> implements Decorator<T> {
027            private final Class<?> moduleClass;
028            private final Method decorateMethod;
029    
030            private final Class<T> serviceClass;
031            private final String serviceId;
032    
033            private final String orderId;
034            private final List<String> orderConstraints;
035    
036            /**
037             * Creates new instance of service decorator definition with defined module class and method used for service
038             * decoration. Decoration method must have return and first argument types equal to service type and be annotated with
039             * {@link Decorate} annotation. For decorators ordering it may also be annotated with {@link Order} annotation.
040             *
041             * @param moduleClass   module class
042             * @param decorateMethod module method used for service decoration
043             */
044            DecoratorImpl(final Class<?> moduleClass, final Method decorateMethod) {
045                    this.moduleClass = moduleClass;
046                    this.decorateMethod = decorateMethod;
047    
048                    //noinspection unchecked
049                    serviceClass = (Class<T>) decorateMethod.getReturnType();
050                    if (!serviceClass.equals(decorateMethod.getParameterTypes()[0])) {
051                            throw new IllegalArgumentException("Decorate method must have equals return type and first argument type");
052                    }
053    
054                    final Decorate decorate = decorateMethod.getAnnotation(Decorate.class);
055                    serviceId = !StringUtils.isEmpty(decorate.serviceId()) ? decorate.serviceId() : null;
056    
057                    final Order order = decorateMethod.getAnnotation(Order.class);
058                    if (order != null) {
059                            orderId = order.value();
060                            orderConstraints = Arrays.asList(order.constraints());
061                    } else {
062                            orderId = "";
063                            orderConstraints = CollectionUtils.newList();
064                    }
065            }
066    
067            /**
068             * {@inheritDoc} This decorator supports service if its service identifier or service class correspond to configured
069             * ones.
070             */
071            public boolean supports(final Service service) {
072                    return serviceId != null ?
073                                    service.getServiceId().equals(serviceId) :
074                                    serviceClass.isAssignableFrom(service.getServiceClass());
075            }
076    
077            /**
078             * {@inheritDoc}
079             */
080            public String getOrderId() {
081                    return orderId;
082            }
083    
084            /**
085             * {@inheritDoc}
086             */
087            public List<String> getOrderConstraints() {
088                    return orderConstraints;
089            }
090    
091            /**
092             * {@inheritDoc} It decorates service by invoking configured module method.
093             */
094            public T decorate(final ServiceResources<T> resources) {
095                    final Logger logger = resources.getResource(Logger.class);
096                    logger.info("Decorating service (%s, %s) from module (%s, %s)", resources.getServiceId(),
097                                    resources.getServiceClass(), moduleClass, decorateMethod);
098    
099                    try {
100                            final Class<T> resultClass = resources.getServiceClass();
101                            final Object moduleInstance =
102                                            Modifier.isStatic(decorateMethod.getModifiers()) ? null : resources.getResource(moduleClass);
103                            final Object[] parameters = InternalUtils.calculateParameters(resources, decorateMethod);
104                            return resultClass.cast(decorateMethod.invoke(moduleInstance, parameters));
105                    } catch (Exception e) {
106                            throw new RuntimeException(
107                                            String.format("Can't create service configuration with id '%s'", resources.getServiceId()), e);
108                    }
109            }
110    }
