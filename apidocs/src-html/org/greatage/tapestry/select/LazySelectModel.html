005    package org.greatage.tapestry.select;
006    
007    import org.apache.tapestry5.OptionGroupModel;
008    import org.apache.tapestry5.OptionModel;
009    import org.apache.tapestry5.internal.OptionModelImpl;
010    import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
011    import org.apache.tapestry5.util.AbstractSelectModel;
012    
013    import java.util.Collection;
014    import java.util.List;
015    
016    /**
017     * @author Ivan Khalopik
018     */
019    public abstract class LazySelectModel<V> extends AbstractSelectModel {
020            private List<OptionModel> optionModels;
021    
022            public List<OptionGroupModel> getOptionGroups() {
023                    return null;
024            }
025    
026            public List<OptionModel> getOptions() {
027                    if (optionModels == null) {
028                            optionModels = createOptionModels();
029                    }
030                    return optionModels;
031            }
032    
033            protected abstract Collection<V> getValues();
034    
035            protected String getLabel(V value) {
036                    return String.valueOf(value);
037            }
038    
039            //todo: add optional values sorting
040    
041            protected List<OptionModel> createOptionModels() {
042                    final List<OptionModel> result = CollectionFactory.newList();
043                    for (V v : getValues()) {
044                            result.add(new OptionModelImpl(getLabel(v), v));
045                    }
046                    return result;
047            }
048    }
