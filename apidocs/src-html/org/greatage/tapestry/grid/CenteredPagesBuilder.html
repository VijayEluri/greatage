005    package org.greatage.tapestry.grid;
006    
007    import java.util.ArrayList;
008    import java.util.List;
009    
010    /**
011     * @author Ivan Khalopik
012     */
013    public class CenteredPagesBuilder implements PagesBuilder {
014            private static final int DEFAULT_RANGE = 5;
015            private final int range;
016    
017            public CenteredPagesBuilder() {
018                    this(DEFAULT_RANGE);
019            }
020    
021            public CenteredPagesBuilder(int range) {
022                    this.range = range;
023            }
024    
025            public List<Page> build(PaginationModel model) {
026                    //todo: defence
027    //              Defense.notNull(model, "model");
028                    final int currentPage = model.getCurrentPage();
029                    final int halfRange = range / 2;
030                    int startPage = currentPage - halfRange;
031                    if (startPage < model.getFirstPage()) {
032                            startPage = model.getFirstPage();
033                    }
034                    int endPage = startPage + (range - 1);
035                    if (endPage > model.getLastPage()) {
036                            endPage = model.getLastPage();
037                    }
038                    startPage = endPage - (range - 1);
039                    if (startPage < model.getFirstPage()) {
040                            startPage = model.getFirstPage();
041                    }
042                    final List<Page> pages = new ArrayList<Page>();
043                    pages.add(Page.first());
044                    pages.add(Page.previous());
045                    pages.add(Page.separator());
046                    for (int i = startPage; i <= endPage; i++) {
047                            pages.add(Page.page(i));
048                    }
049                    pages.add(Page.separator());
050                    pages.add(Page.next());
051                    pages.add(Page.last());
052    
053                    return pages;
054            }
055    }
