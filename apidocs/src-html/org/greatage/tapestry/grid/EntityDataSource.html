005    package org.greatage.tapestry.grid;
006    
007    import org.apache.tapestry5.grid.ColumnSort;
008    import org.apache.tapestry5.grid.GridDataSource;
009    import org.apache.tapestry5.grid.SortConstraint;
010    import org.greatage.domain.Entity;
011    import org.greatage.domain.EntityService;
012    import org.greatage.domain.PaginationBuilder;
013    
014    import java.io.Serializable;
015    import java.util.List;
016    
017    /**
018     * Needs ga:ga-core dependency.
019     *
020     * @author Ivan Khalopik
021     */
022    public class EntityDataSource<PK extends Serializable, E extends Entity<PK>> implements GridDataSource {
023            private final EntityService<PK, E> entityService;
024            private final Class<E> rowType;
025    
026            private List<E> selection;
027            private int indexFrom;
028    
029            public EntityDataSource(final EntityService<PK, E> entityService) {
030                    this.entityService = entityService; //todo: defence
031                    this.rowType = entityService.getEntityClass();
032            }
033    
034            public int getAvailableRows() {
035                    return entityService.getEntitiesCount();
036            }
037    
038            public E getRowValue(final int index) {
039                    return selection.get(index - indexFrom);
040            }
041    
042            public Class<E> getRowType() {
043                    return rowType;
044            }
045    
046            public void prepare(final int startIndex, final int endIndex, final List<SortConstraint> sortConstraints) {
047                    final PaginationBuilder builder = PaginationBuilder.create().start(startIndex).end(endIndex + 1);
048                    for (SortConstraint constraint : sortConstraints) {
049                            final String property = constraint.getPropertyModel().getPropertyName();
050                            final boolean ascending = constraint.getColumnSort() == ColumnSort.ASCENDING;
051                            builder.sort(property, ascending);
052                    }
053                    selection = entityService.getEntities(builder);
054                    indexFrom = startIndex;
055            }
056    }
