005    package org.greatage.tapestry.internal;
006    
007    import org.apache.tapestry5.ioc.annotations.UsesConfiguration;
008    import org.greatage.tapestry.services.ClassResolver;
009    
010    import java.util.Collection;
011    
012    /**
013     * @author Ivan Khalopik
014     */
015    @UsesConfiguration(ClassResolver.class)
016    public class RealClassResolver {
017            private final Collection<ClassResolver> classResolvers;
018    
019            public RealClassResolver(Collection<ClassResolver> classResolvers) {
020                    this.classResolvers = classResolvers;
021            }
022    
023            <T> Class<T> getRealClass(Class<T> clazz) {
024                    for (ClassResolver classResolver : classResolvers) {
025                            final Class<T> realClass = classResolver.getRealClass(clazz);
026                            if (realClass != null) {
027                                    return realClass;
028                            }
029                    }
030                    return clazz;
031            }
032    }
