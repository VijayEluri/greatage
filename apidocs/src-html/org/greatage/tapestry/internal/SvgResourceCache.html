001    package org.greatage.tapestry.internal;
002    
003    import org.apache.tapestry5.internal.services.ResourceCache;
004    import org.apache.tapestry5.internal.services.StreamableResource;
005    import org.apache.tapestry5.ioc.Resource;
006    import org.apache.tapestry5.services.InvalidationListener;
007    
008    /**
009     * @author Ivan Khalopik
010     * @since 1.1
011     */
012    public class SvgResourceCache implements ResourceCache {
013            private final ResourceCache delegate;
014    
015            public SvgResourceCache(final ResourceCache delegate) {
016                    this.delegate = delegate;
017            }
018    
019            public boolean requiresDigest(final Resource resource) {
020                    return delegate.requiresDigest(resource);
021            }
022    
023            public StreamableResource getStreamableResource(final Resource resource) {
024                    final StreamableResource streamableResource = delegate.getStreamableResource(resource);
025                    if (resource.getPath().endsWith(".svg")) {
026                            return new SvgStreamableResource(streamableResource);
027                    }
028                    return streamableResource;
029            }
030    
031            public String getDigest(final Resource resource) {
032                    return delegate.getDigest(resource);
033            }
034    
035            public long getTimeModified(final Resource resource) {
036                    return delegate.getTimeModified(resource);
037            }
038    
039            public void addInvalidationListener(final InvalidationListener listener) {
040                    delegate.addInvalidationListener(listener);
041            }
042    }
