005    package org.greatage.tapestry.internal;
006    
007    import org.apache.tapestry5.ValueEncoder;
008    import org.apache.tapestry5.internal.services.AbstractSessionPersistentFieldStrategy;
009    import org.apache.tapestry5.services.Request;
010    import org.apache.tapestry5.services.ValueEncoderSource;
011    
012    /**
013     * @author Ivan Khalopik
014     */
015    public class EncodedSessionPersistentFieldStrategy extends AbstractSessionPersistentFieldStrategy {
016            private static final String ENCODED_SESSION_PREFIX = "encoded-session:";
017    
018            private final ValueEncoderSource valueEncoderSource;
019            private final RealClassResolver realClassResolver;
020    
021            public EncodedSessionPersistentFieldStrategy(Request request, ValueEncoderSource valueEncoderSource, RealClassResolver realClassResolver) {
022                    this(ENCODED_SESSION_PREFIX, request, valueEncoderSource, realClassResolver);
023            }
024    
025            protected EncodedSessionPersistentFieldStrategy(String prefix, Request request, ValueEncoderSource valueEncoderSource, RealClassResolver realClassResolver) {
026                    super(prefix, request);
027                    this.valueEncoderSource = valueEncoderSource;
028                    this.realClassResolver = realClassResolver;
029            }
030    
031            @Override
032            @SuppressWarnings({"unchecked"})
033            protected Object convertApplicationValueToPersisted(Object newValue) {
034                    if (newValue != null) {
035                            final Class<?> type = getRealClass(newValue.getClass());
036                            final ValueEncoder valueEncoder = getValueEncoder(type);
037                            return new PersistedValue(type, valueEncoder.toClient(newValue));
038                    }
039                    return null;
040            }
041    
042            @Override
043            @SuppressWarnings({"unchecked"})
044            protected Object convertPersistedToApplicationValue(Object persistedValue) {
045                    if (persistedValue != null) {
046                            final PersistedValue value = (PersistedValue) persistedValue;
047                            final ValueEncoder valueEncoder = getValueEncoder(value.getValueClass());
048                            return valueEncoder.toValue(value.getValue());
049                    }
050                    return null;
051            }
052    
053            protected <T> Class<T> getRealClass(Class<T> clazz) {
054                    return realClassResolver.getRealClass(clazz);
055            }
056    
057            protected <T> ValueEncoder<T> getValueEncoder(Class<T> clazz) {
058                    return valueEncoderSource.getValueEncoder(clazz);
059            }
060    }
