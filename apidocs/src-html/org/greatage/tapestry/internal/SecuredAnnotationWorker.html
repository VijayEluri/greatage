005    package org.greatage.tapestry.internal;
006    
007    import org.apache.tapestry5.model.MutableComponentModel;
008    import org.apache.tapestry5.runtime.Component;
009    import org.apache.tapestry5.services.ClassTransformation;
010    import org.apache.tapestry5.services.ComponentClassTransformWorker;
011    import org.apache.tapestry5.services.ComponentInstanceOperation;
012    import org.apache.tapestry5.services.TransformConstants;
013    import org.greatage.security.AccessDeniedException;
014    import org.greatage.security.Authentication;
015    import org.greatage.security.AuthorityConstants;
016    import org.greatage.security.SecurityContext;
017    import org.greatage.security.annotations.Allow;
018    import org.greatage.security.annotations.Deny;
019    import org.greatage.security.annotations.Operation;
020    import org.greatage.util.CollectionUtils;
021    
022    import java.util.Arrays;
023    import java.util.List;
024    
025    /**
026     * @author Ivan Khalopik
027     * @since 1.0
028     */
029    public class SecuredAnnotationWorker implements ComponentClassTransformWorker {
030            private final SecurityContext securityContext;
031    
032            public SecuredAnnotationWorker(final SecurityContext securityContext) {
033                    this.securityContext = securityContext;
034            }
035    
036            public void transform(final ClassTransformation transformation, final MutableComponentModel model) {
037                    final Allow allow = transformation.getAnnotation(Allow.class);
038                    final Deny deny = transformation.getAnnotation(Deny.class);
039                    if (allow != null || deny != null) {
040                            transformation.getOrCreateMethod(TransformConstants.DISPATCH_COMPONENT_EVENT).addOperationBefore(new ComponentInstanceOperation() {
041                                    public void invoke(final Component instance) {
042                                            final List<String> authorities = getAuthorities();
043                                            if (allow != null) {
044                                                    checkAllow(allow, authorities);
045                                            }
046                                            if (deny != null) {
047                                                    checkDeny(deny, authorities);
048                                            }
049                                    }
050                            });
051                    }
052            }
053    
054            private void checkDeny(final Deny deny, final List<String> authorities) {
055                    if (deny.operation() == Operation.AND) {
056                            if (authorities.containsAll(Arrays.asList(deny.value()))) {
057                                    throw new AccessDeniedException(String.format("Access denied for authorities: '%s'", authorities));
058                            }
059                    } else if (deny.operation() == Operation.OR) {
060                            for (String authority : deny.value()) {
061                                    if (authorities.contains(authority)) {
062                                            throw new AccessDeniedException(String.format("Access denied for authorities: '%s'", authorities));
063                                    }
064                            }
065                    }
066            }
067    
068            private void checkAllow(final Allow allow, final List<String> authorities) {
069                    if (allow.operation() == Operation.AND) {
070                            if (!authorities.containsAll(Arrays.asList(allow.value()))) {
071                                    throw new AccessDeniedException(String.format("Access denied for authorities: '%s'", authorities));
072                            }
073                    } else if (allow.operation() == Operation.OR) {
074                            for (String authority : allow.value()) {
075                                    if (authorities.contains(authority)) {
076                                            return;
077                                    }
078                            }
079                            throw new AccessDeniedException(String.format("Access denied for authorities: '%s'", authorities));
080                    }
081            }
082    
083            private List<String> getAuthorities() {
084                    final Authentication user = securityContext.getCurrentUser();
085                    return user != null ? user.getAuthorities() : CollectionUtils.newList(AuthorityConstants.STATUS_ANONYMOUS);
086            }
087    }
