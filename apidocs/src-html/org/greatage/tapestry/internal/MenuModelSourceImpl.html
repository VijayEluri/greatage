005    package org.greatage.tapestry.internal;
006    
007    import org.apache.tapestry5.ComponentResources;
008    import org.apache.tapestry5.ioc.Messages;
009    import org.greatage.tapestry.menu.MenuModel;
010    import org.greatage.tapestry.menu.MenuModelImpl;
011    import org.greatage.tapestry.services.MenuModelSource;
012    
013    import java.util.List;
014    
015    /**
016     * @author Ivan Khalopik
017     */
018    public class MenuModelSourceImpl implements MenuModelSource {
019    
020            public MenuModel createMenuModel(ComponentResources resources, Messages messages, List<String> menuItems, Object... context) {
021                    //todo: defence
022    //              Defense.notNull(resources, "resources");
023    //              Defense.notNull(messages, "messages");
024    //              Defense.notNull(menuItems, "menuItems");
025    //              Defense.notNull(context, "context");
026    
027                    final MenuModel model = new MenuModelImpl(messages);
028                    for (String menuItem : menuItems) {
029                            if (menuItem.startsWith("_")) {
030                                    final String name = menuItem.substring(1);
031                                    model.add(name, resources.createFormEventLink(name, context));
032                            } else {
033                                    model.add(menuItem, resources.createEventLink(menuItem, context));
034                            }
035                    }
036                    return model;
037            }
038    }
