001    package org.greatage.tapestry.internal;
002    
003    import org.apache.tapestry5.services.Request;
004    import org.apache.tapestry5.services.RequestFilter;
005    import org.apache.tapestry5.services.RequestHandler;
006    import org.apache.tapestry5.services.Response;
007    import org.greatage.ioc.scope.ScopeConstants;
008    import org.greatage.ioc.scope.Scope;
009    import org.greatage.ioc.scope.ScopeManager;
010    
011    import java.io.IOException;
012    
013    /**
014     * @author Ivan Khalopik
015     * @since 1.1
016     */
017    public class ThreadCleanupFilter implements RequestFilter {
018            private Scope threadScope;
019    
020            public ThreadCleanupFilter(final ScopeManager scopeManager) {
021                    this.threadScope = scopeManager.getScope(ScopeConstants.THREAD);
022            }
023    
024            public boolean service(final Request request, final Response response, final RequestHandler handler) throws IOException {
025                    try {
026                            return handler.service(request, response);
027                    }
028                    finally {
029                            threadScope.cleanup();
030                    }
031            }
032    }
