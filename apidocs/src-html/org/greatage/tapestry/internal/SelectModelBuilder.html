005    package org.greatage.tapestry.internal;
006    
007    import org.apache.tapestry5.OptionModel;
008    import org.apache.tapestry5.PropertyConduit;
009    import org.apache.tapestry5.SelectModel;
010    import org.apache.tapestry5.internal.SelectModelImpl;
011    import org.apache.tapestry5.ioc.MessageFormatter;
012    import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
013    import org.apache.tapestry5.services.PropertyConduitSource;
014    import org.greatage.tapestry.select.FormattedSelectModel;
015    import org.greatage.tapestry.select.SimpleSelectModel;
016    
017    import java.util.ArrayList;
018    import java.util.Collection;
019    import java.util.List;
020    
021    /**
022     * @author Ivan Khalopik
023     */
024    public class SelectModelBuilder {
025            private static final SelectModel EMPTY_MODEL = new SelectModelImpl(null, CollectionFactory.<OptionModel>newList());
026    
027            private final PropertyConduitSource conduitSource;
028    
029            public SelectModelBuilder(PropertyConduitSource conduitSource) {
030                    this.conduitSource = conduitSource;
031            }
032    
033            public SelectModel buildEmpty() {
034                    return EMPTY_MODEL;
035            }
036    
037            public <V> SelectModel buildSimple(Collection<V> values) {
038                    return new SimpleSelectModel<V>(values);
039            }
040    
041            public <V> SelectModel buildFormatted(Class<V> valueClass, Collection<V> values, String format, String... parameters) {
042                    return new FormattedSelectModel<V>(values, format, getConduits(valueClass, parameters));
043            }
044    
045            public <V> SelectModel buildFormatted(Class<V> valueClass, Collection<V> values, MessageFormatter formatter, String... parameters) {
046                    return new FormattedSelectModel<V>(values, formatter, getConduits(valueClass, parameters));
047            }
048    
049            private <V> List<PropertyConduit> getConduits(Class<V> valueClass, String... parameters) {
050                    final List<PropertyConduit> conduits = new ArrayList<PropertyConduit>();
051                    for (String parameter : parameters) {
052                            final PropertyConduit conduit = conduitSource.create(valueClass, parameter);
053                            conduits.add(conduit);
054                    }
055                    return conduits;
056            }
057    }
