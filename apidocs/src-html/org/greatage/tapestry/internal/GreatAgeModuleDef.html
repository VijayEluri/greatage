005    package org.greatage.tapestry.internal;
006    
007    import org.apache.tapestry5.ioc.def.ContributionDef;
008    import org.apache.tapestry5.ioc.def.DecoratorDef;
009    import org.apache.tapestry5.ioc.def.ModuleDef;
010    import org.apache.tapestry5.ioc.def.ServiceDef;
011    import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
012    import org.greatage.ioc.ServiceLocator;
013    import org.greatage.ioc.ServiceLocatorBuilder;
014    
015    import java.util.Collections;
016    import java.util.Map;
017    import java.util.Set;
018    
019    /**
020     * @author Ivan Khalopik
021     * @since 1.0
022     */
023    public class GreatAgeModuleDef implements ModuleDef {
024            private final Map<String, ServiceDef> services = CollectionFactory.newMap();
025    
026            public GreatAgeModuleDef(final String moduleName) {
027                    try {
028                            final Class<?> moduleClass;
029                            moduleClass = Class.forName(moduleName);
030                            final ServiceLocator locator = ServiceLocatorBuilder.createServiceLocator(moduleClass);
031                            for (String serviceId : locator.getServiceIds()) {
032                                    services.put(serviceId, new GreatAgeServiceDef(locator.getServiceStatus(serviceId)));
033                            }
034                    } catch (ClassNotFoundException e) {
035                            throw new RuntimeException("Couldn't find module class", e);
036                    }
037            }
038    
039            public Set<String> getServiceIds() {
040                    return services.keySet();
041            }
042    
043            public ServiceDef getServiceDef(final String serviceId) {
044                    return services.get(serviceId);
045            }
046    
047            public Set<DecoratorDef> getDecoratorDefs() {
048                    return Collections.emptySet();
049            }
050    
051            public Set<ContributionDef> getContributionDefs() {
052                    return Collections.emptySet();
053            }
054    
055            public Class getBuilderClass() {
056                    return null;
057            }
058    
059            public String getLoggerName() {
060                    return GreatAgeModuleDef.class.getName();
061            }
062    }
