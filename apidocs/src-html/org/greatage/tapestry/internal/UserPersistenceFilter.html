005    package org.greatage.tapestry.internal;
006    
007    import org.apache.tapestry5.services.*;
008    import org.greatage.security.AnonymousAuthenticationToken;
009    import org.greatage.security.Authentication;
010    import org.greatage.security.AuthenticationManager;
011    import org.greatage.security.SecurityContext;
012    
013    import java.io.IOException;
014    
015    /**
016     * @author Ivan Khalopik
017     * @since 1.0
018     */
019    public class UserPersistenceFilter implements RequestFilter {
020            private static final String SECURITY_CONTEXT_KEY = "SECURITY_CONTEXT";
021    
022            private final AuthenticationManager authenticationManager;
023            private final SecurityContext securityContext;
024    
025            public UserPersistenceFilter(final AuthenticationManager authenticationManager, final SecurityContext securityContext) {
026                    this.authenticationManager = authenticationManager;
027                    this.securityContext = securityContext;
028            }
029    
030            @SuppressWarnings({"unchecked"})
031            public boolean service(final Request request, final Response response, final RequestHandler handler) throws IOException {
032                    try {
033                            final Authentication authentication = loadAuthentication(request);
034                            if (authentication != null) {
035                                    securityContext.initCurrentUser(authentication);
036                            } else {
037                                    authenticationManager.signIn(new AnonymousAuthenticationToken());
038                            }
039                            return handler.service(request, response);
040                    } finally {
041                            final Authentication authentication = securityContext.getCurrentUser();
042                            securityContext.clearCurrentUser();
043                            saveAuthentication(request, authentication);
044                    }
045            }
046    
047            private Authentication loadAuthentication(final Request request) {
048                    final Session session = getSession(request, true);
049                    final Object user = session != null ? session.getAttribute(SECURITY_CONTEXT_KEY) : null;
050                    return user != null && user instanceof Authentication ? (Authentication) user : null;
051            }
052    
053            private void saveAuthentication(final Request request, final Authentication user) {
054                    final Session session = getSession(request, true);
055                    session.setAttribute(SECURITY_CONTEXT_KEY, user);
056            }
057    
058            private Session getSession(final Request request, final boolean allowCreate) {
059                    final Session session = request.getSession(false);
060                    return session == null && allowCreate ? request.getSession(true) : session;
061            }
062    }
