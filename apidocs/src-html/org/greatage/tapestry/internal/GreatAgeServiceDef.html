005    package org.greatage.tapestry.internal;
006    
007    import org.apache.tapestry5.ioc.ObjectCreator;
008    import org.apache.tapestry5.ioc.ScopeConstants;
009    import org.apache.tapestry5.ioc.ServiceBuilderResources;
010    import org.apache.tapestry5.ioc.def.ServiceDef;
011    import org.greatage.ioc.ServiceStatus;
012    
013    import java.util.Collections;
014    import java.util.Set;
015    
016    /**
017     * @author Ivan Khalopik
018     * @since 1.0
019     */
020    public class GreatAgeServiceDef implements ServiceDef {
021            private final ServiceStatus service;
022    
023            public GreatAgeServiceDef(final ServiceStatus service) {
024                    this.service = service;
025            }
026    
027            public ObjectCreator createServiceCreator(final ServiceBuilderResources resources) {
028                    return new ObjectCreator() {
029                            public Object createObject() {
030                                    return service.getService();
031                            }
032                    };
033            }
034    
035            public String getServiceId() {
036                    return service.getServiceId();
037            }
038    
039            public Set<Class> getMarkers() {
040                    return Collections.emptySet();
041            }
042    
043            public Class getServiceInterface() {
044                    return service.getServiceClass();
045            }
046    
047            public String getServiceScope() {
048                    return ScopeConstants.DEFAULT;
049            }
050    
051            public boolean isEagerLoad() {
052                    return false;
053            }
054    }
