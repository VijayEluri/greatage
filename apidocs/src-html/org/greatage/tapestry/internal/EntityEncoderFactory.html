005    package org.greatage.tapestry.internal;
006    
007    import org.apache.tapestry5.ValueEncoder;
008    import org.apache.tapestry5.ioc.services.TypeCoercer;
009    import org.apache.tapestry5.services.ValueEncoderFactory;
010    import org.greatage.domain.Entity;
011    import org.greatage.domain.EntityRepository;
012    
013    import java.io.Serializable;
014    
015    /**
016     * Needs ga:ga-core dependency.
017     *
018     * @author Ivan Khalopik
019     */
020    public class EntityEncoderFactory<PK extends Serializable> implements ValueEncoderFactory<Entity<PK>> {
021            private final TypeCoercer typeCoercer;
022            private final EntityRepository repository;
023            private final Class<PK> pkClass;
024    
025            public EntityEncoderFactory(TypeCoercer typeCoercer, EntityRepository repository, Class<PK> pkClass) {
026                    this.repository = repository; //todo: defence
027                    this.typeCoercer = typeCoercer; //todo: defence
028                    this.pkClass = pkClass; //todo: defence
029            }
030    
031            public ValueEncoder<Entity<PK>> create(final Class<Entity<PK>> type) {
032                    return new ValueEncoder<Entity<PK>>() {
033                            public String toClient(Entity<PK> value) {
034                                    return value != null ? typeCoercer.coerce(value.getId(), String.class) : null;
035                            }
036    
037                            public Entity<PK> toValue(String clientValue) {
038                                    final PK pk = typeCoercer.coerce(clientValue, pkClass);
039                                    return pk != null ? repository.get(type, pk) : null;
040                            }
041    
042                            @Override
043                            public String toString() {
044                                    final StringBuilder sb = new StringBuilder("EntityValueEncoder(");
045                                    sb.append("class=").append(type);
046                                    sb.append(")");
047                                    return sb.toString();
048                            }
049                    };
050            }
051    
052            @Override
053            public String toString() {
054                    final StringBuilder sb = new StringBuilder("EntityEncoderFactory(");
055                    sb.append("pkClass=").append(pkClass);
056                    sb.append(")");
057                    return sb.toString();
058            }
059    }
