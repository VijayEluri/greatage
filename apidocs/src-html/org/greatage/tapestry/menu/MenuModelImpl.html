005    package org.greatage.tapestry.menu;
006    
007    import org.apache.tapestry5.Link;
008    import org.apache.tapestry5.ioc.Messages;
009    import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
010    
011    import java.util.List;
012    import java.util.Map;
013    
014    /**
015     * @author Ivan Khalopik
016     */
017    public class MenuModelImpl implements MenuModel {
018            private final List<String> menuItems = CollectionFactory.newList();
019            private final Map<String, MenuItemModel> items = CollectionFactory.newMap();
020            private final Messages messages;
021    
022            public MenuModelImpl(Messages messages) {
023                    this.messages = messages;
024            }
025    
026            public List<String> getNames() {
027                    return menuItems;
028            }
029    
030            public MenuItemModel getByName(String name) {
031                    return items.get(name);
032            }
033    
034            public MenuItemModel add(String name) {
035                    return add(name, null);
036            }
037    
038            public MenuItemModel add(String name, Link link) {
039                    if (!items.containsKey(name)) {
040                            final MenuItemModel menuItemModel = new MenuItemModelImpl(this, name, messages).link(link);
041                            items.put(name, menuItemModel);
042                            menuItems.add(name);
043                    }
044                    return items.get(name);
045            }
046    }
