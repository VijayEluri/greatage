005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.BindingConstants;
008    import org.apache.tapestry5.annotations.Component;
009    import org.apache.tapestry5.annotations.Parameter;
010    import org.apache.tapestry5.annotations.Property;
011    import org.apache.tapestry5.annotations.SupportsInformalParameters;
012    import org.greatage.tapestry.CSSConstants;
013    import org.greatage.tapestry.commonlib.base.components.AbstractGridComponent;
014    
015    import java.util.ArrayList;
016    import java.util.List;
017    
018    /**
019     * @author Ivan Khalopik
020     */
021    @SupportsInformalParameters
022    public class GridSizer extends AbstractGridComponent {
023    
024            @Parameter(value = "2,10,50", defaultPrefix = BindingConstants.LITERAL)
025            private String pages;
026    
027            private List<Integer> pageSizes;
028    
029            @Property
030            private int pageSize;
031    
032            @Component(
033                            parameters = {
034                                            "overrides=overrides",
035                                            "class=prop:sizerClass"},
036                            inheritInformalParameters = true)
037            private Menu menu;
038    
039            @Component(
040                            parameters = {
041                                            "overrides=overrides",
042                                            "selected=current"})
043            private MenuItem menuItem;
044    
045            @Component(
046                            parameters = {
047                                            "event=changePageSize",
048                                            "context=pageSize",
049                                            "disabled=current"})
050            private GridLink pageSizeLink;
051    
052            public List<Integer> getPageSizes() {
053                    if (pageSizes == null) {
054                            pageSizes = createPageSizes();
055                    }
056                    return pageSizes;
057            }
058    
059            public boolean isCurrent() {
060                    return pageSize == getPaginationModel().getPageSize();
061            }
062    
063            public String getSizerClass() {
064                    return CSSConstants.TOOLBAR_MENU_CLASS;
065            }
066    
067            protected List<Integer> createPageSizes() {
068                    final List<Integer> result = new ArrayList<Integer>();
069                    final String[] elements = pages.split(",");
070                    for (String element : elements) {
071                            result.add(Integer.parseInt(element));
072                    }
073                    return result;
074            }
075    }
