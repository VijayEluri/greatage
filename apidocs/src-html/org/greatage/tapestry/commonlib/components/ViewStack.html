005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.Block;
008    import org.apache.tapestry5.annotations.Parameter;
009    import org.apache.tapestry5.annotations.SupportsInformalParameters;
010    import org.greatage.tapestry.commonlib.base.components.AbstractComponent;
011    
012    /**
013     * @author Ivan Khalopik
014     */
015    @SupportsInformalParameters
016    public class ViewStack extends AbstractComponent {
017    
018            @Parameter(required = true)
019            private String state;
020    
021            protected Object beginRender() {
022                    if (state == null) {
023                            return getResources().getBody();
024                    }
025                    final Block block = getOverrides().getOverrideBlock(state);
026                    return block != null ? block : getResources().getBody();
027            }
028    
029            protected boolean beforeRenderBody() {
030                    return false;
031            }
032    }
