005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.BindingConstants;
008    import org.apache.tapestry5.SymbolConstants;
009    import org.apache.tapestry5.annotations.Component;
010    import org.apache.tapestry5.annotations.Import;
011    import org.apache.tapestry5.annotations.Parameter;
012    import org.apache.tapestry5.corelib.components.EventLink;
013    import org.apache.tapestry5.internal.TapestryInternalUtils;
014    import org.apache.tapestry5.ioc.annotations.Inject;
015    import org.apache.tapestry5.ioc.annotations.Symbol;
016    import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
017    import org.apache.tapestry5.services.LocalizationSetter;
018    import org.greatage.tapestry.CSSConstants;
019    
020    import java.util.Arrays;
021    import java.util.List;
022    import java.util.Locale;
023    
024    /**
025     * @author Ivan Khalopik
026     */
027    @Import(stylesheet = "flags.css")
028    public class LocaleMenu {
029    
030            @Parameter(defaultPrefix = BindingConstants.LITERAL, allowNull = false)
031            private String locales;
032    
033            private List<String> localeNames;
034            private String localeName;
035    
036            @Inject
037            private Locale currentLocale;
038    
039            @Inject
040            private LocalizationSetter localizationSetter;
041    
042            @Inject
043            @Symbol(SymbolConstants.SUPPORTED_LOCALES)
044            private String supportedLocales;
045    
046            @Component(parameters = "class=prop:localeMenuClass")
047            private Menu menu;
048    
049            @Component(
050                            parameters = {
051                                            "selected=currentLocale",
052                                            "name=prop:localeName"})
053            private MenuItem menuItem;
054    
055            @Component(
056                            parameters = {
057                                            "event=changeLocale",
058                                            "context=localeName",
059                                            "class=prop:linkClass"})
060            private EventLink localeLink;
061    
062            protected String defaultLocales() {
063                    return supportedLocales;
064            }
065    
066            public String getLocaleMenuClass() {
067                    return CSSConstants.LOCALE_MENU_CLASS;
068            }
069    
070            public String getLinkClass() {
071                    final List<String> classes = CollectionFactory.newList();
072                    classes.add(CSSConstants.FLAG_CLASS);
073                    classes.add(localeName);
074                    return TapestryInternalUtils.toClassAttributeValue(classes);
075            }
076    
077            public List<String> getLocaleNames() {
078                    if (localeNames == null) {
079                            localeNames = Arrays.asList(locales.split(","));
080                    }
081                    return localeNames;
082            }
083    
084            public String getLocaleName() {
085                    return localeName;
086            }
087    
088            public void setLocaleName(String localeName) {
089                    this.localeName = localeName;
090            }
091    
092            public boolean isCurrentLocale() {
093                    return String.valueOf(currentLocale).equals(localeName);
094            }
095    
096            protected void onChangeLocale(String locale) {
097                    localizationSetter.setLocaleFromLocaleName(locale);
098            }
099    }
