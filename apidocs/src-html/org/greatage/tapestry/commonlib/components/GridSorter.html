005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.annotations.Component;
008    import org.apache.tapestry5.annotations.Property;
009    import org.apache.tapestry5.beaneditor.PropertyModel;
010    import org.apache.tapestry5.grid.ColumnSort;
011    import org.greatage.tapestry.CSSConstants;
012    import org.greatage.tapestry.commonlib.base.components.AbstractGridComponent;
013    
014    import java.util.ArrayList;
015    import java.util.List;
016    
017    /**
018     * @author Ivan Khalopik
019     */
020    public class GridSorter extends AbstractGridComponent {
021    
022            private List<String> sortColumns;
023    
024            @Property
025            private String sortColumn;
026    
027            @Component(
028                            parameters = {
029                                            "overrides=overrides",
030                                            "class=prop:sorterClass"},
031                            inheritInformalParameters = true)
032            private Menu menu;
033    
034            @Component(
035                            parameters = {
036                                            "overrides=overrides",
037                                            "selected=current"})
038            private MenuItem menuItem;
039    
040            @Component(
041                            parameters = {
042                                            "event=sort",
043                                            "context=sortColumn"})
044            private GridLink sortLink;
045    
046            public List<String> getSortColumns() {
047                    if (sortColumns == null) {
048                            sortColumns = createSortColumns();
049                    }
050                    return sortColumns;
051            }
052    
053            public boolean isCurrent() {
054                    return getSortModel().getColumnSort(sortColumn) != ColumnSort.UNSORTED;
055            }
056    
057            public String getSorterClass() {
058                    return CSSConstants.TOOLBAR_MENU_CLASS;
059            }
060    
061            @SuppressWarnings({"unchecked"})
062            private List<String> createSortColumns() {
063                    final List<String> result = new ArrayList<String>();
064                    final List<String> propertyNames = getDataModel().getPropertyNames();
065                    for (String name : propertyNames) {
066                            final PropertyModel propertyModel = getDataModel().get(name);
067                            if (propertyModel.isSortable()) {
068                                    result.add(propertyModel.getPropertyName());
069                            }
070                    }
071                    return result;
072            }
073    }
