005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.Asset;
008    import org.apache.tapestry5.BindingConstants;
009    import org.apache.tapestry5.RenderSupport;
010    import org.apache.tapestry5.annotations.Import;
011    import org.apache.tapestry5.annotations.Parameter;
012    import org.apache.tapestry5.ioc.annotations.Inject;
013    import org.apache.tapestry5.ioc.annotations.Symbol;
014    import org.apache.tapestry5.ioc.annotations.Value;
015    import org.apache.tapestry5.services.AssetSource;
016    import org.greatage.tapestry.CommonSymbols;
017    
018    /**
019     * @author Ivan Khalopik
020     */
021    @Import(stylesheet = {
022                    "layout.css",
023                    "reset.css",
024                    "base.css",
025                    "fonts.css"
026    })
027    public class Layout {
028    
029            @Parameter(defaultPrefix = BindingConstants.LITERAL, allowNull = false)
030            private String theme;
031    
032            @Inject
033            private RenderSupport renderSupport;
034    
035            @Inject
036            @Symbol(CommonSymbols.THEME)
037            private String defaultTheme;
038    
039            @Inject
040            @Value("classpath:${" + CommonSymbols.THEME_PATH + "}")
041            private String themePath;
042    
043            @Inject
044            private AssetSource assetSource;
045    
046            protected String defaultTheme() {
047                    return defaultTheme;
048            }
049    
050            protected void afterRender() {
051                    final Asset themeAsset = assetSource.getAsset(null, themePath + "/" + theme + ".css", null);
052                    renderSupport.addStylesheetLink(themeAsset, null);
053            }
054    }
