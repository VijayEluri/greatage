001    package org.greatage.tapestry.commonlib.components;
002    
003    import org.apache.tapestry5.BindingConstants;
004    import org.apache.tapestry5.annotations.Parameter;
005    import org.apache.tapestry5.ioc.annotations.Inject;
006    import org.greatage.security.Authentication;
007    import org.greatage.security.AuthorityConstants;
008    import org.greatage.security.SecurityContext;
009    import org.greatage.tapestry.commonlib.base.components.AbstractComponent;
010    import org.greatage.util.CollectionUtils;
011    
012    import java.util.List;
013    
014    /**
015     * @author Ivan Khalopik
016     * @since 1.0
017     */
018    public class Secured extends AbstractComponent {
019    
020            @Parameter(defaultPrefix = BindingConstants.LITERAL)
021            private String allow;
022    
023            @Parameter(defaultPrefix = BindingConstants.LITERAL)
024            private String deny;
025    
026            @Inject
027            private SecurityContext securityContext;
028    
029            @SuppressWarnings({"RedundantIfStatement"})
030            boolean setupRender() {
031                    final List<String> authorities = getAuthorities();
032                    if (allow != null && !authorities.contains(allow)) {
033                            return false;
034                    }
035                    if (deny != null && authorities.contains(deny)) {
036                            return false;
037                    }
038                    return true;
039            }
040    
041            private List<String> getAuthorities() {
042                    final Authentication user = securityContext.getCurrentUser();
043                    return user != null ? user.getAuthorities() : CollectionUtils.newList(AuthorityConstants.STATUS_ANONYMOUS);
044            }
045    
046    }
