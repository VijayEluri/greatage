005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.ComponentAction;
008    import org.apache.tapestry5.MarkupWriter;
009    import org.apache.tapestry5.ValueEncoder;
010    import org.apache.tapestry5.annotations.Mixin;
011    import org.apache.tapestry5.annotations.Parameter;
012    import org.apache.tapestry5.annotations.SupportsInformalParameters;
013    import org.apache.tapestry5.internal.TapestryInternalUtils;
014    import org.apache.tapestry5.ioc.annotations.Inject;
015    import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
016    import org.apache.tapestry5.services.ComponentDefaultProvider;
017    import org.greatage.tapestry.CSSConstants;
018    import org.greatage.tapestry.commonlib.base.components.AbstractGridComponent;
019    import org.greatage.tapestry.commonlib.mixins.DefineClass;
020    import org.greatage.tapestry.grid.PaginationModel;
021    import org.greatage.tapestry.grid.TableRowContext;
022    
023    import java.util.List;
024    
025    /**
026     * @author Ivan Khalopik
027     */
028    @SupportsInformalParameters
029    public class GridTable extends AbstractGridComponent implements TableRowContext {
030    
031            @Parameter(cache = false)
032            private String rowClass;
033    
034            @Parameter
035            private Object row;
036    
037            @Parameter
038            private int rowIndex;
039    
040            @Parameter(name = "volatile")
041            private boolean volatileState;
042    
043            @Parameter
044            private ValueEncoder encoder;
045    
046            @Inject
047            private ComponentDefaultProvider defaultProvider;
048    
049            @Mixin
050            private DefineClass defineClass;
051    
052            protected ValueEncoder defaultEncoder() {
053                    return defaultProvider.defaultValueEncoder("row", getResources());
054            }
055    
056            public Object getRow() {
057                    return row;
058            }
059    
060            public boolean isHeader() {
061                    return rowIndex < 0;
062            }
063    
064            public String getRowClass() {
065                    final List<String> classes = CollectionFactory.newList();
066                    if (rowClass != null) {
067                            classes.add(rowClass);
068                    }
069                    if (rowIndex == getPaginationModel().getStartIndex()) {
070                            classes.add(CSSConstants.GRID_FIRST_CLASS);
071                    }
072                    if (rowIndex == getPaginationModel().getEndIndex()) {
073                            classes.add(CSSConstants.GRID_LAST_CLASS);
074                    }
075                    return TapestryInternalUtils.toClassAttributeValue(classes);
076            }
077    
078            protected void setupRender() {
079                    rowIndex = -1;
080                    storeAndExecute(this, new SetupDataSource());
081            }
082    
083            protected void beginRender(MarkupWriter writer) {
084                    writer.element("table", "class", CSSConstants.GRID_CLASS);
085                    getResources().renderInformalParameters(writer);
086            }
087    
088            protected void afterRender(MarkupWriter writer) {
089                    writer.end();
090            }
091    
092            @SuppressWarnings({"unchecked"})
093            protected boolean beforeRenderBody(MarkupWriter writer) {
094                    if (rowIndex < 0) {
095                            writer.element("thead");
096                    } else {
097                            if (getPaginationModel().getSize() <= 0) return false;
098    
099                            if (rowIndex == getPaginationModel().getStartIndex()) {
100                                    writer.element("tbody");
101                            }
102                            if (volatileState) {
103                                    if (encoder == null) {
104                                            getFormSupport().store(this, new SetupRow(rowIndex));
105                                    } else {
106                                            final String key = encoder.toClient(row);
107                                            getFormSupport().store(this, new SetupEncodedRow(key));
108                                    }
109                            }
110                            setupRow(rowIndex);
111                    }
112                    writer.element("tr", "class", getRowClass());
113                    storeAndExecute(this, new SetupRowContext());
114                    return true;
115            }
116    
117            protected boolean afterRenderBody(MarkupWriter writer) {
118                    if (rowIndex >= 0 && getPaginationModel().getSize() <= 0) return true;
119    
120                    storeAndExecute(this, new CleanupRowContext());
121                    writer.end();
122                    if (rowIndex < 0) {
123                            rowIndex = getPaginationModel().getStartIndex();
124                            writer.end();
125                    } else if (rowIndex == getPaginationModel().getEndIndex()) {
126                            writer.end();
127                            return true;
128                    } else {
129                            rowIndex++;
130                    }
131                    return false;
132            }
133    
134            protected void setupDataSource() {
135                    final PaginationModel pagination = getPaginationModel();
136                    getDataSource().prepare(pagination.getStartIndex(), pagination.getEndIndex(), getSortModel().getSortConstraints());
137            }
138    
139            private void setupRow(int rowIndex) {
140                    row = getDataSource().getRowValue(rowIndex);
141            }
142    
143            private void setupEncodedRow(String clientValue) {
144                    row = encoder.toValue(clientValue);
145                    if (row == null) {
146                            throw new IllegalArgumentException(
147                                            String.format("%s returned null for client value '%s'.", encoder, clientValue));
148                    }
149            }
150    
151            protected void setupRowContext() {
152                    pushToEnvironment(TableRowContext.class, this);
153            }
154    
155            protected void cleanupRowContext() {
156                    popFromEnvironment(TableRowContext.class);
157            }
158    
159            static class SetupDataSource implements ComponentAction<GridTable> {
160                    public void execute(GridTable component) {
161                            component.setupDataSource();
162                    }
163            }
164    
165            static class SetupRow implements ComponentAction<GridTable> {
166                    private final int rowIndex;
167    
168                    public SetupRow(int rowIndex) {
169                            this.rowIndex = rowIndex;
170                    }
171    
172                    public void execute(GridTable component) {
173                            component.setupRow(rowIndex);
174                    }
175            }
176    
177            static class SetupEncodedRow implements ComponentAction<GridTable> {
178                    private final String clientValue;
179    
180                    public SetupEncodedRow(String clientValue) {
181                            this.clientValue = clientValue;
182                    }
183    
184                    public void execute(GridTable component) {
185                            component.setupEncodedRow(clientValue);
186                    }
187            }
188    
189            static class SetupRowContext implements ComponentAction<GridTable> {
190                    public void execute(GridTable component) {
191                            component.setupRowContext();
192                    }
193            }
194    
195            static class CleanupRowContext implements ComponentAction<GridTable> {
196                    public void execute(GridTable component) {
197                            component.cleanupRowContext();
198                    }
199            }
200    }
