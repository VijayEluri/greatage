005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.ComponentAction;
008    import org.apache.tapestry5.MarkupWriter;
009    import org.apache.tapestry5.annotations.Mixin;
010    import org.apache.tapestry5.annotations.Parameter;
011    import org.apache.tapestry5.annotations.SupportsInformalParameters;
012    import org.apache.tapestry5.beaneditor.BeanModel;
013    import org.apache.tapestry5.ioc.annotations.Inject;
014    import org.apache.tapestry5.services.BeanModelSource;
015    import org.greatage.tapestry.CSSConstants;
016    import org.greatage.tapestry.commonlib.base.components.AbstractComponent;
017    import org.greatage.tapestry.commonlib.mixins.DefineClass;
018    import org.greatage.tapestry.details.DetailsModel;
019    
020    /**
021     * @author Ivan Khalopik
022     */
023    @SupportsInformalParameters
024    public class Details extends AbstractComponent implements DetailsModel {
025            @Parameter(required = true)
026            private Object object;
027    
028            @Parameter
029            private boolean editable;
030    
031            @Inject
032            private BeanModelSource modelSource;
033    
034            @Mixin
035            private DefineClass defineClass;
036    
037            private BeanModel dataModel;
038    
039            public BeanModel getDataModel() {
040                    return dataModel;
041            }
042    
043            public Object getObject() {
044                    return object;
045            }
046    
047            public boolean isEditable() {
048                    return editable;
049            }
050    
051            protected boolean beforeRenderBody() {
052                    return object != null;
053            }
054    
055            protected void setupRender() {
056                    storeAndExecute(this, new SetupModel());
057            }
058    
059            protected void cleanupRender() {
060                    storeAndExecute(this, new CleanupModel());
061            }
062    
063            protected void beginRender(MarkupWriter writer) {
064                    writer.element("div", "class", CSSConstants.DETAILS_CLASS);
065                    getResources().renderInformalParameters(writer);
066            }
067    
068            protected void afterRender(MarkupWriter writer) {
069                    writer.end();
070            }
071    
072            @SuppressWarnings({"unchecked"})
073            protected void setupModel() {
074                    Class objectType = getResources().getBoundType("object");
075                    dataModel = isEditable() ?
076                                    modelSource.createEditModel(objectType, getOverrides().getOverrideMessages()) :
077                                    modelSource.createDisplayModel(objectType, getOverrides().getOverrideMessages());
078                    pushToEnvironment(DetailsModel.class, this);
079            }
080    
081            protected void cleanupModel() {
082                    popFromEnvironment(DetailsModel.class);
083            }
084    
085            static class SetupModel implements ComponentAction<Details> {
086                    public void execute(Details component) {
087                            component.setupModel();
088                    }
089            }
090    
091            static class CleanupModel implements ComponentAction<Details> {
092                    public void execute(Details component) {
093                            component.cleanupModel();
094                    }
095            }
096    }
