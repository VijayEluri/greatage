005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.BindingConstants;
008    import org.apache.tapestry5.annotations.Component;
009    import org.apache.tapestry5.annotations.Import;
010    import org.apache.tapestry5.annotations.Parameter;
011    import org.apache.tapestry5.annotations.SupportsInformalParameters;
012    import org.apache.tapestry5.corelib.components.Any;
013    import org.greatage.tapestry.CSSConstants;
014    import org.greatage.tapestry.commonlib.base.components.AbstractMenuComponent;
015    
016    /**
017     * @author Ivan Khalopik
018     */
019    @Import(library = "menu.js")
020    @SupportsInformalParameters
021    public class MenuButton extends AbstractMenuComponent {
022    
023            @Parameter(required = true, defaultPrefix = BindingConstants.LITERAL)
024            private String label;
025    
026            @Component(
027                            parameters = {
028                                            "id=prop:triggerId",
029                                            "class=prop:menuButtonClass"},
030                            inheritInformalParameters = true)
031            private Any menuButton;
032    
033            @Override
034            public String getMenuClass() {
035                    return CSSConstants.POPUP_MENU_CLASS;
036            }
037    
038            public String getMenuButtonClass() {
039                    return CSSConstants.MENU_BUTTON_CLASS;
040            }
041    
042            public String getLabel() {
043                    return label;
044            }
045    }
