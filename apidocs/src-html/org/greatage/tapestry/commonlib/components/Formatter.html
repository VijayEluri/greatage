005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.BindingConstants;
008    import org.apache.tapestry5.MarkupWriter;
009    import org.apache.tapestry5.annotations.Parameter;
010    import org.apache.tapestry5.ioc.Messages;
011    import org.greatage.tapestry.commonlib.base.components.AbstractComponent;
012    
013    /**
014     * @author Ivan Khalopik
015     */
016    public class Formatter extends AbstractComponent {
017            @Parameter(required = true, defaultPrefix = BindingConstants.LITERAL)
018            private String format;
019    
020            @Parameter
021            private Object[] context;
022    
023            @Parameter
024            private boolean raw;
025    
026            @Parameter(value = "overrideMessages")
027            private Messages messages;
028    
029            protected boolean beginRender(MarkupWriter writer) {
030                    final String message = context != null ? messages.format(format, context) : messages.get(format);
031                    if (raw) {
032                            writer.writeRaw(message);
033                    } else {
034                            writer.write(message);
035                    }
036                    return false;
037            }
038    
039            final void inject(Messages messages, String format, boolean raw, Object... context) {
040                    this.messages = messages;
041                    this.format = format;
042                    this.raw = raw;
043                    this.context = context;
044            }
045    }
