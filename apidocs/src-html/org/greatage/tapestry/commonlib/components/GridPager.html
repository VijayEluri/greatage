005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.annotations.Component;
008    import org.apache.tapestry5.annotations.Parameter;
009    import org.apache.tapestry5.annotations.SupportsInformalParameters;
010    import org.greatage.tapestry.CSSConstants;
011    import org.greatage.tapestry.commonlib.base.components.AbstractGridComponent;
012    import org.greatage.tapestry.grid.CenteredPagesBuilder;
013    import org.greatage.tapestry.grid.Page;
014    import org.greatage.tapestry.grid.PagesBuilder;
015    import org.greatage.tapestry.grid.PaginationModel;
016    
017    import java.util.List;
018    
019    /**
020     * @author Ivan Khalopik
021     */
022    @SupportsInformalParameters
023    public class GridPager extends AbstractGridComponent {
024    
025            @Parameter
026            private PagesBuilder builder;
027    
028            private List<Page> pages;
029    
030            private Page page;
031    
032            @Component(
033                            parameters = {
034                                            "overrides=overrides",
035                                            "class=prop:pagerClass"},
036                            inheritInformalParameters = true)
037            private Menu menu;
038    
039            @Component(
040                            parameters = {
041                                            "overrides=overrides",
042                                            "selected=currentPage"})
043            private MenuItem menuItem;
044    
045            @Component(
046                            parameters = {
047                                            "event=changePage",
048                                            "context=value",
049                                            "disabled=disabled"})
050            private GridLink pageLink;
051    
052            protected PagesBuilder defaultBuilder() {
053                    return new CenteredPagesBuilder();
054            }
055    
056            public List<Page> getPages() {
057                    if (pages == null) {
058                            pages = builder.build(getPaginationModel());
059                    }
060                    return pages;
061            }
062    
063            public Page getPage() {
064                    return page;
065            }
066    
067            public void setPage(Page page) {
068                    this.page = page;
069            }
070    
071            public String getPagerClass() {
072                    return CSSConstants.TOOLBAR_MENU_CLASS;
073            }
074    
075            public String getLabel() {
076                    return getMessages().format(page.getType().name(), page.getValue(getPaginationModel()));
077            }
078    
079            public int getValue() {
080                    return page.getValue(getPaginationModel());
081            }
082    
083            public boolean isDisabled() {
084                    return page.isDisabled(getPaginationModel());
085            }
086    
087            public boolean isCurrentPage() {
088                    final PaginationModel model = getPaginationModel();
089                    return page.getType() == Page.PageType.PAGE && page.getValue(model) == model.getCurrentPage();
090            }
091    }
