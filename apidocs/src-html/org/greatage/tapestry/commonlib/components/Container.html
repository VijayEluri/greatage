005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.BindingConstants;
008    import org.apache.tapestry5.MarkupWriter;
009    import org.apache.tapestry5.annotations.Parameter;
010    import org.apache.tapestry5.annotations.SupportsInformalParameters;
011    import org.apache.tapestry5.internal.TapestryInternalUtils;
012    import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
013    import org.greatage.tapestry.CSSConstants;
014    import org.greatage.tapestry.commonlib.base.components.AbstractComponent;
015    
016    import java.util.List;
017    
018    /**
019     * @author Ivan Khalopik
020     */
021    @SupportsInformalParameters
022    public class Container extends AbstractComponent {
023    
024            @Parameter(defaultPrefix = BindingConstants.LITERAL)
025            private String element;
026    
027            @Parameter(name = "class", defaultPrefix = BindingConstants.LITERAL)
028            private String className;
029    
030            protected String defaultElement() {
031                    return getResources().getElementName("div");
032            }
033    
034            public String getClassName() {
035                    final List<String> classes = CollectionFactory.newList();
036                    classes.add(CSSConstants.CONTAINER_CLASS);
037                    if (className != null) {
038                            classes.add(className);
039                    }
040                    return TapestryInternalUtils.toClassAttributeValue(classes);
041            }
042    
043            protected void beginRender(MarkupWriter writer) {
044                    writer.element(element, "class", getClassName());
045                    getResources().renderInformalParameters(writer);
046            }
047    
048            protected void afterRender(MarkupWriter writer) {
049                    writer.end();
050            }
051    }
