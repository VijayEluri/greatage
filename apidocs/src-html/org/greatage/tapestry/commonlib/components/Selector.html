005    package org.greatage.tapestry.commonlib.components;
006    
007    import org.apache.tapestry5.BindingConstants;
008    import org.apache.tapestry5.ValueEncoder;
009    import org.apache.tapestry5.annotations.Environmental;
010    import org.apache.tapestry5.annotations.Import;
011    import org.apache.tapestry5.annotations.Parameter;
012    import org.apache.tapestry5.annotations.Property;
013    import org.apache.tapestry5.internal.TapestryInternalUtils;
014    import org.apache.tapestry5.ioc.annotations.Inject;
015    import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
016    import org.apache.tapestry5.json.JSONObject;
017    import org.apache.tapestry5.services.ComponentDefaultProvider;
018    import org.apache.tapestry5.services.FormSupport;
019    import org.greatage.tapestry.CSSConstants;
020    import org.greatage.tapestry.commonlib.base.components.AbstractComponent;
021    
022    import java.util.HashMap;
023    import java.util.List;
024    import java.util.Map;
025    
026    /**
027     * @author Ivan Khalopik
028     * @since 1.0
029     */
030    @Import(library = "selector.js")
031    public class Selector extends AbstractComponent {
032    
033            @Parameter
034            private Iterable<?> source;
035    
036            @Parameter
037            private Object value;
038    
039            @Parameter(name = "class", defaultPrefix = BindingConstants.LITERAL)
040            private String className;
041    
042            @Property
043            @Parameter
044            private Object row;
045    
046            @Parameter(required = true)
047            private ValueEncoder encoder;
048    
049            @Environmental(false)
050            private FormSupport formSupport;
051    
052            @Inject
053            private ComponentDefaultProvider defaultProvider;
054    
055            private Map<Object, String> itemIds;
056            private Map<Object, String> itemValues;
057    
058            private String clientId;
059            private String previousId;
060            private String nextId;
061    
062            ValueEncoder defaultEncoder() {
063                    return defaultProvider.defaultValueEncoder("value", getResources());
064            }
065    
066            public String getSelectorClass() {
067                    final List<String> classes = CollectionFactory.newList();
068                    classes.add(CSSConstants.CONTAINER_CLASS);
069                    classes.add(CSSConstants.SELECTOR_CLASS);
070                    if (className != null) {
071                            classes.add(className);
072                    }
073                    return TapestryInternalUtils.toClassAttributeValue(classes);
074            }
075    
076            public boolean isVolatile() {
077                    return formSupport != null;
078            }
079    
080            public String getClientId() {
081                    return clientId;
082            }
083    
084            public String getPreviousId() {
085                    return previousId;
086            }
087    
088            public String getPreviousClass() {
089                    return CSSConstants.SELECTOR_PREVIOUS_CLASS;
090            }
091    
092            public String getNextId() {
093                    return nextId;
094            }
095    
096            public String getNextClass() {
097                    return CSSConstants.SELECTOR_NEXT_CLASS;
098            }
099    
100            public String getRowId() {
101                    return itemIds.get(row);
102            }
103    
104            public String getRowClass() {
105                    return CSSConstants.SELECTOR_ITEM_CLASS + (row.equals(value) ? " " : " invisible");
106            }
107    
108            void setupRender() {
109                    itemIds = new HashMap<Object, String>();
110                    itemValues = new HashMap<Object, String>();
111                    for (Object o : source) {
112                            final String itemId = allocateClientId("item");
113                            itemIds.put(o, itemId);
114                            itemValues.put(o, encoder.toClient(o));
115                    }
116                    if (value == null && source.iterator().hasNext()) {
117                            value = source.iterator().next();
118                    }
119    
120                    clientId = allocateClientId();
121                    previousId = allocateClientId("previous");
122                    nextId = allocateClientId("next");
123            }
124    
125            void afterRender() {
126                    final JSONObject spec = new JSONObject("previousId", previousId, "nextId", nextId, "hiddenId", clientId);
127                    spec.put("selectedId", itemIds.get(value));
128                    spec.put("items", generateItems());
129                    getJavaScriptSupport().addInitializerCall("selector", spec);
130            }
131    
132            private JSONObject generateItems() {
133                    final JSONObject result = new JSONObject();
134                    for (Object item : source) {
135                            result.put(itemIds.get(item), itemValues.get(item));
136                    }
137                    return result;
138            }
139    }
