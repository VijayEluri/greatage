005    package org.greatage.tapestry.commonlib.mixins;
006    
007    import org.apache.tapestry5.BindingConstants;
008    import org.apache.tapestry5.ClientElement;
009    import org.apache.tapestry5.MarkupWriter;
010    import org.apache.tapestry5.RenderSupport;
011    import org.apache.tapestry5.annotations.Environmental;
012    import org.apache.tapestry5.annotations.Import;
013    import org.apache.tapestry5.annotations.InjectContainer;
014    import org.apache.tapestry5.annotations.Parameter;
015    import org.apache.tapestry5.ioc.annotations.Inject;
016    import org.apache.tapestry5.services.FormSupport;
017    
018    /**
019     * @author Ivan Khalopik
020     */
021    @Import(library = "formsubmit.js")
022    public class FormSubmit {
023    
024            @Parameter(defaultPrefix = BindingConstants.LITERAL, value = "change")
025            private String submitEvent;
026    
027            @InjectContainer
028            private ClientElement container;
029    
030            @Environmental
031            private FormSupport formSupport;
032    
033            @Inject
034            private RenderSupport renderSupport;
035    
036            void afterRender(MarkupWriter writer) {
037                    renderSupport.addInit("formSubmit", formSupport.getClientId(), container.getClientId(), submitEvent);
038            }
039    
040    }
