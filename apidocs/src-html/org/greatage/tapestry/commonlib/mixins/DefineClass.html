005    package org.greatage.tapestry.commonlib.mixins;
006    
007    import org.apache.tapestry5.BindingConstants;
008    import org.apache.tapestry5.MarkupWriter;
009    import org.apache.tapestry5.annotations.MixinAfter;
010    import org.apache.tapestry5.annotations.Parameter;
011    
012    /**
013     * @author Ivan Khalopik
014     */
015    @MixinAfter
016    public class DefineClass {
017            @Parameter(name = "class", defaultPrefix = BindingConstants.LITERAL)
018            private String className;
019    
020            protected void beginRender(MarkupWriter writer) {
021                    if (className != null) {
022                            writer.getElement().addClassName(className);
023                    }
024            }
025    }
