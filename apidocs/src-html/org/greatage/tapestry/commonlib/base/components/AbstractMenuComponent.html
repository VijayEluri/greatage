005    package org.greatage.tapestry.commonlib.base.components;
006    
007    import org.apache.tapestry5.annotations.Component;
008    import org.apache.tapestry5.json.JSONObject;
009    import org.greatage.tapestry.commonlib.components.Menu;
010    
011    /**
012     * @author Ivan Khalopik
013     */
014    public abstract class AbstractMenuComponent extends AbstractComponent {
015    
016            private String triggerId;
017            private String menuId;
018    
019            @Component(
020                            parameters = {
021                                            "overrides=overrides",
022                                            "id=prop:menuId",
023                                            "class=prop:menuClass"})
024            private Menu menu;
025    
026            public String getMenuClass() {
027                    return null;
028            }
029    
030            public String getTriggerId() {
031                    return triggerId;
032            }
033    
034            public String getMenuId() {
035                    return menuId;
036            }
037    
038            protected void setupRender() {
039                    triggerId = allocateClientId("trigger");
040                    menuId = allocateClientId("menu");
041            }
042    
043            protected void afterRender() {
044                    getJavaScriptSupport().addInitializerCall("menu", new JSONObject("triggerId", triggerId, "menuId", menuId));
045            }
046    }
