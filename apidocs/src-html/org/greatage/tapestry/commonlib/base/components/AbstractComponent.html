005    package org.greatage.tapestry.commonlib.base.components;
006    
007    import org.apache.tapestry5.Binding;
008    import org.apache.tapestry5.ComponentAction;
009    import org.apache.tapestry5.ComponentResources;
010    import org.apache.tapestry5.PropertyOverrides;
011    import org.apache.tapestry5.annotations.Environmental;
012    import org.apache.tapestry5.annotations.Parameter;
013    import org.apache.tapestry5.ioc.Messages;
014    import org.apache.tapestry5.ioc.annotations.Inject;
015    import org.apache.tapestry5.services.BindingSource;
016    import org.apache.tapestry5.services.Environment;
017    import org.apache.tapestry5.services.FormSupport;
018    import org.apache.tapestry5.services.javascript.JavaScriptSupport;
019    
020    /**
021     * @author Ivan Khalopik
022     */
023    public abstract class AbstractComponent {
024            @Parameter(value = "this", allowNull = false)
025            private PropertyOverrides overrides;
026    
027            @Inject
028            private Messages messages;
029    
030            @Inject
031            private ComponentResources resources;
032    
033            @Inject
034            private JavaScriptSupport javaScriptSupport;
035    
036            @Inject
037            private BindingSource bindingSource;
038    
039            @Inject
040            private Environment environment;
041    
042            @Environmental(false)
043            private FormSupport formSupport;
044    
045            public ComponentResources getResources() {
046                    return resources;
047            }
048    
049            public PropertyOverrides getOverrides() {
050                    return overrides;
051            }
052    
053            public Messages getMessages() {
054                    return messages;
055            }
056    
057            public Messages getOverrideMessages() {
058                    return overrides.getOverrideMessages();
059            }
060    
061            protected JavaScriptSupport getJavaScriptSupport() {
062                    return javaScriptSupport;
063            }
064    
065            protected FormSupport getFormSupport() {
066                    return formSupport;
067            }
068    
069            protected <T> void storeAndExecute(T component, ComponentAction<T> action) {
070                    if (formSupport != null) {
071                            formSupport.storeAndExecute(component, action);
072                    } else {
073                            action.execute(component);
074                    }
075            }
076    
077            protected Binding newBinding(final String description, final ComponentResources container, final String defaultPrefix, final String expression) {
078                    return bindingSource.newBinding(description, container, defaultPrefix, expression);
079            }
080    
081            protected Binding newBinding(final String description, final String defaultPrefix, final String expression) {
082                    return bindingSource.newBinding(description, resources, defaultPrefix, expression);
083            }
084    
085            protected <T> T popFromEnvironment(Class<T> type) {
086                    return environment.pop(type);
087            }
088    
089            protected <T> T pushToEnvironment(Class<T> type, T instance) {
090                    return environment.push(type, instance);
091            }
092    
093            protected String allocateClientId() {
094                    return allocateClientId(null);
095            }
096    
097            protected String allocateClientId(String id) {
098                    return id != null ?
099                                    getJavaScriptSupport().allocateClientId(id) :
100                                    getJavaScriptSupport().allocateClientId(getResources());
101            }
102    }
