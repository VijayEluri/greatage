005    package org.greatage.tapestry.services;
006    
007    import org.apache.tapestry5.internal.services.GenericValueEncoderFactory;
008    import org.apache.tapestry5.internal.services.ResourceCache;
009    import org.apache.tapestry5.ioc.Configuration;
010    import org.apache.tapestry5.ioc.MappedConfiguration;
011    import org.apache.tapestry5.ioc.ServiceBinder;
012    import org.apache.tapestry5.ioc.annotations.Symbol;
013    import org.apache.tapestry5.services.BeanBlockContribution;
014    import org.apache.tapestry5.services.LibraryMapping;
015    import org.apache.tapestry5.services.PersistentFieldStrategy;
016    import org.apache.tapestry5.services.ValueEncoderFactory;
017    import org.greatage.tapestry.CommonSymbols;
018    import org.greatage.tapestry.PersistenceConstants;
019    import org.greatage.tapestry.ThemeConstants;
020    import org.greatage.tapestry.internal.EncodedFlashPersistentFieldStrategy;
021    import org.greatage.tapestry.internal.EncodedSessionPersistentFieldStrategy;
022    import org.greatage.tapestry.internal.LocaleValueEncoder;
023    import org.greatage.tapestry.internal.MenuModelSourceImpl;
024    import org.greatage.tapestry.internal.RealClassResolver;
025    import org.greatage.tapestry.internal.SelectModelBuilder;
026    import org.greatage.tapestry.internal.SvgResourceCache;
027    import org.greatage.tapestry.internal.TimeZoneValueEncoder;
028    
029    import java.util.Locale;
030    import java.util.TimeZone;
031    
032    /**
033     * @author Ivan Khalopik
034     */
035    public class CommonModule {
036    
037            public static void bind(ServiceBinder binder) {
038                    binder.bind(MenuModelSource.class, MenuModelSourceImpl.class);
039                    binder.bind(SelectModelBuilder.class, SelectModelBuilder.class);
040                    binder.bind(RealClassResolver.class, RealClassResolver.class);
041            }
042    
043            public static void contributeFactoryDefaults(MappedConfiguration<String, String> configuration) {
044                    configuration.add(CommonSymbols.THEME_PATH, "org/greatage/tapestry/theme");
045                    configuration.add(CommonSymbols.THEME, ThemeConstants.WHITE_THEME);
046            }
047    
048            public static void contributeComponentClassResolver(Configuration<LibraryMapping> configuration) {
049                    configuration.add(new LibraryMapping("common", "org.greatage.tapestry.commonlib"));
050            }
051    
052            public void contributeValueEncoderSource(final MappedConfiguration<Class, ValueEncoderFactory> configuration) {
053                    configuration.add(Locale.class, GenericValueEncoderFactory.create(new LocaleValueEncoder()));
054                    configuration.add(TimeZone.class, GenericValueEncoderFactory.create(new TimeZoneValueEncoder()));
055            }
056    
057            public static void contributeClasspathAssetAliasManager(MappedConfiguration<String, String> configuration,
058                                                                                                                            @Symbol(CommonSymbols.THEME_PATH) String themePath) {
059                    configuration.add("theme", themePath);
060            }
061    
062            public static void contributePersistentFieldManager(MappedConfiguration<String, PersistentFieldStrategy> configuration) {
063                    configuration.addInstance(PersistenceConstants.ENCODED_SESSION, EncodedSessionPersistentFieldStrategy.class);
064                    configuration.addInstance(PersistenceConstants.ENCODED_FLASH, EncodedFlashPersistentFieldStrategy.class);
065            }
066    
067            public void contributeBeanBlockOverrideSource(Configuration<BeanBlockContribution> configuration) {
068                    addEditBlock(configuration, "text");
069                    addEditBlock(configuration, "number");
070                    addEditBlock(configuration, "enum");
071                    addEditBlock(configuration, "boolean");
072                    addEditBlock(configuration, "date");
073                    addEditBlock(configuration, "password");
074                    addEditBlock(configuration, "longtext");
075            }
076    
077            private void addEditBlock(Configuration<BeanBlockContribution> configuration, String dataType) {
078                    addEditBlock(configuration, dataType, dataType);
079            }
080    
081            private void addEditBlock(Configuration<BeanBlockContribution> configuration, String dataType, String blockId) {
082                    configuration.add(new BeanBlockContribution(dataType, "common/EditBlocks", blockId, true));
083            }
084    
085            public ResourceCache decorateResourceCache(final ResourceCache resourceCache) {
086                    return new SvgResourceCache(resourceCache);
087            }
088    }
