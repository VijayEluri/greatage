005    package org.greatage.tapestry.details;
006    
007    import org.apache.tapestry5.beaneditor.BeanModel;
008    
009    import java.util.ArrayList;
010    import java.util.Arrays;
011    import java.util.List;
012    
013    /**
014     * @author Ivan Khalopik
015     */
016    public abstract class BeanModelUtils {
017            private static final String COLUMNS_DELIMITER = "\\s*,\\s*";
018    
019            public static void setColumns(final BeanModel<?> model, final String columns) {
020                    final List<String> propertyNames = model.getPropertyNames();
021                    final List<String> includeColumns = new ArrayList<String>();
022                    final List<String> addColumns = new ArrayList<String>();
023                    final String[] cols = columns.split(COLUMNS_DELIMITER);
024                    for (String column : cols) {
025                            if (propertyNames.contains(column)) {
026                                    includeColumns.add(column);
027                            } else {
028                                    addColumns.add(column);
029                            }
030                    }
031                    model.include(includeColumns.toArray(new String[]{""}));
032                    for (String column : addColumns) {
033                            model.add(column, null);
034                    }
035                    model.reorder(cols);
036            }
037    
038            public static void setSortColumns(final BeanModel<?> model, final String sortColumns) {
039                    final List<String> propertyNames = model.getPropertyNames();
040                    final List<String> columns = Arrays.asList(sortColumns.split(COLUMNS_DELIMITER));
041                    for (String name : propertyNames) {
042                            model.get(name).sortable(columns.contains(name));
043                    }
044            }
045    }
