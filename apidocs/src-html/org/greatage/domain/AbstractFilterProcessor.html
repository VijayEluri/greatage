005    package org.greatage.domain;
006    
007    import org.greatage.util.DescriptionBuilder;
008    
009    import java.io.Serializable;
010    
011    /**
012     * This class represents abstract implementation of {@link EntityFilterProcessor} that works only with filters of
013     * specified class.
014     *
015     * @author Ivan Khalopik
016     * @param <PK>       type of entities primary key
017     * @param <E>        type of entities
018     * @param <F>        type of entities filter
019     * @since 1.0
020     */
021    public abstract class AbstractFilterProcessor<PK extends Serializable, E extends Entity<PK>, F extends EntityFilter<PK, E>>
022                    implements EntityFilterProcessor {
023    
024            private final Class<? extends Entity> supportedEntityClass;
025    
026            /**
027             * Constructor with supported entity class initialization.
028             *
029             * @param supportedEntityClass supported entity class
030             */
031            protected AbstractFilterProcessor(final Class<? extends Entity> supportedEntityClass) {
032                    this.supportedEntityClass = supportedEntityClass;
033            }
034    
035            /**
036             * Executes some logic to inject into entity criteria parameters stored in filter, if specified filter are supported by
037             * this filter processor.
038             *
039             * @param criteria   entity criteria
040             * @param filter         entity filter
041             * @param pagination pagination
042             * @param <SPK>      type of entities primary key
043             * @param <SE>       type of entities
044             */
045            @SuppressWarnings({"unchecked"})
046            public <SPK extends Serializable, SE extends Entity<SPK>>
047            void process(final EntityCriteria criteria, final EntityFilter<SPK, SE> filter, final Pagination pagination) {
048                    if (supports(filter)) {
049                            processFilter(criteria, (F) filter);
050                    }
051            }
052    
053            /**
054             * Determines if this filter processor supports specified filter. If not, it will not be processed.
055             *
056             * @param filter entity filter
057             * @param <SPK>  type of entities primary key
058             * @param <SE>   type of entities
059             * @return true if filter is supported, false otherwise
060             */
061            protected <SPK extends Serializable, SE extends Entity<SPK>>
062            boolean supports(final EntityFilter<SPK, SE> filter) {
063                    return supportedEntityClass.isAssignableFrom(filter.getEntityClass());
064            }
065    
066            /**
067             * Executes some logic to inject into entity criteria parameters stored in filter.
068             *
069             * @param criteria entity criteria
070             * @param filter   processed entity filter
071             */
072            protected abstract void processFilter(final EntityCriteria criteria, final F filter);
073    
074            @Override
075            public String toString() {
076                    final DescriptionBuilder builder = new DescriptionBuilder(getClass());
077                    builder.append(supportedEntityClass);
078                    return builder.toString();
079            }
080    }
