005    package org.greatage.domain;
006    
007    import org.greatage.util.DescriptionBuilder;
008    import org.greatage.util.ReflectionUtils;
009    
010    import java.io.Serializable;
011    import java.util.Collection;
012    import java.util.List;
013    import java.util.Map;
014    
015    /**
016     * @author Ivan Khalopik
017     * @since 1.0
018     */
019    public abstract class AbstractEntityRepository implements EntityRepository {
020            private final Map<Class, Class> entityMapping;
021    
022            protected AbstractEntityRepository(final Map<Class, Class> entityMapping) {
023                    this.entityMapping = entityMapping;
024            }
025    
026            public <PK extends Serializable, E extends Entity<PK>>
027            int count(final Class<E> entityClass) {
028                    return count(createDefaultFilter(entityClass));
029            }
030    
031            public <PK extends Serializable, E extends Entity<PK>>
032            List<E> find(final Class<E> entityClass, final Pagination pagination) {
033                    return find(createDefaultFilter(entityClass), pagination);
034            }
035    
036            public <PK extends Serializable, E extends Entity<PK>>
037            List<PK> findKeys(final Class<E> entityClass, final Pagination pagination) {
038                    return findKeys(createDefaultFilter(entityClass), pagination);
039            }
040    
041            public <PK extends Serializable, E extends Entity<PK>>
042            List<Map<String, Object>> findValueObjects(final Class<E> entityClass, final Map<String, String> projection, final Pagination pagination) {
043                    return findValueObjects(createDefaultFilter(entityClass), projection, pagination);
044            }
045    
046            public <PK extends Serializable, E extends Entity<PK>>
047            E create(final Class<E> entityClass) {
048                    return ReflectionUtils.newInstance(getImplementation(entityClass));
049            }
050    
051            public <PK extends Serializable, E extends Entity<PK>>
052            void saveOrUpdate(final E entity) {
053                    if (entity.isNew()) {
054                            save(entity);
055                    } else {
056                            update(entity);
057                    }
058            }
059    
060            /**
061             * Creates default entity filter for entityClass.
062             *
063             * @param entityClass entity class
064             * @param <PK>        type of entities primary key
065             * @param <E>         type of entities
066             * @return default entity filter for entityClass
067             */
068            protected <PK extends Serializable, E extends Entity<PK>>
069            EntityFilter<PK, E> createDefaultFilter(final Class<E> entityClass) {
070                    return new DefaultEntityFilter<PK, E>(entityClass);
071            }
072    
073            @SuppressWarnings({"unchecked"})
074            protected <T> Class<? extends T> getImplementation(final Class<T> entityClass) {
075                    final Class implementation = entityMapping.get(entityClass);
076                    return implementation != null ? implementation : entityClass;
077            }
078    
079            @Override
080            public String toString() {
081                    final DescriptionBuilder builder = new DescriptionBuilder(getClass());
082                    builder.append("mapping", entityMapping);
083                    return builder.toString();
084            }
085    
086            class DefaultEntityFilter<PK extends Serializable, E extends Entity<PK>> implements EntityFilter<PK, E> {
087                    private final Class<E> entityClass;
088    
089                    public DefaultEntityFilter(final Class<E> entityClass) {
090                            this.entityClass = entityClass;
091                    }
092    
093                    public Class<E> getEntityClass() {
094                            return entityClass;
095                    }
096    
097                    public Collection<PK> getIncludeKeys() {
098                            return null;
099                    }
100    
101                    public Collection<PK> getExcludeKeys() {
102                            return null;
103                    }
104    
105                    public String getQueryString() {
106                            return null;
107                    }
108            }
109    
110    }
