005    package org.greatage.domain.hibernate;
006    
007    import org.greatage.ioc.resource.Resource;
008    import org.greatage.ioc.resource.ResourceLocator;
009    import org.hibernate.cfg.Configuration;
010    
011    import java.io.IOException;
012    import java.util.Properties;
013    
014    /**
015     * @author Ivan Khalopik
016     * @since 1.0
017     */
018    public class HibernateBaseConfiguration implements HibernateConfiguration {
019            private final ResourceLocator resourceLocator;
020            private final String hibernatePropertiesResource;
021    
022            public HibernateBaseConfiguration(final ResourceLocator resourceLocator, final String hibernatePropertiesResource) {
023                    this.resourceLocator = resourceLocator;
024                    this.hibernatePropertiesResource = hibernatePropertiesResource;
025            }
026    
027            public void configure(final Configuration configuration) {
028                    final Resource resource = resourceLocator.getResource(hibernatePropertiesResource);
029                    if (resource.exists()) {
030                            final Properties properties = new Properties();
031                            try {
032                                    properties.load(resource.open());
033                                    configuration.setProperties(properties);
034                            } catch (IOException e) {
035                                    throw new RuntimeException("Can not find hibernate properties", e);
036                            }
037                    }
038            }
039    }
