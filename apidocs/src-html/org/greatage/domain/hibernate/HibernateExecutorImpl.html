005    package org.greatage.domain.hibernate;
006    
007    import org.greatage.domain.Transaction;
008    import org.hibernate.Session;
009    import org.hibernate.SessionFactory;
010    
011    /**
012     * @author Ivan Khalopik
013     * @since 1.0
014     */
015    public class HibernateExecutorImpl implements HibernateExecutor {
016            private final SessionFactory sessionFactory;
017    
018            private Session session;
019    
020            public HibernateExecutorImpl(final SessionFactory sessionFactory) {
021                    this.sessionFactory = sessionFactory;
022            }
023    
024            public <T> T execute(final HibernateCallback<T> callback) {
025                    try {
026                            final Session session = getSession();
027                            final T result = callback.doInSession(session);
028                            session.flush();
029                            return result;
030                    } catch (RuntimeException ex) {
031                            throw ex;
032                    } catch (Throwable throwable) {
033                            throw new RuntimeException(throwable);
034                    }
035            }
036    
037            public void clear() {
038                    if (session != null) {
039                            session.close();
040                            session = null;
041                    }
042            }
043    
044            public Transaction begin() {
045                    return new HibernateTransaction(getSession().beginTransaction());
046            }
047    
048            private Session getSession() {
049                    if (session == null) {
050                            session = sessionFactory.openSession();
051                    }
052                    return session;
053            }
054    }
