005    package org.greatage.domain.hibernate;
006    
007    import org.greatage.domain.EntityCriterion;
008    import org.greatage.util.DescriptionBuilder;
009    import org.hibernate.criterion.Conjunction;
010    import org.hibernate.criterion.Criterion;
011    import org.hibernate.criterion.Disjunction;
012    import org.hibernate.criterion.Restrictions;
013    
014    /**
015     * @author Ivan Khalopik
016     * @since 1.0
017     */
018    public class HibernateCriterion implements EntityCriterion {
019            private final Criterion criterion;
020    
021            public HibernateCriterion(final Criterion criterion) {
022                    this.criterion = criterion;
023            }
024    
025            Criterion getCriterion() {
026                    return criterion;
027            }
028    
029            public EntityCriterion or(final EntityCriterion... criterions) {
030                    final Disjunction disjunction = Restrictions.disjunction();
031                    disjunction.add(getCriterion());
032                    for (EntityCriterion criterion : criterions) {
033                            disjunction.add(((HibernateCriterion) criterion).getCriterion());
034                    }
035                    return new HibernateCriterion(disjunction);
036            }
037    
038            public EntityCriterion and(final EntityCriterion... criterions) {
039                    final Conjunction conjunction = Restrictions.conjunction();
040                    conjunction.add(getCriterion());
041                    for (EntityCriterion criterion : criterions) {
042                            conjunction.add(((HibernateCriterion) criterion).getCriterion());
043                    }
044                    return new HibernateCriterion(conjunction);
045            }
046    
047            public EntityCriterion not() {
048                    return new HibernateCriterion(Restrictions.not(getCriterion()));
049            }
050    
051            @Override
052            public String toString() {
053                    final DescriptionBuilder builder = new DescriptionBuilder(getClass());
054                    builder.append(criterion);
055                    return builder.toString();
056            }
057    }
