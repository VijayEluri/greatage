005    package org.greatage.domain.jdo;
006    
007    import org.greatage.domain.Transaction;
008    
009    import javax.jdo.PersistenceManager;
010    import javax.jdo.PersistenceManagerFactory;
011    
012    /**
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public class JdoExecutorImpl implements JdoExecutor {
017            private final PersistenceManagerFactory persistenceManagerFactory;
018    
019            private PersistenceManager persistenceManager;
020    
021            public JdoExecutorImpl(final PersistenceManagerFactory persistenceManagerFactory) {
022                    this.persistenceManagerFactory = persistenceManagerFactory;
023            }
024    
025            public <T> T execute(final JdoCallback<T> callback) {
026                    try {
027                            final PersistenceManager persistenceManager = getPersistenceManager();
028                            return callback.doInJdo(persistenceManager);
029                    } catch (RuntimeException ex) {
030                            throw ex;
031                    } catch (Throwable throwable) {
032                            throw new RuntimeException(throwable);
033                    }
034            }
035    
036            private PersistenceManager getPersistenceManager() {
037                    if (persistenceManager == null) {
038                            persistenceManager = persistenceManagerFactory.getPersistenceManager();
039                    }
040                    return persistenceManager;
041            }
042    
043            public Transaction begin() {
044                    final javax.jdo.Transaction transaction = getPersistenceManager().currentTransaction();
045                    transaction.begin();
046                    return new JdoTransaction(transaction);
047            }
048    }
