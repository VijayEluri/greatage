005    package org.greatage.domain.jdo;
006    
007    import org.greatage.domain.*;
008    import org.greatage.domain.sql.SqlCriterion;
009    import org.greatage.util.DescriptionBuilder;
010    import org.greatage.util.StringUtils;
011    
012    import javax.jdo.Extent;
013    import javax.jdo.PersistenceManager;
014    import javax.jdo.Query;
015    import java.util.HashMap;
016    import java.util.Map;
017    
018    /**
019     * @author Ivan Khalopik
020     * @since 1.0
021     */
022    public class JdoCriteria implements EntityCriteria {
023            private final Query query;
024            private final String alias;
025    
026            private final Map<String, EntityProperty> properties = new HashMap<String, EntityProperty>();
027            private final Map<String, JdoCriteria> criterias = new HashMap<String, JdoCriteria>();
028    
029            private SqlCriterion filter;
030    
031            private JdoCriteria(final Query query, final String alias) {
032                    this.query = query;
033                    this.alias = alias;
034            }
035    
036            public static JdoCriteria forClass(final PersistenceManager pm, final Class<? extends Entity> entityClass) {
037                    final Extent<? extends Entity> extent = pm.getExtent(entityClass, true);
038                    final Query query = pm.newQuery(extent);
039                    return new JdoCriteria(query, null);
040            }
041    
042            Query assign() {
043                    if (filter != null) {
044                            query.setFilter(filter.getSql());
045                    }
046                    return query;
047            }
048    
049            public String getAlias() {
050                    return alias;
051            }
052    
053            public EntityCriteria root() {
054                    return this;
055            }
056    
057            public JdoCriteria getCriteria(final String path) {
058                    if (!criterias.containsKey(path)) {
059                            criterias.put(path, createCriteria(path));
060                    }
061                    return criterias.get(path);
062            }
063    
064            public EntityProperty getProperty(final String path) {
065                    if (!properties.containsKey(path)) {
066                            properties.put(path, createProperty(path));
067                    }
068                    return properties.get(path);
069            }
070    
071            public void add(final EntityCriterion criterion) {
072                    if (filter == null) {
073                            filter = (SqlCriterion) criterion;
074                    } else {
075                            filter.and(criterion);
076                    }
077            }
078    
079            public void setPagination(final Pagination pagination) {
080                    if (pagination.getCount() >= 0) {
081                            final int start = pagination.getStart() > 0 ? pagination.getStart() : 0;
082                            final int end = start + pagination.getCount();
083                            query.setRange(start, end);
084                    }
085            }
086    
087            private JdoCriteria createCriteria(final String path) {
088                    if (StringUtils.isEmpty(path)) {
089                            throw new IllegalArgumentException("Empty path");
090                    }
091                    return new JdoCriteria(query, path);
092            }
093    
094            private EntityProperty createProperty(final String path) {
095                    if (StringUtils.isEmpty(path)) {
096                            throw new IllegalArgumentException("Empty path");
097                    }
098                    final int i = path.lastIndexOf('.');
099                    final JdoCriteria entityCriteria = i > 0 ? getCriteria(path.substring(0, i)) : this;
100                    final String property = i > 0 ? path.substring(i + 1) : path;
101                    return new JdoProperty(entityCriteria.query, entityCriteria.getAlias(), property);
102            }
103    
104            @Override
105            public String toString() {
106                    final DescriptionBuilder builder = new DescriptionBuilder(getClass());
107                    builder.append(query);
108                    return builder.toString();
109            }
110    }
