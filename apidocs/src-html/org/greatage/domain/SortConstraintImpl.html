005    package org.greatage.domain;
006    
007    /**
008     * @author Ivan Khalopik
009     * @since 1.0
010     */
011    public class SortConstraintImpl implements SortConstraint {
012            private String property;
013            private boolean ascending;
014            private boolean ignoreCase;
015    
016            public SortConstraintImpl(final String property, final boolean ascending, final boolean ignoreCase) {
017                    this.property = property;
018                    this.ascending = ascending;
019                    this.ignoreCase = ignoreCase;
020            }
021    
022            public String getProperty() {
023                    return property;
024            }
025    
026            public boolean isIgnoreCase() {
027                    return ignoreCase;
028            }
029    
030            public boolean isAscending() {
031                    return ascending;
032            }
033    }
