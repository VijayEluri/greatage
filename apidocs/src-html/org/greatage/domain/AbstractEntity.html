005    package org.greatage.domain;
006    
007    import org.greatage.util.DescriptionBuilder;
008    
009    import java.io.Serializable;
010    
011    /**
012     * This class represents implementation of utility methods of {@link org.greatage.domain.Entity}.
013     *
014     * @author Ivan Khalopik
015     * @param <PK> type of entity primary key
016     * @since 1.0
017     */
018    public abstract class AbstractEntity<PK extends Serializable> implements Entity<PK> {
019    
020            public boolean isNew() {
021                    return getId() == null;
022            }
023    
024            @Override
025            public boolean equals(Object o) {
026                    if (this == o) {
027                            return true;
028                    }
029                    if (o == null || !(o instanceof Entity)) {
030                            return false;
031                    }
032                    final Entity entity = (Entity) o;
033                    return !(getId() != null ? !getId().equals(entity.getId()) : entity.getId() != null);
034            }
035    
036            @Override
037            public int hashCode() {
038                    return getId() == null ? super.hashCode() : getId().hashCode();
039            }
040    
041            @Override
042            public String toString() {
043                    final DescriptionBuilder builder = new DescriptionBuilder(getClass());
044                    builder.append("id", isNew() ? "new" : getId());
045                    return builder.toString();
046            }
047    
048    }
