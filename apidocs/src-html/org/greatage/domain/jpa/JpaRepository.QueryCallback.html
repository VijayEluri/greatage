005    package org.greatage.domain.jpa;
006    
007    import org.greatage.domain.*;
008    import org.greatage.util.DescriptionBuilder;
009    
010    import javax.persistence.EntityManager;
011    import javax.persistence.PersistenceException;
012    import javax.persistence.Query;
013    import java.io.Serializable;
014    import java.util.List;
015    import java.util.Map;
016    
017    /**
018     * @author Ivan Khalopik
019     * @since 1.0
020     */
021    public class JpaRepository extends AbstractEntityRepository {
022            private final JpaExecutor executor;
023            private final EntityFilterProcessor filterProcessor;
024    
025            public JpaRepository(final Map<Class, Class> entityMapping,
026                                                     final JpaExecutor executor,
027                                                     final EntityFilterProcessor filterProcessor) {
028                    super(entityMapping);
029                    this.executor = executor;
030                    this.filterProcessor = filterProcessor;
031            }
032    
033            public <PK extends Serializable, E extends Entity<PK>>
034            int count(final EntityFilter<PK, E> filter) {
035                    return execute("select count() from entityClass", filter, Pagination.ALL, new QueryCallback<Integer>() {
036                            public Integer doInQuery(final Query query) {
037                                    return (Integer) query.getSingleResult();
038                            }
039                    });
040            }
041    
042            public <PK extends Serializable, E extends Entity<PK>>
043            List<E> find(final EntityFilter<PK, E> filter, final Pagination pagination) {
044                    return execute("select from entityClass", filter, pagination, new QueryCallback<List<E>>() {
045                            @SuppressWarnings({"unchecked"})
046                            public List<E> doInQuery(final Query query) {
047                                    return query.getResultList();
048                            }
049                    });
050            }
051    
052            public <PK extends Serializable, E extends Entity<PK>>
053            List<PK> findKeys(final EntityFilter<PK, E> filter, final Pagination pagination) {
054                    throw new UnsupportedOperationException("cannot find keys");
055            }
056    
057            public <PK extends Serializable, E extends Entity<PK>>
058            List<Map<String, Object>> findValueObjects(final EntityFilter<PK, E> filter, final Map<String, String> projection, final Pagination pagination) {
059                    throw new UnsupportedOperationException("cannot find value objects");
060            }
061    
062            public <PK extends Serializable, E extends Entity<PK>>
063            E findUnique(final EntityFilter<PK, E> filter) {
064                    return execute("select from entityClass", filter, Pagination.UNIQUE, new QueryCallback<E>() {
065                            @SuppressWarnings({"unchecked"})
066                            public E doInQuery(final Query query) {
067                                    return (E) query.getSingleResult();
068                            }
069                    });
070            }
071    
072            public <PK extends Serializable, E extends Entity<PK>>
073            E get(final Class<E> entityClass, final PK pk) {
074                    return executor.execute(new JpaCallback<E>() {
075                            public E doInJpa(final EntityManager em) throws Throwable {
076                                    return em.find(getImplementation(entityClass), pk);
077                            }
078                    });
079            }
080    
081            public <PK extends Serializable, E extends Entity<PK>>
082            void save(final E entity) {
083                    executor.execute(new JpaCallback<Object>() {
084                            public Object doInJpa(final EntityManager em) throws Throwable {
085                                    em.persist(entity);
086                                    return null;
087                            }
088                    });
089            }
090    
091            public <PK extends Serializable, E extends Entity<PK>>
092            void update(final E entity) {
093                    executor.execute(new JpaCallback<Object>() {
094                            public Object doInJpa(final EntityManager em) throws Throwable {
095                                    em.merge(entity);
096                                    return null;
097                            }
098                    });
099            }
100    
101            public <PK extends Serializable, E extends Entity<PK>>
102            void delete(final E entity) {
103                    executor.execute(new JpaCallback<Object>() {
104                            public Object doInJpa(final EntityManager em) throws Throwable {
105                                    em.remove(entity);
106                                    return null;
107                            }
108                    });
109            }
110    
111            protected <T, PK extends Serializable, E extends Entity<PK>>
112            T execute(final String queryString, final EntityFilter<PK, E> filter, final Pagination pagination, final QueryCallback<T> callback) {
113                    return executor.execute(new JpaCallback<T>() {
114                            public T doInJpa(EntityManager em) throws PersistenceException {
115                                    final Query query = em.createQuery(queryString.replaceAll("entityClass", filter.getEntityClass().getName()));
116                                    if (pagination.getStart() > 0) {
117                                            query.setFirstResult(pagination.getStart());
118                                    }
119                                    if (pagination.getCount() >= 0) {
120                                            query.setMaxResults(pagination.getCount());
121                                    }
122                                    return callback.doInQuery(query);
123                            }
124                    });
125            }
126    
127            public interface QueryCallback<T> {
128    
129                    T doInQuery(Query query);
130    
131            }
132    
133            @Override
134            public String toString() {
135                    final DescriptionBuilder builder = new DescriptionBuilder(getClass());
136                    builder.append("executor", executor);
137                    builder.append("filterProcessor", filterProcessor);
138                    return builder.toString();
139            }
140    }
