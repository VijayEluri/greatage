005    package org.greatage.domain.jpa;
006    
007    import org.greatage.domain.Transaction;
008    
009    import javax.persistence.EntityManager;
010    import javax.persistence.EntityManagerFactory;
011    import javax.persistence.EntityTransaction;
012    
013    /**
014     * @author Ivan Khalopik
015     * @since 1.0
016     */
017    public class JpaExecutorImpl implements JpaExecutor {
018            private final EntityManagerFactory entityManagerFactory;
019    
020            private EntityManager entityManager;
021    
022            public JpaExecutorImpl(final EntityManagerFactory entityManagerFactory) {
023                    this.entityManagerFactory = entityManagerFactory;
024            }
025    
026            public <T> T execute(final JpaCallback<T> callback) {
027                    try {
028                            final EntityManager entityManager = getEntityManager();
029                            return callback.doInJpa(entityManager);
030                    } catch (RuntimeException ex) {
031                            throw ex;
032                    } catch (Throwable throwable) {
033                            throw new RuntimeException(throwable);
034                    }
035            }
036    
037            private EntityManager getEntityManager() {
038                    if (entityManager == null) {
039                            entityManager = entityManagerFactory.createEntityManager();
040                    }
041                    return entityManager;
042            }
043    
044            public Transaction begin() {
045                    final EntityTransaction transaction = getEntityManager().getTransaction();
046                    transaction.begin();
047                    return new JpaTransaction(transaction);
048            }
049    }
