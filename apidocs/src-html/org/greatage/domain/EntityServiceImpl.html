005    package org.greatage.domain;
006    
007    import org.greatage.domain.annotations.Transactional;
008    import org.greatage.util.DescriptionBuilder;
009    import org.greatage.util.ReflectionUtils;
010    
011    import java.io.Serializable;
012    import java.util.List;
013    
014    /**
015     * This class represents default implementation of {@link EntityService}.
016     *
017     * @author Ivan Khalopik
018     * @param <PK>       type of entities primary key
019     * @param <E>        type of entities
020     * @param <Q>        type of entities query
021     * @since 1.0
022     */
023    public class EntityServiceImpl<PK extends Serializable, E extends Entity<PK>, Q extends EntityQuery<PK, E, Q>>
024                    implements EntityService<PK, E> {
025    
026            private final EntityRepository repository;
027            private final Class<E> entityClass;
028            private final Class<Q> queryClass;
029            private final String entityName;
030    
031            /**
032             * Constructor with specified repository and entity class.
033             *
034             * @param repository  entity repository
035             * @param entityClass entity class
036             */
037            public EntityServiceImpl(final EntityRepository repository, final Class<E> entityClass) {
038                    this(repository, entityClass, null);
039            }
040    
041            /**
042             * Constructor with specified repository and entity class.
043             *
044             * @param repository  entity repository
045             * @param entityClass entity class
046             * @param queryClass  entity filter class
047             */
048            public EntityServiceImpl(final EntityRepository repository, final Class<E> entityClass, final Class<Q> queryClass) {
049                    this(repository, entityClass, queryClass, null);
050            }
051    
052            /**
053             * Constructor with specified repository and entity class.
054             *
055             * @param repository  entity repository
056             * @param entityClass entity class
057             * @param queryClass  entity filter class
058             * @param entityName  entity name
059             */
060            public EntityServiceImpl(final EntityRepository repository, final Class<E> entityClass, final Class<Q> queryClass, final String entityName) {
061                    this.repository = repository;
062                    this.entityClass = entityClass;
063                    this.queryClass = queryClass;
064                    this.entityName = entityName != null ? entityName : entityClass.getName();
065            }
066    
067            public Class<E> getEntityClass() {
068                    return entityClass;
069            }
070    
071            public String getEntityName() {
072                    return entityName;
073            }
074    
075            public E create() {
076                    return repository().create(getEntityClass());
077            }
078    
079            @Transactional
080            public void saveOrUpdate(final E entity) {
081                    if (entity.isNew()) {
082                            save(entity);
083                    } else {
084                            update(entity);
085                    }
086            }
087    
088            @Transactional
089            public void save(final E entity) {
090                    repository().save(entity);
091            }
092    
093            @Transactional
094            public void update(final E entity) {
095                    repository().update(entity);
096            }
097    
098            @Transactional
099            public void delete(final E entity) {
100                    repository().delete(entity);
101            }
102    
103            public E get(final PK pk) {
104                    return repository().get(getEntityClass(), pk);
105            }
106    
107            public int getEntitiesCount() {
108                    return createQuery().count();
109            }
110    
111            public List<E> getEntities() {
112                    return createQuery().list(createDefaultPagination());
113            }
114    
115            public List<E> getEntities(final Pagination pagination) {
116                    return createQuery().list(pagination);
117            }
118    
119            /**
120             * Gets entity repository.
121             *
122             * @return entity repository
123             */
124            protected EntityRepository repository() {
125                    return repository;
126            }
127    
128            /**
129             * Gets default pagination for entities selection.
130             *
131             * @return default pagination for entities selection
132             */
133            protected Pagination createDefaultPagination() {
134                    return Pagination.ALL;
135            }
136    
137            /**
138             * Creates default filter for entities selection.
139             *
140             * @return default filter for entities selection
141             */
142            protected Q createQuery() {
143                    return ReflectionUtils.newInstance(queryClass).assign(repository());
144            }
145    
146            @Override
147            public String toString() {
148                    final DescriptionBuilder builder = new DescriptionBuilder(getClass());
149                    builder.append("class", entityClass);
150                    builder.append("name", entityName);
151                    return builder.toString();
152            }
153    }
