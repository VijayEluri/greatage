005    package org.greatage.domain;
006    
007    import java.io.Serializable;
008    import java.util.Collection;
009    
010    /**
011     * This interface represents base filter for selection entities from repositories.
012     *
013     * @author Ivan Khalopik
014     * @param <PK>       type of entities primary key
015     * @param <E>        type of entities
016     * @see EntityRepository
017     * @see EntityService
018     * @since 1.0
019     */
020    public interface EntityFilter<PK extends Serializable, E extends Entity<PK>> {
021    
022            /**
023             * Gets entity class.
024             *
025             * @return entity class
026             */
027            Class<E> getEntityClass();
028    
029            /**
030             * Gets collection of entity primary keys that will be included into select result.
031             *
032             * @return collection of entity primary keys that will be included into select result
033             */
034            Collection<PK> getIncludeKeys();
035    
036            /**
037             * Gets collection of entity primary keys that will be excluded from select result.
038             *
039             * @return collection of entity primary keys that will be excluded from select result
040             */
041            Collection<PK> getExcludeKeys();
042    
043            /**
044             * Gets search query string for entity full-text search.
045             *
046             * @return search query string for entity full-text search
047             */
048            String getQueryString();
049    
050    }
