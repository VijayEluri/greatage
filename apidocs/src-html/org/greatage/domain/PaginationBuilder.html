005    package org.greatage.domain;
006    
007    import org.greatage.util.CollectionUtils;
008    import org.greatage.util.DescriptionBuilder;
009    
010    import java.util.Collection;
011    import java.util.List;
012    
013    /**
014     * This class represents featured implementation of {@link org.greatage.domain.Pagination} with programmatic build and
015     * lazy initialization.
016     *
017     * @author Ivan Khalopik
018     * @since 1.0
019     */
020    public class PaginationBuilder implements Pagination {
021            private int start;
022            private int count;
023    
024            private int from;
025            private int to;
026            private int page;
027            private int size = -1;
028            private final List<SortConstraint> sortDefinitions = CollectionUtils.newList();
029    
030            private boolean modified = true;
031    
032            public int getStart() {
033                    lazyBuild();
034                    return start;
035            }
036    
037            public int getCount() {
038                    lazyBuild();
039                    return count;
040            }
041    
042            public Collection<SortConstraint> getSortConstraints() {
043                    lazyBuild();
044                    return sortDefinitions;
045            }
046    
047            /**
048             * Sets start pagination position.
049             *
050             * @param start start pagination position
051             * @return this
052             */
053            public PaginationBuilder start(final int start) {
054                    this.from = start;
055                    return modify();
056            }
057    
058            /**
059             * Sets end pagination position.
060             *
061             * @param end end pagination position
062             * @return this
063             * @throws IllegalStateException if page or size parameter are specified
064             */
065            public PaginationBuilder end(final int end) {
066                    if (page > 0 || size > 0) {
067                            throw new IllegalStateException("page or size attributes already exists");
068                    }
069                    this.to = end;
070                    return modify();
071            }
072    
073            /**
074             * Sets pagination count of rows.
075             *
076             * @param count pagination count of rows
077             * @return this
078             * @see #size(int)
079             */
080            public PaginationBuilder count(final int count) {
081                    return size(count);
082            }
083    
084            /**
085             * Sets pagination size of selection.
086             *
087             * @param size pagination size of selection
088             * @return this
089             * @throws IllegalStateException if end parameter are specified
090             */
091            public PaginationBuilder size(final int size) {
092                    if (to > 0) {
093                            throw new IllegalStateException("to attribute already exists");
094                    }
095                    this.size = size;
096                    return modify();
097            }
098    
099            /**
100             * Sets selection page.
101             *
102             * @param page selection page
103             * @return this
104             * @throws IllegalStateException if end parameter are specified
105             */
106            public PaginationBuilder page(final int page) {
107                    if (to > 0) {
108                            throw new IllegalStateException("to attribute already exists");
109                    }
110                    this.page = page;
111                    return modify();
112            }
113    
114            /**
115             * Sets case sensitive ascending sort order by property.
116             *
117             * @param property sort property
118             * @return this
119             */
120            public PaginationBuilder sort(final String property) {
121                    return sort(property, true);
122            }
123    
124            /**
125             * Sets case insensitive sort order by property, parameterized by ascending.
126             *
127             * @param property  sort property
128             * @param ascending ascending sort order
129             * @return this
130             */
131            public PaginationBuilder sort(final String property, final boolean ascending) {
132                    return sort(property, ascending, true);
133            }
134    
135            /**
136             * Sets sort order by property, parameterized by ascending and ignoreCase.
137             *
138             * @param property   sort property
139             * @param ascending  ascending sort order
140             * @param ignoreCase ignore case
141             * @return this
142             */
143            public PaginationBuilder sort(final String property, final boolean ascending, final boolean ignoreCase) {
144                    sortDefinitions.add(new SortConstraintImpl(property, ascending, ignoreCase));
145                    return modify();
146            }
147    
148            /**
149             * Clears sort orders.
150             *
151             * @return this
152             */
153            public PaginationBuilder unsorted() {
154                    sortDefinitions.clear();
155                    return modify();
156            }
157    
158            /**
159             * Sets modified flag, that determines if pagination parameters were modified.
160             *
161             * @return this
162             */
163            private PaginationBuilder modify() {
164                    this.modified = true;
165                    return this;
166            }
167    
168            /**
169             * Initializes general parameters (start and count) if parameters were modified.
170             */
171            private void lazyBuild() {
172                    if (modified) {
173                            if (page > 0 && size >= 0) {
174                                    start = from + (page - 1) * size;
175                                    count = size;
176                            } else if (to > 0 && to >= from) {
177                                    start = from;
178                                    count = to - from;
179                            } else {
180                                    start = from;
181                                    count = size >= 0 ? size : -1;
182                            }
183                            modified = false;
184                    }
185            }
186    
187            @Override
188            public String toString() {
189                    final DescriptionBuilder builder = new DescriptionBuilder(Pagination.class);
190                    builder.append("start", start);
191                    builder.append("count", count);
192                    return builder.toString();
193            }
194    
195            /**
196             * Creates default pagination builder.
197             *
198             * @return default pagination builder
199             */
200            public static PaginationBuilder create() {
201                    return new PaginationBuilder();
202            }
203    }
