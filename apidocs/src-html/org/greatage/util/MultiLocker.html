005    package org.greatage.util;
006    
007    import java.util.Map;
008    
009    /**
010     * This class represent utility that locks some code by key to prevent use it again.
011     *
012     * @param <K> type of keys
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public class MultiLocker<K> {
017            private final Map<K, Locker> lockers = CollectionUtils.newConcurrentMap();
018    
019            /**
020             * Checks if specified key is locked.
021             *
022             * @param key key
023             * @throws IllegalStateException if instance is locked
024             */
025            public synchronized void check(final K key) {
026                    if (lockers.containsKey(key)) {
027                            final Locker locker = lockers.get(key);
028                            locker.check();
029                    }
030            }
031    
032            /**
033             * Locks specified key to prevent use it again.
034             *
035             * @param key key
036             * @throws IllegalStateException if instance is locked
037             */
038            public synchronized void lock(final K key) {
039                    if (!lockers.containsKey(key)) {
040                            lockers.put(key, new Locker());
041                    }
042                    lockers.get(key).lock();
043            }
044    
045            @Override
046            public String toString() {
047                    final DescriptionBuilder builder = new DescriptionBuilder(getClass());
048                    builder.append("lockers", lockers);
049                    return builder.toString();
050            }
051    }
