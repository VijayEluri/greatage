005    package org.greatage.util;
006    
007    import java.util.List;
008    
009    /**
010     * This class represents utility that helps to implement toString() method. It builds entity description according to
011     * configured entity name and entity parameters. The result string will look like
012     * <tt>EntityName(key1=value1,key2=value2,value3,key4=value4...)</tt>.
013     *
014     * @author Ivan Khalopik
015     * @since 1.0
016     */
017    public class DescriptionBuilder {
018            private final String entityName;
019            private final List<String> parameters = CollectionUtils.newList();
020    
021            /**
022             * Creates new utility that helps to implement toString() method and initializes entity name with specified parameter.
023             *
024             * @param entityName entity name
025             */
026            public DescriptionBuilder(final String entityName) {
027                    this.entityName = entityName;
028            }
029    
030            /**
031             * Creates new utility that helps to implement toString() method and initializes entity name with simplified class
032             * name.
033             *
034             * @param entityClass entity class
035             */
036            public DescriptionBuilder(final Class entityClass) {
037                    this.entityName = entityClass.getSimpleName();
038            }
039    
040            /**
041             * Appends key-value pair to entity description. They will appear in result string like <tt>(parameter=value,
042             * parameter=value...)</tt>.
043             *
044             * @param key   key
045             * @param value value
046             * @return this description builder instance
047             */
048            public DescriptionBuilder append(final String key, final Object value) {
049                    parameters.add(key + "=" + value);
050                    return this;
051            }
052    
053            /**
054             * Appends value to entity description. It will appear in result string like <tt>(value, value...)</tt>.
055             *
056             * @param value value
057             * @return this description builder instance
058             */
059            public DescriptionBuilder append(final Object value) {
060                    parameters.add(String.valueOf(value));
061                    return this;
062            }
063    
064            /**
065             * Builds entity description according to configured earlier entity name and entity parameters. The result string will
066             * look like <tt>EntityName(key1=value1,key2=value2,value3,key4=value4...)</tt>
067             *
068             * @return entity description built according to configured earlier entity name and entity parameters
069             */
070            public String toString() {
071                    final StringBuilder builder = new StringBuilder(entityName);
072                    if (!CollectionUtils.isEmpty(parameters)) {
073                            builder.append("(");
074                            final StringBuilder parametersBuilder = new StringBuilder();
075                            for (String parameter : parameters) {
076                                    if (parametersBuilder.length() > 0) {
077                                            parametersBuilder.append(", ");
078                                    }
079                                    parametersBuilder.append(parameter);
080                            }
081                            builder.append(parametersBuilder.toString());
082                            builder.append(")");
083                    }
084                    return builder.toString();
085            }
086    }
