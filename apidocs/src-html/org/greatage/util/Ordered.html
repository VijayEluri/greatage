005    package org.greatage.util;
006    
007    import java.util.List;
008    
009    /**
010     * This interface represents item that can be ordered according to its order identifier and constraints.
011     *
012     * @author Ivan Khalopik
013     * @since 1.0
014     */
015    public interface Ordered {
016    
017            /**
018             * Prefix that indicates that this item must be before item specified in suffix.
019             */
020            String BEFORE = "before:";
021    
022            /**
023             * Prefix that indicates that this item must be after item specified in suffix.
024             */
025            String AFTER = "after:";
026    
027            /**
028             * Gets order identifier.
029             *
030             * @return order identifier
031             */
032            String getOrderId();
033    
034            /**
035             * Gets list of order constraints. They must be like <tt>after:<another order id></tt>.
036             *
037             * @return list of order constraints or empty list
038             */
039            List<String> getOrderConstraints();
040    }
