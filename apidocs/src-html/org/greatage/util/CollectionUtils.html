005    package org.greatage.util;
006    
007    import java.util.ArrayList;
008    import java.util.Arrays;
009    import java.util.Collection;
010    import java.util.HashMap;
011    import java.util.HashSet;
012    import java.util.List;
013    import java.util.Map;
014    import java.util.Set;
015    import java.util.concurrent.ConcurrentHashMap;
016    import java.util.concurrent.CopyOnWriteArrayList;
017    import java.util.concurrent.CopyOnWriteArraySet;
018    
019    /**
020     * This class represents utility methods for working with java arrays and collections.
021     *
022     * @author Ivan Khalopik
023     * @since 1.0
024     */
025    public abstract class CollectionUtils {
026    
027            /**
028             * Checks if collection is empty.
029             *
030             * @param collection collection to check
031             * @return true if collection is not null and has at last one element, false otherwise
032             */
033            public static boolean isEmpty(final Collection collection) {
034                    return (collection == null || collection.isEmpty());
035            }
036    
037            /**
038             * Creates new empty generic set.
039             *
040             * @param <V> type of values
041             * @return new empty set
042             */
043            public static <V> Set<V> newSet() {
044                    return new HashSet<V>();
045            }
046    
047            /**
048             * Creates new generic set initialized with specified values.
049             *
050             * @param <V>    type of values
051             * @param values values to initialize set with
052             * @return new generic set initialized with specified values
053             */
054            public static <V> Set<V> newSet(final Collection<? extends V> values) {
055                    return new HashSet<V>(values);
056            }
057    
058            /**
059             * Creates new generic set initialized with specified values.
060             *
061             * @param <V>    type of values
062             * @param <T>    type of values
063             * @param values values to initialize set with
064             * @return new generic set initialized with specified values
065             */
066            public static <V, T extends V> Set<V> newSet(final T... values) {
067                    return new HashSet<V>(Arrays.asList(values));
068            }
069    
070            /**
071             * Creates new empty thread safe set.
072             *
073             * @param <V> type of values
074             * @return new empty thread safe set
075             */
076            public static <V> Set<V> newConcurrentSet() {
077                    return new CopyOnWriteArraySet<V>();
078            }
079    
080            /**
081             * Creates new thread safe set initialized with specified values.
082             *
083             * @param <V>    type of values
084             * @param values values to initialize set with
085             * @return new thread safe set initialized with specified values
086             */
087            public static <V> Set<V> newConcurrentSet(final Collection<? extends V> values) {
088                    return new CopyOnWriteArraySet<V>(values);
089            }
090    
091            /**
092             * Creates new thread safe set initialized with specified values.
093             *
094             * @param <V>    type of values
095             * @param <T>    type of values
096             * @param values values to initialize set with
097             * @return new thread safe set initialized with specified values
098             */
099            public static <V, T extends V> Set<V> newConcurrentSet(final T... values) {
100                    return new CopyOnWriteArraySet<V>(Arrays.asList(values));
101            }
102    
103            /**
104             * Creates new empty generic list.
105             *
106             * @param <V> type of values
107             * @return new empty list
108             */
109            public static <V> List<V> newList() {
110                    return new ArrayList<V>();
111            }
112    
113            /**
114             * Creates new generic list initialized with specified values.
115             *
116             * @param <V>    type of values
117             * @param values values to initialize list with
118             * @return new generic list initialized with specified values
119             */
120            public static <V> List<V> newList(final Collection<? extends V> values) {
121                    return new ArrayList<V>(values);
122            }
123    
124            /**
125             * Creates new generic list initialized with specified values.
126             *
127             * @param <V>    type of values
128             * @param <T>    type of values
129             * @param values values to initialize list with
130             * @return new generic list initialized with specified values
131             */
132            public static <V, T extends V> List<V> newList(final T... values) {
133                    return new ArrayList<V>(Arrays.asList(values));
134            }
135    
136            /**
137             * Creates new empty thread safe list.
138             *
139             * @param <V> type of values
140             * @return new empty thread safe list
141             */
142            public static <V> List<V> newConcurrentList() {
143                    return new CopyOnWriteArrayList<V>();
144            }
145    
146            /**
147             * Creates new thread safe list initialized with specified values.
148             *
149             * @param <V>    type of values
150             * @param values values to initialize list with
151             * @return new thread safe list initialized with specified values
152             */
153            public static <V> List<V> newConcurrentList(final Collection<? extends V> values) {
154                    return new CopyOnWriteArrayList<V>(values);
155            }
156    
157            /**
158             * Creates new thread safe list initialized with specified values.
159             *
160             * @param <V>    type of values
161             * @param <T>    type of values
162             * @param values values to initialize list with
163             * @return new thread safe list initialized with specified values
164             */
165            public static <V, T extends V> List<V> newConcurrentList(final T... values) {
166                    return new CopyOnWriteArrayList<V>(Arrays.asList(values));
167            }
168    
169            /**
170             * Creates new empty generic map.
171             *
172             * @param <K> type of keys
173             * @param <V> type of values
174             * @return new empty generic map
175             */
176            public static <K, V> Map<K, V> newMap() {
177                    return new HashMap<K, V>();
178            }
179    
180            /**
181             * Creates new generic map initialized with specified values.
182             *
183             * @param <K>    type of keys
184             * @param <V>    type of values
185             * @param values values to initialize map with
186             * @return new generic map initialized with specified values
187             */
188            public static <K, V> Map<K, V> newMap(final Map<? extends K, ? extends V> values) {
189                    return new HashMap<K, V>(values);
190            }
191    
192            /**
193             * Creates new generic map initialized with keys like even elements and values like odd elements.
194             *
195             * @param <K>    type of keys
196             * @param <V>    type of values
197             * @param values values to initialize map with
198             * @return new generic map initialized with keys like array even elements and values like odd elements
199             */
200            public static <K, V> Map<K, V> newMap(final Object... values) {
201                    final Map<K, V> map = newMap();
202                    return fillMap(map, values);
203            }
204    
205            /**
206             * Creates new empty thread safe map.
207             *
208             * @param <K> type of keys
209             * @param <V> type of values
210             * @return new empty thread safe map
211             */
212            public static <K, V> Map<K, V> newConcurrentMap() {
213                    return new ConcurrentHashMap<K, V>();
214            }
215    
216            /**
217             * Creates new thread safe map initialized with specified values.
218             *
219             * @param <K>    type of keys
220             * @param <V>    type of values
221             * @param values values to initialize map with
222             * @return new thread safe map initialized with specified values
223             */
224            public static <K, V> Map<K, V> newConcurrentMap(final Map<? extends K, ? extends V> values) {
225                    return new ConcurrentHashMap<K, V>(values);
226            }
227    
228            /**
229             * Creates new thread safe map initialized with keys like even elements and values like odd elements.
230             *
231             * @param <K>    type of keys
232             * @param <V>    type of values
233             * @param values values to initialize map with
234             * @return new thread safe map initialized with keys like array even elements and values like odd elements
235             */
236            public static <K, V> Map<K, V> newConcurrentMap(final Object... values) {
237                    final Map<K, V> map = newConcurrentMap();
238                    return fillMap(map, values);
239            }
240    
241            /**
242             * Fills map with keys like even elements and values like odd elements.
243             *
244             * @param <K>    type of keys
245             * @param <V>    type of values
246             * @param map   map to fill
247             * @param values values to fill map with
248             * @return filled with specified values map
249             */
250            @SuppressWarnings("unchecked")
251            private static <K, V> Map<K, V> fillMap(final Map<K, V> map, final Object... values) {
252                    if (values == null || values.length % 2 == 1) {
253                            throw new IllegalArgumentException("Values length must be ord");
254                    }
255                    for (int i = 0; i < values.length; i += 2) {
256                            map.put((K) values[i], (V) values[i + 1]);
257                    }
258                    return map;
259            }
260    }
