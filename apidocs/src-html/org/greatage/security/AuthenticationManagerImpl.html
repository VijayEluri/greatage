005    package org.greatage.security;
006    
007    import java.util.List;
008    
009    /**
010     * @author Ivan Khalopik
011     * @since 1.0
012     */
013    public class AuthenticationManagerImpl<T extends Authentication> implements AuthenticationManager {
014            private final List<AuthenticationProvider<T>> providers;
015            private final SecurityContext<T> securityContext;
016    
017            public AuthenticationManagerImpl(final List<AuthenticationProvider<T>> providers, final SecurityContext<T> securityContext) {
018                    this.providers = providers;
019                    this.securityContext = securityContext;
020            }
021    
022            public void signIn(final AuthenticationToken token) throws AuthenticationException {
023                    final T user = doSignIn(token);
024                    securityContext.initCurrentUser(user);
025            }
026    
027            public void signOut() throws AuthenticationException {
028                    final T user = securityContext.getCurrentUser();
029                    doSignOut(user);
030                    securityContext.removeCurrentUser();
031            }
032    
033            private T doSignIn(final AuthenticationToken token) throws AuthenticationException {
034                    for (AuthenticationProvider<T> provider : providers) {
035                            final T user = provider.signIn(token);
036                            if (user != null) {
037                                    return user;
038                            }
039                    }
040                    throw new AuthenticationException(String.format("Authentication token are not supported %s", token));
041            }
042    
043            private void doSignOut(final T authentication) throws AuthenticationException {
044                    for (AuthenticationProvider<T> provider : providers) {
045                            provider.signOut(authentication);
046                    }
047            }
048    }
