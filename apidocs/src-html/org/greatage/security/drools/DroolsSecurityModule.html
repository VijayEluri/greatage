005    package org.greatage.security.drools;
006    
007    import org.drools.KnowledgeBase;
008    import org.drools.KnowledgeBaseFactory;
009    import org.drools.builder.KnowledgeBuilder;
010    import org.drools.builder.KnowledgeBuilderError;
011    import org.drools.builder.KnowledgeBuilderFactory;
012    import org.drools.builder.ResourceType;
013    import org.drools.io.ResourceFactory;
014    import org.greatage.ioc.ServiceBinder;
015    import org.greatage.ioc.annotations.Bind;
016    import org.greatage.ioc.annotations.Build;
017    import org.greatage.ioc.annotations.Dependency;
018    import org.greatage.ioc.resource.Resource;
019    import org.greatage.ioc.resource.ResourceLocator;
020    import org.greatage.security.AccessControlManager;
021    import org.greatage.security.SecurityModule;
022    
023    import java.io.IOException;
024    import java.util.Arrays;
025    import java.util.Collection;
026    
027    /**
028     * @author Ivan Khalopik
029     * @since 1.0
030     */
031    @Dependency(SecurityModule.class)
032    public class DroolsSecurityModule {
033    
034            @Bind
035            public static void bind(final ServiceBinder binder) {
036                    binder.bind(AccessControlManager.class, DroolsAccessControlManager.class);
037            }
038    
039            @Build
040            public KnowledgeBase buildKnowledgeBase(
041                            final Collection<String> resources,
042                            final ResourceLocator resourceLocator) throws IOException {
043                    final KnowledgeBuilder builder = KnowledgeBuilderFactory.newKnowledgeBuilder();
044    
045                    for (String resource : resources) {
046                            final Resource res = resourceLocator.getResource(resource);
047                            builder.add(ResourceFactory.newInputStreamResource(res.open()), ResourceType.DRL);
048                    }
049    
050                    if (builder.hasErrors()) {
051                            final StringBuilder sb = new StringBuilder();
052                            for (KnowledgeBuilderError error : builder.getErrors()) {
053                                    sb.append("Error at lines ").append(Arrays.toString(error.getErrorLines())).append(":").append("\n");
054                                    sb.append(error.getMessage()).append("\n");
055                            }
056                            throw new RuntimeException(sb.toString());
057                    }
058                    final KnowledgeBase knowledgeBase = KnowledgeBaseFactory.newKnowledgeBase();
059                    knowledgeBase.addKnowledgePackages(builder.getKnowledgePackages());
060                    return knowledgeBase;
061            }
062    }
