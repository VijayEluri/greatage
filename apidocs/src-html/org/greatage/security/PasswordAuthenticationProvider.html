005    package org.greatage.security;
006    
007    /**
008     * @author Ivan Khalopik
009     * @since 1.0
010     */
011    public abstract class PasswordAuthenticationProvider<T extends PasswordAuthentication>
012                    extends AbstractAuthenticationProvider<T, PasswordAuthenticationToken> {
013    
014            private final PasswordEncoder passwordEncoder;
015    
016            public PasswordAuthenticationProvider(final PasswordEncoder passwordEncoder) {
017                    super(PasswordAuthenticationToken.class);
018                    this.passwordEncoder = passwordEncoder;
019            }
020    
021            @Override
022            protected T doSignIn(final PasswordAuthenticationToken token) {
023                    final T authentication = getAuthentication(token.getName());
024                    if (authentication == null) {
025                            throw new AuthenticationException(String.format("Wrong name '%s'", token.getName()));
026                    }
027    
028                    checkPassword(token.getPassword(), authentication.getPassword());
029                    return authentication;
030            }
031    
032            protected abstract T getAuthentication(final String name);
033    
034            protected void checkPassword(final String password, final String expected) {
035                    final String actual = passwordEncoder.encode(password);
036                    if ((actual != null && !actual.equals(expected)) ||
037                                    (actual == null && expected != null)) {
038                            throw new AuthenticationException(String.format("Wrong password '%s'", password));
039                    }
040            }
041    }
