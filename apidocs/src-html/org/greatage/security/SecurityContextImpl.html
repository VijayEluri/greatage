005    package org.greatage.security;
006    
007    import org.greatage.util.CollectionUtils;
008    
009    import java.util.List;
010    import java.util.Map;
011    
012    /**
013     * @author Ivan Khalopik
014     * @since 1.0
015     */
016    public class SecurityContextImpl<T extends Authentication> implements SecurityContext<T> {
017            private final ThreadLocal<T> currentUser = new ThreadLocal<T>();
018            private final Map<String, T> loggedUsers = CollectionUtils.newConcurrentMap();
019    
020            public T getCurrentUser() {
021                    return currentUser.get();
022            }
023    
024            public T getUser(final String name) {
025                    return loggedUsers.get(name);
026            }
027    
028            public List<T> getLoggedUsers() {
029                    return CollectionUtils.newList(loggedUsers.values());
030            }
031    
032            public void initCurrentUser(final T user) {
033                    if (user != null) {
034                            loggedUsers.put(user.getName(), user);
035                    }
036                    currentUser.set(user);
037            }
038    
039            public void clearCurrentUser() {
040                    currentUser.set(null);
041            }
042    
043            public void removeCurrentUser() {
044                    final T user = currentUser.get();
045                    if (user != null) {
046                            loggedUsers.remove(user.getName());
047                    }
048                    currentUser.set(null);
049            }
050    }
