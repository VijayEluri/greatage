001    package org.greatage.security;
002    
003    /**
004     * @author Ivan Khalopik
005     * @since 1.0
006     */
007    public abstract class AbstractAuthenticationProvider<T extends Authentication, E extends AuthenticationToken>
008                    implements AuthenticationProvider<T> {
009    
010            private final Class<E> supportedTokenClass;
011    
012            protected AbstractAuthenticationProvider(final Class<E> supportedTokenClass) {
013                    this.supportedTokenClass = supportedTokenClass;
014            }
015    
016            @SuppressWarnings({"unchecked"})
017            public T signIn(final AuthenticationToken token) throws AuthenticationException {
018                    if (supports(token)) {
019                            return doSignIn((E) token);
020                    }
021    
022                    return null;
023            }
024    
025            public void signOut(final T authentication) throws AuthenticationException {
026                    //do nothing by default
027            }
028    
029            protected boolean supports(final AuthenticationToken token) {
030                    return token != null && supportedTokenClass.isAssignableFrom(token.getClass());
031            }
032    
033            protected abstract T doSignIn(final E token);
034    
035    }
