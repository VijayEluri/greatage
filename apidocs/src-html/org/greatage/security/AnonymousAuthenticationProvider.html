001    package org.greatage.security;
002    
003    /**
004     * @author Ivan Khalopik
005     * @since 1.0
006     */
007    public abstract class AnonymousAuthenticationProvider<T extends Authentication>
008                    extends AbstractAuthenticationProvider<T, AnonymousAuthenticationToken> {
009    
010            public AnonymousAuthenticationProvider() {
011                    super(AnonymousAuthenticationToken.class);
012            }
013    
014            @Override
015            protected T doSignIn(final AnonymousAuthenticationToken token) {
016                    return getAntonymousAuthentication();
017            }
018    
019            protected abstract T getAntonymousAuthentication();
020    
021    }
