005    package org.greatage.security;
006    
007    import org.greatage.ioc.proxy.Invocation;
008    import org.greatage.ioc.proxy.MethodAdvice;
009    import org.greatage.security.annotations.Allow;
010    import org.greatage.security.annotations.Deny;
011    import org.greatage.security.annotations.Operation;
012    import org.greatage.util.CollectionUtils;
013    
014    import java.util.Arrays;
015    import java.util.List;
016    
017    /**
018     * @author Ivan Khalopik
019     * @since 1.0
020     */
021    public class AuthoritySecurityAdvice implements MethodAdvice {
022            private final SecurityContext securityContext;
023    
024            public AuthoritySecurityAdvice(final SecurityContext securityContext) {
025                    this.securityContext = securityContext;
026            }
027    
028            public Object advice(final Invocation invocation, final Object... parameters) throws Throwable {
029                    final Allow allow = invocation.getAnnotation(Allow.class);
030                    final Deny deny = invocation.getAnnotation(Deny.class);
031                    if (allow != null || deny != null) {
032                            final List<String> authorities = getAuthorities();
033                            if (allow != null) {
034                                    checkAllow(allow, authorities);
035                            }
036                            if (deny != null) {
037                                    checkDeny(deny, authorities);
038                            }
039                    }
040                    return invocation.proceed(parameters);
041            }
042    
043            private void checkDeny(final Deny deny, final List<String> authorities) {
044                    if (deny.operation() == Operation.AND) {
045                            if (authorities.containsAll(Arrays.asList(deny.value()))) {
046                                    throw new AccessDeniedException(String.format("Access denied for authorities: '%s'", authorities));
047                            }
048                    } else if (deny.operation() == Operation.OR) {
049                            for (String authority : deny.value()) {
050                                    if (authorities.contains(authority)) {
051                                            throw new AccessDeniedException(String.format("Access denied for authorities: '%s'", authorities));
052                                    }
053                            }
054                    }
055            }
056    
057            private void checkAllow(final Allow allow, final List<String> authorities) {
058                    if (allow.operation() == Operation.AND) {
059                            if (!authorities.containsAll(Arrays.asList(allow.value()))) {
060                                    throw new AccessDeniedException(String.format("Access denied for authorities: '%s'", authorities));
061                            }
062                    } else if (allow.operation() == Operation.OR) {
063                            for (String authority : allow.value()) {
064                                    if (authorities.contains(authority)) {
065                                            return;
066                                    }
067                            }
068                            throw new AccessDeniedException(String.format("Access denied for authorities: '%s'", authorities));
069                    }
070            }
071    
072            private List<String> getAuthorities() {
073                    final Authentication user = securityContext.getCurrentUser();
074                    return user != null ? user.getAuthorities() : CollectionUtils.newList(AuthorityConstants.STATUS_ANONYMOUS);
075            }
076    }
