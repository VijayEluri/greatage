011    public class AccessDeniedException extends SecurityException {
012    
013            public AccessDeniedException() {
014            }
015    
016            public AccessDeniedException(final String s) {
017                    super(s);
018            }
019    
020            public AccessDeniedException(final String message, final Throwable cause) {
021                    super(message, cause);
022            }
023    
024            public AccessDeniedException(final Throwable cause) {
025                    super(cause);
026            }
027    }
