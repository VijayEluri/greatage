005    package org.greatage.javassist;
006    
007    import javassist.CannotCompileException;
008    import javassist.CtClass;
009    import javassist.CtConstructor;
010    import javassist.CtField;
011    import javassist.CtMethod;
012    import javassist.NotFoundException;
013    import org.greatage.util.DescriptionBuilder;
014    
015    /**
016     * This class represents utility that helps to create classes in runtime uses javassist.
017     *
018     * @param <T> type of class that will be created by {@link #build()} method
019     * @author Ivan Khalopik
020     * @since 1.0
021     */
022    public class ClassBuilder<T> {
023            private final ClassPoolEx pool;
024            private final CtClass ctClass;
025    
026            /**
027             * Constructor that creates class builder and initializes it with specified class name. Class instance will be always
028             * created by {@link #build()} method. {@link Object} will be a superclass of created class.
029             *
030             * @param pool ClassPoolEx instance
031             * @param name class name
032             */
033            public ClassBuilder(final ClassPoolEx pool, final String name) {
034                    this(pool, name, false);
035            }
036    
037            /**
038             * Constructor that creates class builder and initializes it with specified class name and isInterface parameters. If
039             * isInterface parameter is true interface instance will be created by {@link #build()} method. For class creation
040             * {@link Object} will be a superclass of created class.
041             *
042             * @param pool          ClassPoolEx instance
043             * @param name          class name
044             * @param isInterface isInterface parameter
045             */
046            public ClassBuilder(final ClassPoolEx pool, final String name, final boolean isInterface) {
047                    this(pool, name, isInterface, null);
048            }
049    
050            /**
051             * Constructor that creates class builder and initializes it with specified class name, isInterface and superClass
052             * parameters. If isInterface parameter is true interface instance will be created by {@link #build()} method. For
053             * class creation if superClass is null {@link Object} will be a superclass of created class, if superClass parameter
054             * is interface {@link Object} will be a superclass of created class and specified superClass parameter will be added
055             * to this class like a implemented interface.
056             *
057             * @param pool          ClassPoolEx instance
058             * @param name          class name
059             * @param isInterface isInterface parameter
060             * @param superClass  superClass parameter, may be null
061             */
062            public ClassBuilder(final ClassPoolEx pool, final String name, final boolean isInterface,
063                                                    final Class<T> superClass) {
064                    this.pool = pool;
065    
066                    if (isInterface) {
067                            ctClass = superClass == null ?
068                                            pool.makeInterface(name) :
069                                            pool.makeInterface(name, toCtClass(superClass));
070                    } else {
071                            ctClass = superClass == null || superClass.isInterface() ?
072                                            pool.makeClass(name) :
073                                            pool.makeClass(name, toCtClass(superClass));
074                            if (superClass != null && superClass.isInterface()) {
075                                    addInterface(superClass);
076                            }
077                    }
078            }
079    
080            /**
081             * Builds new class instance` represented by this class builder.
082             *
083             * @return new class instance
084             */
085            @SuppressWarnings("unchecked")
086            public Class<T> build() {
087                    return toClass(ctClass);
088            }
089    
090            /**
091             * Adds implementation of specified interface to class represented by this class builder.
092             *
093             * @param interfaceClass interface class
094             * @return this class builder instance
095             */
096            public ClassBuilder<T> addInterface(final Class interfaceClass) {
097                    final CtClass ctInterfaceClass = toCtClass(interfaceClass);
098                    ctClass.addInterface(ctInterfaceClass);
099                    return this;
100            }
101    
102            /**
103             * Adds new field declaration with specified name modifiers and type to class represented by this class builder.
104             *
105             * @param name    field name
106             * @param modifiers field modifiers
107             * @param type    field class
108             * @return this class builder instance
109             */
110            public ClassBuilder<T> addField(final String name, final int modifiers, final Class type) {
111                    final CtClass ctType = toCtClass(type);
112                    try {
113                            final CtField field = new CtField(ctType, name, ctClass);
114                            field.setModifiers(modifiers);
115                            ctClass.addField(field);
116                    } catch (CannotCompileException ex) {
117                            throw new RuntimeException(String.format("Can't add field '%s' to class '%s'", name, ctClass), ex);
118                    }
119                    return this;
120            }
121    
122            /**
123             * Adds new constructor declaration with specified parameter types, exception declarations and constructor body.
124             *
125             * @param parameterTypes constructor parameter types
126             * @param exceptionTypes constructor exception types
127             * @param body             constructor body
128             * @return this class builder instance
129             */
130            public ClassBuilder<T> addConstructor(final Class[] parameterTypes, final Class[] exceptionTypes,
131                                                                                      final String body) {
132                    final CtClass[] ctParameterTypes = toCtClasses(parameterTypes);
133                    final CtClass[] ctExceptionTypes = toCtClasses(exceptionTypes);
134    
135                    try {
136                            final CtConstructor constructor = new CtConstructor(ctParameterTypes, ctClass);
137                            constructor.setExceptionTypes(ctExceptionTypes);
138                            constructor.setBody(body);
139                            ctClass.addConstructor(constructor);
140                    } catch (Exception ex) {
141                            throw new RuntimeException(String.format("Can't add constructor to class '%s'", ctClass), ex);
142                    }
143                    return this;
144            }
145    
146            /**
147             * Adds new method declaration with specified name, modifiers, signature and method body.
148             *
149             * @param name             method name
150             * @param modifiers       method modifiers
151             * @param returnType     method return type
152             * @param parameterTypes method parameter types
153             * @param exceptionTypes method exception types
154             * @param body             method body
155             * @return this class builder instance
156             */
157            public ClassBuilder<T> addMethod(final String name, final int modifiers, final Class returnType,
158                                                                             final Class[] parameterTypes, final Class[] exceptionTypes, final String body) {
159                    final CtClass ctReturnType = toCtClass(returnType);
160                    final CtClass[] ctParameterTypes = toCtClasses(parameterTypes);
161                    final CtClass[] ctExceptionTypes = toCtClasses(exceptionTypes);
162    
163                    try {
164                            final CtMethod method = new CtMethod(ctReturnType, name, ctParameterTypes, ctClass);
165                            method.setModifiers(modifiers);
166                            method.setBody(body);
167                            method.setExceptionTypes(ctExceptionTypes);
168                            ctClass.addMethod(method);
169                    } catch (Exception ex) {
170                            throw new RuntimeException(String.format("Can't add method '%s' to class '%s'", name, ctClass), ex);
171                    }
172                    return this;
173            }
174    
175            /**
176             * Converts original java classes to java assist class representations.
177             *
178             * @param inputClasses original java classes
179             * @return java assist class representations
180             */
181            private CtClass[] toCtClasses(final Class[] inputClasses) {
182                    if (inputClasses == null || inputClasses.length == 0) {
183                            return null;
184                    }
185    
186                    final CtClass[] result = new CtClass[inputClasses.length];
187                    for (int i = 0; i < inputClasses.length; i++) {
188                            result[i] = toCtClass(inputClasses[i]);
189                    }
190                    return result;
191            }
192    
193            /**
194             * Converts original java class to java assist class representation.
195             *
196             * @param inputClass original java class
197             * @return java assist class representation
198             */
199            private CtClass toCtClass(final Class inputClass) {
200                    try {
201                            pool.importClass(inputClass);
202                            return pool.getCtClass(toClassName(inputClass));
203                    } catch (NotFoundException ex) {
204                            throw new RuntimeException(String.format("Can't convert class '%s'", inputClass), ex);
205                    }
206            }
207    
208            /**
209             * Converts java assist class to original java class.
210             *
211             * @param inputClass java assist class
212             * @return original java class
213             */
214            private Class toClass(final CtClass inputClass) {
215                    try {
216                            return pool.toClass(inputClass);
217                    } catch (CannotCompileException ex) {
218                            throw new RuntimeException(String.format("Can't convert javassist class '%s'", inputClass), ex);
219                    }
220            }
221    
222            /**
223             * Gets string representation of class name. It differs from original class name only for arrays.
224             *
225             * @param inputClass class
226             * @return string representation of class name
227             */
228            private String toClassName(final Class inputClass) {
229                    if (inputClass.isArray()) {
230                            return toClassName(inputClass.getComponentType()) + "[]";
231                    }
232    
233                    return inputClass.getName();
234            }
235    
236            @Override
237            public String toString() {
238                    final DescriptionBuilder builder = new DescriptionBuilder(getClass());
239                    builder.append("name", ctClass.getName());
240                    return builder.toString();
241            }
242    }
