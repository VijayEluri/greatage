005    package org.greatage.javassist;
006    
007    import javassist.ClassPool;
008    import javassist.LoaderClassPath;
009    import org.greatage.util.CollectionUtils;
010    
011    import java.util.Set;
012    
013    /**
014     * This class represent javassist class pool fix that works on web application class loaders.
015     *
016     * @author Ivan Khalopik
017     * @since 1.0
018     */
019    public class ClassPoolEx extends ClassPool {
020            private static final Set<ClassLoader> CLASS_LOADERS = CollectionUtils.newConcurrentSet();
021    
022            /**
023             * Creates default javassist class pool with fix that works on web application class loaders.
024             */
025            public ClassPoolEx() {
026                    super(true);
027            }
028    
029            /**
030             * Adds missing class loader classpath for specified class.
031             *
032             * @param clazz class
033             */
034            public void importClass(final Class clazz) {
035                    final ClassLoader loader = clazz.getClassLoader();
036                    if (loader != null && !CLASS_LOADERS.contains(loader)) {
037                            CLASS_LOADERS.add(loader);
038                            insertClassPath(new LoaderClassPath(loader));
039                    }
040            }
041    }
